# Angular Manuel Utilisateur

Fonctionnalités principales d'Angular et conseils d'utilisation.

## Développement de composants

### Créer un composant

```bash
ng generate component components/user-list
```

### Bases des composants

```typescript
import { Component } from '@angular/core';

@Component({
  selector: 'app-user-list',
  template: `
    <h2>Liste des utilisateurs</h2>
    <ul>
      <li *ngFor="let user of users">

      </li>
    </ul>

  styles: [`

    { name: 'Jean', email: 'jean@example.com' },

```

### Communication entre composants

**Parent vers enfant (Input):**

```typescript
// Composant enfant
import { Component, Input } from '@angular/core';

@Component({
  selector: 'app-user-card',
  template: `
    <div class="card">
      <h3>{{ title }}</h3>
      <p>{{ user?.name }}</p>
    </div>
  `
})
export class UserCardComponent {
  @Input() title = '';
  @Input() user?: { name: string; email: string };
}
```

```typescript
// Composant parent
@Component({
  selector: 'app-parent',
  template: `
    <app-user-card 
      title="Info utilisateur" 
      [user]="currentUser">
    </app-user-card>
  `
})
export class ParentComponent {
  currentUser = { name: 'Alice', email: 'alice@example.com' };
}
```

**Enfant vers parent (Output):**

```typescript
// Composant enfant
import { Component, Output, EventEmitter } from '@angular/core';

@Component({
  selector: 'app-child',
  template: `<button (click)="onClick()">Cliquez</button>`
})
export class ChildComponent {
  @Output() buttonClick = new EventEmitter<string>();

  onClick() {
    this.buttonClick.emit('Bonjour depuis enfant!');
  }
}
```

```typescript
// Composant parent
@Component({
  selector: 'app-parent',
  template: `
    <app-child (buttonClick)="handleClick($event)"></app-child>
  `
})
export class ParentComponent {
  handleClick(message: string) {
    console.log(message);
  }
}
```

## Services et Injection de dépendances

### Créer un service

```bash
ng generate service services/user
```

### Bases des services

```typescript
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';

export interface User {
  id: number;
  name: string;
  email: string;
}

@Injectable({
  providedIn: 'root'
})
export class UserService {
  private apiUrl = 'https://api.example.com/users';

  constructor(private http: HttpClient) {}

  getUsers(): Observable<User[]> {
    return this.http.get<User[]>(this.apiUrl);
  }

  getUser(id: number): Observable<User> {
    return this.http.get<User>(`${this.apiUrl}/${id}`);
  }

  createUser(user: Partial<User>): Observable<User> {
    return this.http.post<User>(this.apiUrl, user);
  }

  updateUser(id: number, user: Partial<User>): Observable<User> {
    return this.http.put<User>(`${this.apiUrl}/${id}`, user);
  }

  deleteUser(id: number): Observable<void> {
    return this.http.delete<void>(`${this.apiUrl}/${id}`);
  }
}
```

### Utiliser un service

```typescript
import { Component, OnInit } from '@angular/core';
import { UserService, User } from './services/user.service';

@Component({
  selector: 'app-users',
  template: `
    <div *ngFor="let user of users">
      {{ user.name }}
    </div>
  `
})
export class UsersComponent implements OnInit {
  users: User[] = [];

  constructor(private userService: UserService) {}

  ngOnInit() {
    this.userService.getUsers().subscribe(data => {
      this.users = data;
    });
  }
}
```

## Routage

### Paramètres de route

```typescript
// Configuration de route
{ path: 'user/:id', component: UserDetailComponent }

// Naviguer avec paramètres
<a [routerLink]="['/user', user.id]">Voir</a>
<button (click)="goToDetail(123)">Voir</button>

// Obtenir les paramètres dans le composant
import { ActivatedRoute } from '@angular/router';

@Component({...})
export class UserDetailComponent implements OnInit {
  constructor(private route: ActivatedRoute) {}

  ngOnInit() {
    this.route.params.subscribe(params => {
      const id = params['id'];
      console.log('ID utilisateur:', id);
    });
  }

  goToDetail(id: number) {
    this.router.navigate(['/user', id]);
  }
}
```

## Formulaires

### Formulaires réactifs

```typescript
import { Component } from '@angular/core';
import { FormBuilder, FormGroup, Validators, ReactiveFormsModule } from '@angular/forms';
import { CommonModule } from '@angular/common';

@Component({
  selector: 'app-login',
  template: `
    <form [formGroup]="form" (ngSubmit)="onSubmit()">
      <div>
        <label>Email</label>
        <input formControlName="email" type="email">
        <span *ngIf="form.get('email')?.errors?.['required']">Requis</span>
        <span *ngIf="form.get('email')?.errors?.['email']">Email invalide</span>
      </div>
      <div>
        <label>Mot de passe</label>
        <input formControlName="password" type="password">
      </div>
      <button type="submit" [disabled]="form.invalid">Connexion</button>
    </form>
  `,
  standalone: true,
  imports: [ReactiveFormsModule, CommonModule]
})
export class LoginComponent {
  form: FormGroup;

  constructor(private fb: FormBuilder) {
    this.form = this.fb.group({
      email: ['', [Validators.required, Validators.email]],
      password: ['', [Validators.required, Validators.minLength(6)]]
    });
  }

  onSubmit() {
    if (this.form.valid) {
      console.log('Valeur du formulaire:', this.form.value);
    }
  }
}
```

```html

<!-- Sortie: BONJOUR -->

<!-- Pipe JSON -->

```

### Pipe personnalisé

```bash
ng generate pipe pipes/truncate
```

```typescript
import { Pipe, PipeTransform } from '@angular/core';

@Pipe({
  name: 'truncate'
})
export class TruncatePipe implements PipeTransform {
  transform(value: string, limit: number = 10, trail: string = '...'): string {
    if (value.length <= limit) return value;
    return value.substring(0, limit) + trail;
  }
}
```

Utilisation:

```html
<p>{{ 'Ceci est un long texte' | truncate:5 }}</p>
<!-- Sortie: Ceci ... -->
```

## Hooks de cycle de vie

```typescript
import { 
  Component, 
  OnInit, 
  OnChanges, 
  OnDestroy,
  AfterViewInit,
  AfterContentInit
} from '@angular/core';

@Component({...})
export class MyComponent implements 
  OnInit, 
  OnChanges, 
  OnDestroy,
  AfterViewInit,
  AfterContentInit {
  
  // Appelé après le constructeur
  ngOnInit() {
    console.log('ngOnInit - Composant initialisé');
  }
  
  // Appelé quand les propriétés d'entrée changent
  ngOnChanges(changes: SimpleChanges) {
    console.log('ngOnChanges - Propriétés d\'entrée changées', changes);
  }
  
  // Appelé après l'initialisation de la vue
  ngAfterViewInit() {
    console.log('ngAfterViewInit - Vue initialisée');
  }
  
  // Appelé après l'initialisation du contenu projeté
  ngAfterContentInit() {
    console.log('ngAfterContentInit - Contenu projeté');
  }
  
  // Appelé avant la destruction du composant
  ngOnDestroy() {
    console.log('ngOnDestroy - Composant sur le point d\'être détruit');
  }
```

## Meilleures pratiques

### Organisation du code

```typescript
// Utiliser des Composants Standalone
import { Component, signal, computed } from '@angular/core';
import { CommonModule } from '@angular/common';

@Component({
  selector: 'app-counter',
  standalone: true,
  imports: [CommonModule],
  template: `
    <p>Compteur: {{ count() }}</p>
    <p>Double: {{ doubleCount() }}</p>
    <button (click)="increment()">+1</button>
  `
})
export class CounterComponent {
  // Utiliser les Signals
  count = signal(0);
  
  // Valeur calculée
  doubleCount = computed(() => this.count() * 2);
  
  increment() {
    this.count.update(c => c + 1);
  }
}
```

### Opérations asynchrones

```typescript
import { Component, inject } from '@angular/core';
import { AsyncPipe } from '@angular/common';
import { Observable } from 'rxjs';

@Component({
  selector: 'app-users',
  template: `
    <ul *ngIf="users$ | async as users">
      <li *ngFor="let user of users">{{ user.name }}</li>
    </ul>
  `,
  standalone: true,
  imports: [AsyncPipe]
})
export class UsersComponent {
  private userService = inject(UserService);
  users$ = this.userService.getUsers();
}
```
