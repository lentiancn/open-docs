# Vue 使用指南

本指南介紹如何在應用程式中使用 Vue.js。

## 基本概念

### Vue 實例

```javascript
import { createApp } from 'vue'

const app = createApp({
  data() {
    return {
      message: 'Hello Vue!'
    }
  }
})

app.mount('#app')
```

### 模板語法

```html

<div v-for="item in items">{{ item }}</div>
<button @click="handleClick">點擊</button>
<input v-model="inputValue">
```

## 響應式

### Vue 3 Composition API（推薦）

```javascript
import { ref, reactive, computed } from 'vue'

export default {
  setup() {
    // ref 用於原始類型
    const count = ref(0)
    
    // reactive 用於物件
    const state = reactive({
      name: 'Vue',
      version: 3
    })
    
    // 計算屬性
    const doubleCount = computed(() => count.value * 2)
    
    // 方法
    const increment = () => {
      count.value++
    }
    
    return { count, state, doubleCount, increment }
  }
}
```

### Vue 2 Options API

```javascript
export default {
  data() {
    return {
      count: 0,
      name: 'Vue',
      version: 2
    }
  },
  computed: {
    doubleCount() {
      return this.count * 2
    }
  },
  methods: {
    increment() {
      this.count++
    }
  }
}
```

## 元件

### 建立元件

```vue
<!-- MyComponent.vue -->
<template>
  <div class="my-component">
    <h2>{{ title }}</h2>
    <p>{{ content }}</p>
    <button @click="emitEvent">點擊我</button>
  </div>
</template>

<script setup>
import { defineProps, defineEmits } from 'vue'

const props = defineProps({
  title: {
    type: String,
    required: true
  },
  content: {
    type: String,
    default: '預設內容'
  }
})

const emit = defineEmits(['click'])

const emitEvent = () => {
  emit('click', 'Hello from component!')
}
</script>

<style scoped>
.my-component {
  padding: 20px;
}
</style>
```

### 使用元件

```vue
<!-- ParentComponent.vue -->
<template>
  <MyComponent 
    title="我的標題"
    content="自訂內容"
    @click="handleClick"
  />
</template>

<script setup>
import MyComponent from './MyComponent.vue'

const handleClick = (message) => {
  console.log(message)
}
</script>
```

## 元件通訊

### Props（父到子）

```vue
<!-- Child.vue -->
<script setup>
defineProps({
  message: String,
  count: Number
})
</script>

<template>
  <p>{{ message }} - {{ count }}</p>
</template>

<!-- Parent.vue -->
<template>
  <Child message="Hello" :count="5" />
</template>
```

### 事件（子到父）

```vue
<!-- Child.vue -->
<script setup>
const emit = defineEmits(['update'])

const sendData = () => {
  emit('update', { name: 'Vue' })
}
</script>

<template>
  <button @click="sendData">傳送</button>
</template>

<!-- Parent.vue -->
<template>
  <Child @update="handleUpdate" />
</template>

<script setup>
const handleUpdate = (data) => {
  console.log(data)
}
</script>
```

### Provide/Inject

```vue
<!-- Parent.vue -->
<script setup>
import { provide } from 'vue'

provide('theme', 'dark')
provide('user', { name: 'John' })
</script>

<!-- Child.vue -->
<script setup>
import { inject } from 'vue'

const theme = inject('theme')
const user = inject('user')
</script>
```

## 事件處理

```html
<!-- 點擊事件 -->
<button @click="handleClick">點擊</button>

<!-- 事件物件 -->
<button @click="handleClick($event)">點擊</button>

<!-- 事件修飾符 -->
<button @click.stop="handleClick">阻止冒泡</button>
<button @click.prevent="handleSubmit">阻止預設</button>

<!-- 鍵盤修飾符 -->
<input @keyup.enter="submit" />
<input @keyup.ctrl.enter="submit" />
```

## 表單輸入綁定

```vue
<template>
  <form @submit.prevent="submitForm">
    <!-- 文字輸入 -->
    <input v-model="form.name" type="text" />
    
    <!-- 核取方塊 -->
    <input v-model="form.checked" type="checkbox" />
    
    <!-- 單選 -->
    <input v-model="form.gender" value="male" type="radio">男
    <input v-model="form.gender" value="female" type="radio">女
    
    <!-- 選擇器 -->
    <select v-model="form.country">
      <option value="cn">中國</option>
      <option value="us">美國</option>
    </select>
    
    <button type="submit">提交</button>
  </form>
</template>

<script setup>
import { reactive } from 'vue'

const form = reactive({
  name: '',
  checked: false,
  gender: '',
  country: ''
})

const submitForm = () => {
  console.log(form)
}
</script>
```

## 計算屬性

```javascript
import { ref, computed } from 'vue'

const firstName = ref('John')
const lastName = ref('Doe')

// 計算屬性 - 快取，只有依賴變化時才重新計算
const fullName = computed(() => {
  return `${firstName.value} ${lastName.value}`
})
```

## 監聽器

```javascript
import { ref, watch } from 'vue'

const count = ref(0)

// 簡單監聽器
watch(count, (newValue, oldValue) => {
  console.log(`從 ${oldValue} 變為 ${newValue}`)
})

// 深度監聽
const state = ref({ name: 'Vue' })
watch(state, (newValue) => {
  console.log('狀態改變:', newValue)
}, { deep: true })
```

## 生命週期鉤子

### Vue 3 Composition API

| 鉤子 | 描述 |
|------|------|
| `setup()` | 在其他鉤子之前呼叫 |
| `onMounted()` | 元件掛載後 |
| `onUpdated()` | 元件更新後 |
| `onUnmounted()` | 元件卸載後 |

```javascript
import { onMounted, onUpdated, onUnmounted } from 'vue'

export default {
  setup() {
    onMounted(() => {
      console.log('已掛載!')
    })
    onUpdated(() => {
      console.log('已更新!')
    })
    onUnmounted(() => {
      console.log('已卸載!')
    })
  }
}
```

### Vue 2 Options API

| 鉤子 | 描述 |
|------|------|
| `beforeCreate` | 實例建立前 |
| `created` | 實例建立後 |
| `beforeMount` | 掛載前 |
| `mounted` | 掛載後 |
| `beforeUpdate` | 更新前 |
| `updated` | 更新後 |
| `beforeDestroy` | 銷毀前 |
| `destroyed` | 銷毀後 |

## Vue Router

### 基本設定

```javascript
// router/index.js
import { createRouter, createWebHistory } from 'vue-router'
import Home from '../views/Home.vue'
import About from '../views/About.vue'

const routes = [
  { path: '/', component: Home },
  { path: '/about', component: About }
]

const router = createRouter({
  history: createWebHistory(),
  routes
})

export default router
```

### 導航

```vue
<template>
  <!-- 程式化導航 -->
  <button @click="$router.push('/about')">關於</button>
  
  <!-- Router Link -->
  <router-link to="/">首頁</router-link>
  <router-link to="/about">關於</router-link>
</template>
```

## Pinia 狀態管理

### 定義 Store

```javascript
// stores/user.js
import { defineStore } from 'pinia'

export const useUserStore = defineStore('user', {
  state: () => ({
    name: 'John',
    email: 'john@example.com'
  }),
  
  getters: {
    displayName: (state) => `${state.name} (${state.email})`
  },
  
  actions: {
    updateName(newName) {
      this.name = newName
    }
  }
})
```

### 使用 Store

```vue
<script setup>
import { useUserStore } from '../stores/user'

const userStore = useUserStore()

// 存取 state
console.log(userStore.name)

// 呼叫 actions
userStore.updateName('Jane')
</script>
```
