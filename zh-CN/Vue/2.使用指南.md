# Vue 使用指南

本指南介绍如何在应用中使用 Vue.js。

## 基本概念

### Vue 实例

```javascript
import { createApp } from 'vue'

const app = createApp({
  data() {
    return {
      message: 'Hello Vue!'
    }
  }
})

app.mount('#app')
```

### 模板语法

```html

<div v-for="item in items">{{ item }}</div>

<input v-model="inputValue">
```

## 响应式

### Vue 3 Composition API（推荐）

```javascript
import { ref, reactive, computed } from 'vue'

export default {
  setup() {
    // ref 用于原始类型
    const count = ref(0)
    
    // reactive 用于对象
    const state = reactive({
      name: 'Vue',
      version: 3
    })
    
    // 计算属性
    const doubleCount = computed(() => count.value * 2)
    
    // 方法
    const increment = () => {
      count.value++
    }
    
    return { count, state, doubleCount, increment }
  }
}
```

### Vue 2 Options API

```javascript
export default {
  data() {
    return {
      count: 0,
      name: 'Vue',
      version: 2
    }
  },
  computed: {
    doubleCount() {
      return this.count * 2
    }
  },
  methods: {
    increment() {
      this.count++
    }
  }
}
```

## 组件

### 创建组件

```vue
<!-- MyComponent.vue -->
<template>
  <div class="my-component">
    <h2>{{ title }}</h2>
    <p>{{ content }}</p>
    <button @click="emitEvent">点击我</button>
  </div>
</template>

<script setup>
import { defineProps, defineEmits } from 'vue'

const props = defineProps({
  title: {
    type: String,
    required: true
  },
  content: {
    type: String,
    default: '默认内容'
  }
})

const emit = defineEmits(['click'])

const emitEvent = () => {
  emit('click', 'Hello from component!')
}
</script>

<style scoped>
.my-component {
  padding: 20px;
}
</style>
```

### 使用组件

```vue
<!-- ParentComponent.vue -->
<template>
  <MyComponent 
    title="我的标题"
    content="自定义内容"
    @click="handleClick"
  />
</template>

<script setup>
import MyComponent from './MyComponent.vue'

const handleClick = (message) => {
  console.log(message)
}
</script>
```

## 组件通信

### Props（父到子）

```vue
<!-- Child.vue -->
<script setup>
defineProps({
  message: String,
  count: Number
})
</script>

<template>
  <p>{{ message }} - {{ count }}</p>
</template>

<!-- Parent.vue -->
<template>
  <Child message="Hello" :count="5" />
</template>
```

### 事件（子到父）

```vue
<!-- Child.vue -->
<script setup>
const emit = defineEmits(['update'])

const sendData = () => {
  emit('update', { name: 'Vue' })
}
</script>

<template>
  <button @click="sendData">发送</button>
</template>

<!-- Parent.vue -->
<template>
  <Child @update="handleUpdate" />
</template>

<script setup>
const handleUpdate = (data) => {
  console.log(data)
}
</script>
```

### Provide/Inject

```vue
<!-- Parent.vue -->
<script setup>
import { provide } from 'vue'

provide('theme', 'dark')
provide('user', { name: 'John' })
</script>

<!-- Child.vue -->
<script setup>
import { inject } from 'vue'

const theme = inject('theme')
const user = inject('user')
</script>
```

## 事件处理

```html
<!-- 点击事件 -->
<button @click="handleClick">点击</button>

<!-- 事件对象 -->
<button @click="handleClick($event)">点击</button>

<!-- 事件修饰符 -->
<button @click.stop="handleClick">阻止冒泡</button>
<button @click.prevent="handleSubmit">阻止默认</button>

<!-- 键盘修饰符 -->
<input @keyup.enter="submit" />
<input @keyup.ctrl.enter="submit" />
```

## 表单输入绑定

```vue
<template>
  <form @submit.prevent="submitForm">
    <!-- 文本输入 -->
    <input v-model="form.name" type="text" />
    
    <!-- 复选框 -->
    <input v-model="form.checked" type="checkbox" />
    
    <!-- 单选 -->
    <input v-model="form.gender" value="male" type="radio">男
    <input v-model="form.gender" value="female" type="radio">女
    
    <!-- 选择器 -->
    <select v-model="form.country">
      <option value="cn">中国</option>
      <option value="us">美国</option>
    </select>
    
    <button type="submit">提交</button>
  </form>
</template>

<script setup>
import { reactive } from 'vue'

const form = reactive({
  name: '',
  checked: false,
  gender: '',
  country: ''
})

const submitForm = () => {
  console.log(form)
}
</script>
```

## 计算属性

```javascript
import { ref, computed } from 'vue'

const firstName = ref('John')
const lastName = ref('Doe')

// 计算属性 - 缓存，只有依赖变化时才重新计算
const fullName = computed(() => {
  return `${firstName.value} ${lastName.value}`
})
```

## 监听器

```javascript
import { ref, watch } from 'vue'

const count = ref(0)

// 简单监听器
watch(count, (newValue, oldValue) => {
  console.log(`从 ${oldValue} 变为 ${newValue}`)
})

// 深度监听
const state = ref({ name: 'Vue' })
watch(state, (newValue) => {
  console.log('状态改变:', newValue)
}, { deep: true })
```

## 生命周期钩子

### Vue 3 Composition API

| 钩子 | 描述 |
|------|------|
| `setup()` | 在其他钩子之前调用 |
| `onMounted()` | 组件挂载后 |
| `onUpdated()` | 组件更新后 |
| `onUnmounted()` | 组件卸载后 |

```javascript
import { onMounted, onUpdated, onUnmounted } from 'vue'

export default {
  setup() {
    onMounted(() => {
      console.log('已挂载!')
    })
    onUpdated(() => {
      console.log('已更新!')
    })
    onUnmounted(() => {
      console.log('已卸载!')
    })
  }
}
```

### Vue 2 Options API

| 钩子 | 描述 |
|------|------|
| `beforeCreate` | 实例创建前 |
| `created` | 实例创建后 |
| `beforeMount` | 挂载前 |
| `mounted` | 挂载后 |
| `beforeUpdate` | 更新前 |
| `updated` | 更新后 |
| `beforeDestroy` | 销毁前 |
| `destroyed` | 销毁后 |

## Vue Router

### 基础设置

```javascript
// router/index.js
import { createRouter, createWebHistory } from 'vue-router'
import Home from '../views/Home.vue'
import About from '../views/About.vue'

const routes = [
  { path: '/', component: Home },
  { path: '/about', component: About }
]

const router = createRouter({
  history: createWebHistory(),
  routes
})

export default router
```

### 导航

```vue
<template>
  <!-- 编程式导航 -->
  <button @click="$router.push('/about')">关于</button>
  
  <!-- Router Link -->
  <router-link to="/">首页</router-link>
  <router-link to="/about">关于</router-link>
</template>
```

## Pinia 状态管理

### 定义 Store

```javascript
// stores/user.js
import { defineStore } from 'pinia'

export const useUserStore = defineStore('user', {
  state: () => ({
    name: 'John',
    email: 'john@example.com'
  }),
  
  getters: {
    displayName: (state) => `${state.name} (${state.email})`
  },
  
  actions: {
    updateName(newName) {
      this.name = newName
    }
  }
})
```

### 使用 Store

```vue
<script setup>
import { useUserStore } from '../stores/user'

const userStore = useUserStore()

// 访问 state
console.log(userStore.name)

// 调用 actions
userStore.updateName('Jane')
</script>
```

## 最佳实践

### 1. 使用 Composition API

```javascript

export default {
  setup() {
    const count = ref(0)
    return { count }
  }
}
```

### 2. 使用 Script Setup

```vue
<script setup>
// 更简洁的语法
</script>
```

### 3. 使用计算属性处理派生状态

```javascript
const firstName = ref('John')
const lastName = ref('Doe')

// 好 - 计算属性
```
