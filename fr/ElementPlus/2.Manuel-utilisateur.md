## 1. Bases des composants

### 1.1 Bouton de base

Les boutons sont l'un des composants les plus utilisés. Element Plus fournit plusieurs types de boutons.

```vue
<template>
  <!-- Bouton primaire -->
  <el-button type="primary">Bouton primaire</el-button>
  
  <!-- Bouton succès -->
  <el-button type="success">Bouton succès</el-button>
  
  <!-- Bouton info -->
  <el-button type="info">Bouton info</el-button>
  
  <!-- Bouton avertissement -->
  <el-button type="warning">Bouton avertissement</el-button>
  
  <!-- Bouton danger -->
  <el-button type="danger">Bouton danger</el-button>
  
  <!-- Bouton texte -->
  <el-button type="text">Bouton texte</el-button>
  
  <!-- Bouton simple -->
  <el-button plain>Bouton simple</el-button>
  
  <!-- Bouton arrondi -->
  <el-button round>Bouton arrondi</el-button>
  
  <!-- Bouton avec icône -->
  <el-button icon="el-icon-search">Rechercher</el-button>
</template>
```

### 1.2 Attributs du bouton

| Attribut | Description | Type | Valeurs optionnelles | Par défaut |
|------|------|------|-----------|----------|
| type | Type de bouton | string | primary / success / warning / danger / info / text | — |
| plain | Bouton simple | boolean | — | false |
| round | Bouton arrondi | boolean | — | false |
| circle | Bouton circulaire | boolean | — | false |
| loading | En cours de chargement | boolean | — | false |
| disabled | Désactivé | boolean | — | false |
| icon | Nom de classe d'icône | string | — | — |
| size | Taille | string | large / default / small | — |

### 1.3 Événements du bouton

```vue
<template>
  <el-button @click="handleClick">Événement clic</el-button>
</template>

<script setup>
const handleClick = () => {
  console.log('Bouton cliqué')
}
</script>
```

## 2. Composants de formulaire

### 2.1 Formulaire de base

```vue
<template>
  <el-form :model="form" :rules="rules" ref="formRef" label-width="100px">
    <el-form-item label="Nom d'utilisateur" prop="username">
      <el-input v-model="form.username" placeholder="Entrez le nom d'utilisateur" />
    </el-form-item>
    
    <el-form-item label="Mot de passe" prop="password">
      <el-input v-model="form.password" type="password" placeholder="Entrez le mot de passe" show-password />
    </el-form-item>
    
    <el-form-item label="Email" prop="email">
      <el-input v-model="form.email" placeholder="Entrez l'email" />
    </el-form-item>
    
    <el-form-item label="Genre" prop="gender">
      <el-radio-group v-model="form.gender">
        <el-radio label="male">Masculin</el-radio>
        <el-radio label="female">Féminin</el-radio>
      </el-radio-group>
    </el-form-item>
    
    <el-form-item label="Loisirs" prop="hobby">
      <el-checkbox-group v-model="form.hobby">
        <el-checkbox label="Basketball" />
        <el-checkbox label="Football" />
        <el-checkbox label="Natation" />
      </el-checkbox-group>
    </el-form-item>
    
    <el-form-item label="Bio" prop="bio">
      <el-input v-model="form.bio" type="textarea" :rows="3" placeholder="Entrez votre bio" />
    </el-form-item>
    
    <el-form-item>
      <el-button type="primary" @click="submitForm">Soumettre</el-button>
      <el-button @click="resetForm">Réinitialiser</el-button>
    </el-form-item>
  </el-form>
</template>

<script setup>
import { reactive, ref } from 'vue'

const formRef = ref(null)

const form = reactive({
  username: '',
  password: '',
  email: '',
  gender: '',
  hobby: [],
  bio: ''
})

const rules = {
  username: [
    { required: true, message: 'Veuillez entrer le nom utilisateur', trigger: 'blur' },
    { min: 3, max: 20, message: 'Le nom doit contenir 3-20 caractères', trigger: 'blur' }
  ],
  password: [
    { required: true, message: 'Veuillez entrer le mot de passe', trigger: 'blur' },
    { min: 6, message: 'Le mot de passe doit contenir au moins 6 caractères', trigger: 'blur' }
  ],
  email: [
    { required: true, message: 'Veuillez entrer email', trigger: 'blur' },
    { type: 'email', message: 'Veuillez entrer un email valide', trigger: 'blur' }
  ],
  gender: [
    { required: true, message: 'Veuillez sélectionner le genre', trigger: 'change' }
  ]
}

const submitForm = async () => {
  if (!formRef.value) return
  
  await formRef.value.validate((valid) => {
    if (valid) {
      console.log('Données du formulaire:', form)
      ElMessage.success('Soumis avec succès')
    } else {
      ElMessage.error('Veuillez vérifier le formulaire')
    }
  })
}

const resetForm = () => {
  formRef.value?.resetFields()
}
</script>
```

```vue
<template>

        <el-button type="danger" size="small" @click="handleDelete(row)">Supprimer</el-button>

import { ref } from 'vue'

const tableData = ref([

  { date: '2024-01-02', name: 'Marie Martin', address: 'Lyon' },
  { date: '2024-01-03', name: 'Pierre Bernard', address: 'Marseille' }
])

const handleEdit = (row) => {
  console.log('Modifier:', row)
}

const handleDelete = (row) => {
  console.log('Supprimer:', row)
}
</script>
```

### 3.2 Table avec pagination

```vue
<template>
  <div>
    <el-table :data="tableData" v-loading="loading">
      <el-table-column prop="id" label="ID" width="80" />
      <el-table-column prop="title" label="Titre" />
      <el-table-column prop="author" label="Auteur" width="120" />
      <el-table-column prop="date" label="Date de publication" width="180" />
    </el-table>
    
    <el-pagination
      v-model:current-page="currentPage"
      v-model:page-size="pageSize"
      :total="total"
      :page-sizes="[10, 20, 50, 100]"
      layout="total, sizes, prev, pager, next, jumper"
      @size-change="handleSizeChange"
      @current-change="handleCurrentChange"
    />
  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue'

const tableData = ref([])
const loading = ref(false)
const currentPage = ref(1)
const pageSize = ref(10)
const total = ref(0)

const loadData = async () => {
  loading.value = true
  // Simuler appel API
  setTimeout(() => {
    tableData.value = [
      { id: 1, title: 'Titre article 1', author: 'Auteur 1', date: '2024-01-01' },
      { id: 2, title: 'Titre article 2', author: 'Auteur 2', date: '2024-01-02' }
    ]
    total.value = 100
    loading.value = false
  }, 500)
}

const handleSizeChange = (val) => {
  pageSize.value = val
  loadData()
}

const handleCurrentChange = (val) => {
  currentPage.value = val
  loadData()
}

onMounted(() => {
  loadData()
})
</script>
```

## 4. Dialogue et tiroir

### 4.1 Dialogue

```vue
<template>
  <el-button type="primary" @click="dialogVisible = true">Ouvrir dialogue</el-button>
  
  <el-dialog
    v-model="dialogVisible"
    title="Modifier informations"
    width="500px"
    :close-on-click-modal="false"
    @close="handleClose"
  >
    <el-form :model="form" label-width="80px">
      <el-form-item label="Nom">
        <el-input v-model="form.name" />
      </el-form-item>
      <el-form-item label="Description">
        <el-input v-model="form.desc" type="textarea" :rows="3" />
      </el-form-item>
    </el-form>
    
    <template #footer>
      <el-button @click="dialogVisible = false">Annuler</el-button>
      <el-button type="primary" @click="handleConfirm">Confirmer</el-button>
    </template>
  </el-dialog>
</template>

<script setup>
import { ref, reactive } from 'vue'

const dialogVisible = ref(false)

const form = reactive({
  name: '',
  desc: ''
})

const handleClose = () => {
  form.name = ''
  form.desc = ''
}

const handleConfirm = () => {
  console.log('Soumettre:', form)
  dialogVisible.value = false
  ElMessage.success('Succès')
}
</script>
```

## 5. Notifications

### 5.1 Messages globaux

```vue
<script setup>
import { ElMessage } from 'element-plus'

// Message succès
const showSuccess = () => {
  ElMessage.success('Opération réussie')
}

// Message avertissement
const showWarning = () => {
  ElMessage.warning('Avertissement')
}

// Message erreur
const showError = () => {
  ElMessage.error('Opération échouée')
}

// Message info
const showInfo = () => {
  ElMessage.info('Message info')
}
</script>
```

### 5.2 Notifications

```vue
<script setup>
import { ElNotification } from 'element-plus'

// Notification succès
const notifySuccess = () => {
  ElNotification({
    title: 'Succès',
    message: 'Opération terminée avec succès',
    type: 'success',
  })
}

// Notification erreur
const notifyError = () => {
  ElNotification({
    title: 'Erreur',
    message: "L'opération a échoué",
    type: 'error',
  })
}
</script>
```
