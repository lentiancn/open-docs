## Komponentenentwicklung

### Komponente erstellen

```bash
ng generate component components/user-list
```

### Komponenten-Grundlagen

```typescript
import { Component } from '@angular/core';

@Component({
  selector: 'app-user-list',
  template: `
    <h2>Benutzerliste</h2>
    <ul>
      <li *ngFor="let user of users">
        {{ user.name }} - {{ user.email }}
      </li>
    </ul>
  `,
  styles: [`
    ul { list-style: none; padding: 0; }

  users = [
    { name: 'Max', email: 'max@example.com' },
    { name: 'Anna', email: 'anna@example.com' }
  ];
}
```

### Komponentenkommunikation

**Eltern zu Kind (Input):**

```typescript
// Kind-Komponente
import { Component, Input } from '@angular/core';

@Component({
  selector: 'app-user-card',
  template: `
    <div class="card">
      <h3>{{ title }}</h3>
      <p>{{ user?.name }}</p>
    </div>
  `
})
export class UserCardComponent {
  @Input() title = '';
  @Input() user?: { name: string; email: string };
}
```

```typescript
// Eltern-Komponente
@Component({
  selector: 'app-parent',
  template: `
    <app-user-card 
      title="Benutzerinfo" 
      [user]="currentUser">
    </app-user-card>
  `
})
export class ParentComponent {
  currentUser = { name: 'Alice', email: 'alice@example.com' };
}
```

**Kind zu Eltern (Output):**

```typescript
// Kind-Komponente
import { Component, Output, EventEmitter } from '@angular/core';

@Component({
  selector: 'app-child',
  template: `<button (click)="onClick()">Klicken</button>`
})
export class ChildComponent {
  @Output() buttonClick = new EventEmitter<string>();

  onClick() {
    this.buttonClick.emit('Hallo vom Kind!');
  }
}
```

```typescript
// Eltern-Komponente
@Component({
  selector: 'app-parent',
  template: `
    <app-child (buttonClick)="handleClick($event)"></app-child>
  `
})
export class ParentComponent {
  handleClick(message: string) {
    console.log(message);
  }
}
```

## Services und Dependency Injection

### Service erstellen

```bash
ng generate service services/user
```

### Service-Grundlagen

```typescript
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';

export interface User {
  id: number;
  name: string;
  email: string;
}

@Injectable({
  providedIn: 'root'
})
export class UserService {
  private apiUrl = 'https://api.example.com/users';

  constructor(private http: HttpClient) {}

  getUsers(): Observable<User[]> {
    return this.http.get<User[]>(this.apiUrl);
  }

  getUser(id: number): Observable<User> {
    return this.http.get<User>(`}/${id}`);
  }

  createUser(user: Partial<User${this.apiUrl>): Observable<User> {
    return this.http.post<User>(this.apiUrl, user);
  }

  updateUser(id: number, user: Partial<User>): Observable<User> {
    return this.http.put<User>(`${this.apiUrl}/${id}`, user);
  }

  deleteUser(id: number): Observable<void> {
    return this.http.delete<void>(`${this.apiUrl}/${id}`);
  }
}
```

### Service verwenden

```typescript
import { Component, OnInit } from '@angular/core';
import { UserService, User } from './services/user.service';

@Component({
  selector: 'app-users',
  template: `
    <div *ngFor="let user of users">
      {{ user.name }}
    </div>
  `
})
export class UsersComponent implements OnInit {
  users: User[] = [];

  constructor(private userService: UserService) {}

  ngOnInit() {
    this.userService.getUsers().subscribe(data => {
      this.users = data;
    });
  }
}
```

### Routen-Parameter

```typescript
// Routen-Konfiguration

// Mit Parametern navigieren
<a [routerLink]="['/user', user.id]">Ansehen</a>
<button (click)="goToDetail(123)">Ansehen</button>

// Parameter in Komponente abrufen
import { ActivatedRoute } from '@angular/router';

@Component({...})
export class UserDetailComponent implements OnInit {

  ngOnInit() {
    this.route.params.subscribe(params => {

      console.log('Benutzer-ID:', id);
    });
  }

  goToDetail(id: number) {
    this.router.navigate(['/user', id]);
  }
}
```

## Formulare

### Reactive Forms

```typescript
import { Component } from '@angular/core';
import { FormBuilder, FormGroup, Validators, ReactiveFormsModule } from '@angular/forms';
import { CommonModule } from '@angular/common';

@Component({
  selector: 'app-login',
  template: `
    <form [formGroup]="form" (ngSubmit)="onSubmit()">
      <div>
        <label>E-Mail</label>
        <input formControlName="email" type="email">
        <span *ngIf="form.get('email')?.errors?.['required']">Erforderlich</span>
        <span *ngIf="form.get('email')?.errors?.['email']">Ungültige E-Mail</span>
      </div>
      <div>
        <label>Passwort</label>
        <input formControlName="password" type="password">
      </div>
      <button type="submit" [disabled]="form.invalid">Anmelden</button>
    </form>
  `,
  standalone: true,
  imports: [ReactiveFormsModule, CommonModule]
})
export class LoginComponent {
  form: FormGroup;

  constructor(private fb: FormBuilder) {
    this.form = this.fb.group({
      email: ['', [Validators.required, Validators.email]],
      password: ['', [Validators.required, Validators.minLength(6)]]
    });
  }

  onSubmit() {
    if (this.form.valid) {
      console.log('Formularwert:', this.form.value);
    }
  }
}
```

### Template-Driven Forms

  }
}

## Pipes

### Eingebaute Pipes

```html
<!-- Datums-Pipe -->
<p>{{ today | date:'dd.MM.yyyy' }}</p>
<!-- Ausgabe: 01.01.2024 -->

<!-- Währungs-Pipe -->

<!-- Ausgabe: 100,00 € -->

<!-- Zahlen-Pipe -->

<!-- Ausgabe: 3,14 -->

<!-- Großbuchstaben-Pipe -->
<p>{{ 'Hallo' | uppercase }}</p>
<!-- Ausgabe: HALLO -->

<!-- JSON-Pipe -->
<p>{{ obj | json }}</p>
```

### Eigene Pipe

```bash
ng generate pipe pipes/truncate
```

```typescript
import { Pipe, PipeTransform } from '@angular/core';

@Pipe({
  name: 'truncate'
})
export class TruncatePipe implements PipeTransform {
  transform(value: string, limit: number = 10, trail: string = '...'): string {
    if (value.length <= limit) return value;
    return value.substring(0, limit) + trail;
  }
}
```

Verwendung:

```html
<p>{{ 'Dies ist ein langer Text' | truncate:5 }}</p>
<!-- Ausgabe: Dies i... -->
```

## Lebenszyklus-Hooks

```typescript
import { 
  Component, 
  OnInit, 
  OnChanges, 
  OnDestroy,
  AfterViewInit,
  AfterContentInit
} from '@angular/core';

@Component({...})
export class MyComponent implements 
  OnInit, 
  OnChanges, 
  OnDestroy,
  AfterViewInit,
  AfterContentInit {
  
  // Wird nach dem Konstruktor aufgerufen
  ngOnInit() {
    console.log('ngOnInit - Komponente initialisiert');
  }
  
  // Wird aufgerufen, wenn sich Eingabeeigenschaften ändern
  ngOnChanges(changes: SimpleChanges) {
    console.log('ngOnChanges - Eingabeeigenschaften geändert', changes);
  }
  
  // Wird nach der Ansichts-Initialisierung aufgerufen
  ngAfterViewInit() {
    console.log('ngAfterViewInit - Ansicht initialisiert');
  }
  
  // Wird nach der Content-Projection-Initialisierung aufgerufen
  ngAfterContentInit() {
    console.log('ngAfterContentInit - Content projiziert');
  }
  
  // Wird vor der Komponenten-Zerstörung aufgerufen
  ngOnDestroy() {
    console.log('ngOnDestroy - Komponente wird zerstört');
  }
}
```

## Best Practices

### Code-Organisation

```typescript
// Standalone-Komponenten verwenden
import { Component, signal, computed } from '@angular/core';
import { CommonModule } from '@angular/common';

@Component({
  selector: 'app-counter',
  standalone: true,
  imports: [CommonModule],
  template: `
    <p>Zähler: {{ count() }}</p>
    <p>Doppel: {{ doubleCount() }}</p>
    <button (click)="increment()">+1</button>
  `
})
export class CounterComponent {
  // Signale verwenden
  count = signal(0);
  
  // Berechneter Wert
  doubleCount = computed(() => this.count() * 2);
  
  increment() {
    this.count.update(c => c + 1);
  }
}
```

### Asynchrone Operationen

```typescript
import { Component, inject } from '@angular/core';
import { AsyncPipe } from '@angular/common';
import { Observable } from 'rxjs';

@Component({
  selector: 'app-users',
  template: `
    <ul *ngIf="users$ | async as users">
      <li *ngFor="let user of users">{{ user.name }}</li>
    </ul>
  `,
  standalone: true,
  imports: [AsyncPipe]
})
export class UsersComponent {
  private userService = inject(UserService);
  users$ = this.userService.getUsers();
}
```
