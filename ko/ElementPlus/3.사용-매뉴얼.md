## 일반적인 컴포넌트

### 버튼

<el-button>버튼</el-button>
<el-button type="primary">주요</el-button>
<el-button size="small">작게</el-button>
<el-button disabled>비활성화</el-button>
<el-button icon="Search">검색</el-button>

### 입력

<el-input v-model="input" placeholder="입력하세요" />
<el-input v-model="input" prefix-icon="Search" />
<el-input v-model="input" clearable />

### 선택

<el-select v-model="value" placeholder="선택하세요">
  <el-option label="옵션 1" value="1" />
</el-select>

### 테이블

<el-table :data="tableData">
  <el-table-column prop="name" label="이름" />
</el-table>

### 폼

<el-form :model="form">
  <el-form-item label="이름">
    <el-input v-model="form.name" />
  </el-form-item>
</el-form>
