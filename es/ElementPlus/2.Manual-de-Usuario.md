## 1. Fundamentos de Componentes

### 1.1 Botón Básico

Los botones son uno de los componentes más utilizados. Element Plus proporciona múltiples tipos de botones.

```vue
<template>
  <!-- Botón Primario -->
  <el-button type="primary">Botón Primario</el-button>
  
  <!-- Botón de Éxito -->
  <el-button type="success">Botón de Éxito</el-button>
  
  <!-- Botón de Información -->
  <el-button type="info">Botón de Información</el-button>
  
  <!-- Botón de Advertencia -->
  <el-button type="warning">Botón de Advertencia</el-button>
  
  <!-- Botón de Peligro -->
  <el-button type="danger">Botón de Peligro</el-button>
  
  <!-- Botón de Texto -->
  <el-button type="text">Botón de Texto</el-button>
  
  <!-- Botón Simple -->
  <el-button plain>Botón Simple</el-button>
  
  <!-- Botón Redondeado -->
  <el-button round>Botón Redondeado</el-button>
  
  <!-- Botón con Icono -->
  <el-button icon="el-icon-search">Buscar</el-button>
</template>
```

### 1.2 Atributos del Botón

| Atributo | Descripción | Tipo | Valores opcionales | Valor por defecto |
|------|------|------|-----------|----------|
| type | Tipo de botón | string | primary / success / warning / danger / info / text | — |
| plain | Si es botón simple | boolean | — | false |
| round | Si es botón redondeado | boolean | — | false |
| circle | Si es botón circular | boolean | — | false |
| loading | Si está cargando | boolean | — | false |
| disabled | Si está deshabilitado | boolean | — | false |
| icon | Nombre de clase del icono | string | — | — |
| size | Tamaño | string | large / default / small | — |

### 1.3 Eventos del Botón

```vue
<template>
  <el-button @click="handleClick">Evento de Clic</el-button>
</template>

<script setup>
const handleClick = () => {
  console.log('Botón presionado')
}
</script>
```

## 2. Componentes de Formulario

### 2.1 Formulario Básico

```vue
<template>
  <el-form :model="form" :rules="rules" ref="formRef" label-width="100px">
    <el-form-item label="Usuario" prop="username">
      <el-input v-model="form.username" placeholder="Ingresa usuario" />
    </el-form-item>
    
    <el-form-item label="Contraseña" prop="password">
      <el-input v-model="form.password" type="password" placeholder="Ingresa contraseña" show-password />
    </el-form-item>
    
    <el-form-item label="Email" prop="email">
      <el-input v-model="form.email" placeholder="Ingresa email" />
    </el-form-item>
    
    <el-form-item label="Género" prop="gender">
      <el-radio-group v-model="form.gender">
        <el-radio label="male">Masculino</el-radio>
        <el-radio label="female">Femenino</el-radio>
      </el-radio-group>
    </el-form-item>
    
    <el-form-item label="Pasatiempos" prop="hobby">
      <el-checkbox-group v-model="form.hobby">
        <el-checkbox label="Baloncesto" />
        <el-checkbox label="Fútbol" />
        <el-checkbox label="Natación" />
      </el-checkbox-group>
    </el-form-item>
    
    <el-form-item label="Biografía" prop="bio">
      <el-input v-model="form.bio" type="textarea" :rows="3" placeholder="Ingresa biografía" />
    </el-form-item>
    
    <el-form-item>
      <el-button type="primary" @click="submitForm">Enviar</el-button>
      <el-button @click="resetForm">Reiniciar</el-button>
    </el-form-item>
  </el-form>
</template>

<script setup>
import { reactive, ref } from 'vue'

const formRef = ref(null)

const form = reactive({
  username: '',
  password: '',
  email: '',
  gender: '',
  hobby: [],
  bio: ''
})

const rules = {
  username: [
    { required: true, message: 'Por favor ingresa usuario', trigger: 'blur' },
    { min: 3, max: 20, message: 'Usuario entre 3 y 20 caracteres', trigger: 'blur' }
  ],
  password: [
    { required: true, message: 'Por favor ingresa contraseña', trigger: 'blur' },
    { min: 6, message: 'Contraseña mínimo 6 caracteres', trigger: 'blur' }
  ],
  email: [
    { required: true, message: 'Por favor ingresa email', trigger: 'blur' },
    { type: 'email', message: 'Por favor ingresa email válido', trigger: 'blur' }
  ],
  gender: [
    { required: true, message: 'Por favor selecciona género', trigger: 'change' }
  ]
}

const submitForm = async () => {
  if (!formRef.value) return
  
  await formRef.value.validate((valid) => {
    if (valid) {
      console.log('Datos del formulario:', form)
      ElMessage.success('Enviado exitosamente')
    } else {
      ElMessage.error('Por favor verifica el formulario')
    }
  })
}

const resetForm = () => {
  formRef.value?.resetFields()
}
</script>
```

## 3. Componente de Tabla

### 3.1 Tabla Básica

```vue
<template>
  <el-table :data="tableData" stripe border>
    <el-table-column prop="date" label="Fecha" width="180" />
    <el-table-column prop="name" label="Nombre" width="180" />
    <el-table-column prop="address" label="Dirección" />
    <el-table-column label="Acciones" width="200">
      <template #default="{ row }">
        <el-button type="primary" size="small" @click="handleEdit(row)">Editar</el-button>
        <el-button type="danger" size="small" @click="handleDelete(row)">Eliminar</el-button>
      </template>
    </el-table-column>
  </el-table>
</template>

<script setup>
import { ref } from 'vue'

const tableData = ref([
  { date: '2024-01-01', name: 'Juan Pérez', address: 'Ciudad de México' },
  { date: '2024-01-02', name: 'María García', address: 'Guadalajara' },
  { date: '2024-01-03', name: 'Carlos López', address: 'Monterrey' }
])

const handleEdit = (row) => {
  console.log('Editar:', row)
}

const handleDelete = (row) => {
  console.log('Eliminar:', row)
}
</script>
```

### 3.2 Tabla con Paginación

```vue
<template>
  <div>
    <el-table :data="tableData" v-loading="loading">
      <el-table-column prop="id" label="ID" width="80" />
      <el-table-column prop="title" label="Título" />
      <el-table-column prop="author" label="Autor" width="120" />
      <el-table-column prop="date" label="Fecha de Publicación" width="180" />
    </el-table>
    
    <el-pagination
      v-model:current-page="currentPage"
      v-model:page-size="pageSize"
      :total="total"
      :page-sizes="[10, 20, 50, 100]"
      layout="total, sizes, prev, pager, next, jumper"
      @size-change="handleSizeChange"
      @current-change="handleCurrentChange"
    />
  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue'

const tableData = ref([])
const loading = ref(false)
const currentPage = ref(1)
const pageSize = ref(10)
const total = ref(0)

const loadData = async () => {
  loading.value = true
  // Simular llamada API
  setTimeout(() => {
    tableData.value = [
      { id: 1, title: 'Título del Artículo 1', autor: 'Autor 1', date: '2024-01-01' },
      { id: 2, title: 'Título del Artículo 2', autor: 'Autor 2', date: '2024-01-02' }
    ]
    total.value = 100
    loading.value = false
  }, 500)
}

const handleSizeChange = (val) => {
  pageSize.value = val
  loadData()
}

const handleCurrentChange = (val) => {
  currentPage.value = val
  loadData()
}

onMounted(() => {
  loadData()
})
</script>
```

## 4. Diálogo y Cajón

### 4.1 Diálogo

```template
<template>
  <el-button type="primary" @click="dialogVisible = true">Abrir Diálogo</el-button>
  
  <el-dialog
    v-model="dialogVisible"
    title="Editar Información"
    width="500px"
    :close-on-click-modal="false"
    @close="handleClose"
  >
    <el-form :model="form" label-width="80px">
      <el-form-item label="Nombre">
        <el-input v-model="form.name" />
      </el-form-item>
      <el-form-item label="Descripción">
        <el-input v-model="form.desc" type="textarea" :rows="3" />
      </el-form-item>
    </el-form>
    
    <template #footer>
      <el-button @click="dialogVisible = false">Cancelar</el-button>
      <el-button type="primary" @click="handleConfirm">Confirmar</el-button>
    </template>
  </el-dialog>
</template>

<script setup>
import { ref, reactive } from 'vue'

const dialogVisible = ref(false)

const form = reactive({
  name: '',
  desc: ''
})

const handleClose = () => {
  form.name = ''
  form.desc = ''
}

const handleConfirm = () => {
  console.log('Enviar:', form)
  dialogVisible.value = false
  ElMessage.success('Operación exitosa')
}
</script>
```

## 5. Notificaciones de Mensajes

### 5.1 Mensajes Globales

```vue
<script setup>
import { ElMessage } from 'element-plus'

// Mensaje de éxito
const showSuccess = () => {
  ElMessage.success('Operación exitosa')
}

// Mensaje de advertencia
const showWarning = () => {
  ElMessage.warning('Advertencia')
}

// Mensaje de error
const showError = () => {
  ElMessage.error('Operación fallida')
}

// Mensaje de información
const showInfo = () => {
  ElMessage.info('Mensaje de información')
}
</script>
```

### 5.2 Notificaciones

```vue
<script setup>
import { ElNotification } from 'element-plus'

// Notificación de éxito
const notifySuccess = () => {
  ElNotification({
    title: 'Éxito',
    message: 'Operación completada exitosamente',
    type: 'success',
  })
}

// Notificación de error
const notifyError = () => {
  ElNotification({
    title: 'Error',
    message: 'La operación falló',
    type: 'error',
  })
}
</script>
```
