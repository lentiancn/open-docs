# Vue Usage Guide

This guide covers how to use Vue.js effectively in your applications.

## Basic Concepts

### Vue Instance

```javascript
import { createApp } from 'vue'

const app = createApp({
  data() {
    return {
      message: 'Hello Vue!'
    }
  }
})

app.mount('#app')
```

### Template Syntax

```html
<!-- Text interpolation -->
<h1>{{ message }}</h1>

<!-- Directives -->
<div v-if="show">Content</div>
<div v-for="item in items">{{ item }}</div>
<button @click="handleClick">Click</button>
<input v-model="inputValue">
```

## Reactivity

### Vue 3 Composition API (Recommended)

```javascript
import { ref, reactive, computed } from 'vue'

export default {
  setup() {
    // Ref for primitive values
    const count = ref(0)
    
    // Reactive for objects
    const state = reactive({
      name: 'Vue',
      version: 3
    })
    
    // Computed
    const doubleCount = computed(() => count.value * 2)
    
    // Methods
    const increment = () => {
      count.value++
    }
    
    return { count, state, doubleCount, increment }
  }
}
```

### Vue 2 Options API

```javascript
export default {
  data() {
    return {
      count: 0,
      name: 'Vue',
      version: 2
    }
  },
  computed: {
    doubleCount() {
      return this.count * 2
    }
  },
  methods: {
    increment() {
      this.count++
    }
  }
}
```

## Components

### Creating a Component

```vue
<!-- MyComponent.vue -->
<template>
  <div class="my-component">
    <h2>{{ title }}</h2>
    <p>{{ content }}</p>
    <button @click="emitEvent">Click Me</button>
  </div>
</template>

<script setup>
import { defineProps, defineEmits } from 'vue'

const props = defineProps({
  title: {
    type: String,
    required: true
  },
  content: {
    type: String,
    default: 'Default content'
  }
})

const emit = defineEmits(['click'])

const emitEvent = () => {
  emit('click', 'Hello from component!')
}
</script>

<style scoped>
.my-component {
  padding: 20px;
}
</style>
```

### Using Components

```vue
<!-- ParentComponent.vue -->
<template>
  <MyComponent 
    title="My Title"
    content="Custom content"
    @click="handleClick"
  />
</template>

<script setup>
import MyComponent from './MyComponent.vue'

const handleClick = (message) => {
  console.log(message)
}
</script>
```

## Component Communication

### Props (Parent to Child)

```vue
<!-- Child.vue -->
<script setup>
defineProps({
  message: String,
  count: Number
})
</script>

<template>
  <p>{{ message }} - {{ count }}</p>
</template>

<!-- Parent.vue -->
<template>
  <Child message="Hello" :count="5" />
</template>
```

### Events (Child to Parent)

```vue
<!-- Child.vue -->
<script setup>
const emit = defineEmits(['update'])

const sendData = () => {
  emit('update', { name: 'Vue' })
}
</script>

<template>
  <button @click="sendData">Send</button>
</template>

<!-- Parent.vue -->
<template>
  <Child @update="handleUpdate" />
</template>

<script setup>
const handleUpdate = (data) => {
  console.log(data)
}
</script>
```

### Provide/Inject

```vue
<!-- Parent.vue -->
<script setup>
import { provide } from 'vue'

provide('theme', 'dark')
provide('user', { name: 'John' })
</script>

<!-- Child.vue -->
<script setup>
import { inject } from 'vue'

const theme = inject('theme')
const user = inject('user')
</script>
```

### Refs (Template Refs)

```vue
<script setup>
import { ref, onMounted } from 'vue'

const inputRef = ref(null)

onMounted(() => {
  inputRef.value.focus()
})
</script>

<template>
  <input ref="inputRef" type="text" />
</template>
```

## Conditional Rendering

```html
<!-- v-if -->
<div v-if="show">Content</div>
<div v-else-if="show2">Content 2</div>
<div v-else>Default</div>

<!-- v-show -->
<div v-show="visible">Toggle visibility</div>
```

## List Rendering

```html
<!-- v-for with array -->
<ul>
  <li v-for="(item, index) in items" :key="item.id">
    {{ index }}. {{ item.name }}
  </li>
</ul>

<!-- v-for with object -->
<div v-for="(value, key) in object" :key="key">
  {{ key }}: {{ value }}
</div>

<!-- v-for with range -->
<span v-for="n in 10" :key="n">{{ n }}</span>
```

## Event Handling

```html
<!-- Click event -->
<button @click="handleClick">Click</button>

<!-- Event object -->
<button @click="handleClick($event)">Click</button>

<!-- Multiple handlers -->
<button @click="handler1(), handler2()">Click</button>

<!-- Event modifiers -->
<button @click.stop="handleClick">Stop propagation</button>
<button @click.prevent="handleSubmit">Prevent default</button>
<button @click.stop.prevent="handleClick">Both</button>

<!-- Key modifiers -->
<input @keyup.enter="submit" />
<input @keyup.ctrl.enter="submit" />

<!-- Mouse modifiers -->
<button @click.left="leftClick">Left</button>
<button @click.right="rightClick">Right</button>
```

## Form Input Binding

```vue
<template>
  <form @submit.prevent="submitForm">
    <!-- Text input -->
    <input v-model="form.name" type="text" />
    
    <!-- Textarea -->
    <textarea v-model="form.message"></textarea>
    
    <!-- Checkbox -->
    <input v-model="form.checked" type="checkbox" />
    
    <!-- Multiple checkboxes -->
    <input v-model="form.colors" value="red" type="checkbox">Red
    <input v-model="form.colors" value="blue" type="checkbox">Blue
    
    <!-- Radio -->
    <input v-model="form.gender" value="male" type="radio">Male
    <input v-model="form.gender" value="female" type="radio">Female
    
    <!-- Select -->
    <select v-model="form.country">
      <option value="cn">China</option>
      <option value="us">USA</option>
    </select>
    
    <!-- Modifiers -->
    <input v-model.lazy="form.name" />
    <input v-model.number="form.age" />
    <input v-model.trim="form.nickname" />
    
    <button type="submit">Submit</button>
  </form>
</template>

<script setup>
import { reactive } from 'vue'

const form = reactive({
  name: '',
  message: '',
  checked: false,
  colors: [],
  gender: '',
  country: '',
  age: 0,
  nickname: ''
})

const submitForm = () => {
  console.log(form)
}
</script>
```

## Computed Properties

```javascript
import { ref, computed } from 'vue'

const firstName = ref('John')
const lastName = ref('Doe')

// Computed - cached, only re-evaluates when dependencies change
const fullName = computed(() => {
  return `${firstName.value} ${lastName.value}`
})

// Method - re-evaluates every time
const getFullName = () => {
  return `${firstName.value} ${lastName.value}`
}
```

## Watchers

```javascript
import { ref, watch } from 'vue'

const count = ref(0)

// Simple watcher
watch(count, (newValue, oldValue) => {
  console.log(`Changed from ${oldValue} to ${newValue}`)
})

// Deep watcher
const state = ref({ name: 'Vue' })
watch(state, (newValue) => {
  console.log('State changed:', newValue)
}, { deep: true })

// Watch multiple sources
const firstName = ref('John')
const lastName = ref('Doe')
watch([firstName, lastName], ([newFirst, newLast]) => {
  console.log(`Name: ${newFirst} ${newLast}`)
})
```

## Lifecycle Hooks

### Vue 3 Composition API

| Hook | Description |
|------|-------------|
| `setup()` | Called before other hooks, setup |
| `onMounted()` | After component is mounted |
| `onUpdated()` | After component is updated |
| `onUnmounted()` | After component is unmounted |
| `onErrorCaptured()` | When error is captured |

```javascript
import { 
  onMounted, 
  onUpdated, 
  onUnmounted 
} from 'vue'

export default {
  setup() {
    onMounted(() => {
      console.log('Mounted!')
    })
    onUpdated(() => {
      console.log('Updated!')
    })
    onUnmounted(() => {
      console.log('Unmounted!')
    })
  }
}
```

### Vue 2 Options API

| Hook | Description |
|------|-------------|
| `beforeCreate` | Before instance is created |
| `created` | After instance is created |
| `beforeMount` | Before mounting |
| `mounted` | After mounting |
| `beforeUpdate` | Before update |
| `updated` | After update |
| `beforeDestroy` | Before destruction |
| `destroyed` | After destruction |

## Slots

### Default Slot

```vue
<!-- Parent.vue -->
<MyCard>
  <p>This goes to default slot</p>
</MyCard>

<!-- MyCard.vue -->
<template>
  <div class="card">
    <slot></slot>
  </div>
</template>
```

### Named Slots

```vue
<!-- Parent.vue -->
<MyLayout>
  <template #header>Header</template>
  <template #default>Content</template>
  <template #footer>Footer</template>
</MyLayout>

<!-- MyLayout.vue -->
<template>
  <div>
    <header><slot name="header"></slot></header>
    <main><slot></slot></main>
    <footer><slot name="footer"></slot></footer>
  </div>
</template>
```

### Scoped Slots

```vue
<!-- MyList.vue -->
<template>
  <ul>
    <li v-for="item in items" :key="item.id">
      <slot :item="item"></slot>
    </li>
  </ul>
</template>

<script setup>
defineProps(['items'])
</script>

<!-- Parent.vue -->
<template>
  <MyList :items="items">
    <template #default="{ item }">
      {{ item.name }} - {{ item.price }}
    </template>
  </MyList>
</template>
```

## Vue Router

### Basic Setup

```javascript
// router/index.js
import { createRouter, createWebHistory } from 'vue-router'
import Home from '../views/Home.vue'
import About from '../views/About.vue'
import User from '../views/User.vue'

const routes = [
  { path: '/', component: Home },
  { path: '/about', component: About },
  { path: '/user/:id', component: User, props: true }
]

const router = createRouter({
  history: createWebHistory(),
  routes
})

export default router
```

### Navigation

```vue
<template>
  <!-- Programmatic navigation -->
  <button @click="$router.push('/about')">About</button>
  <button @click="$router.back()">Back</button>
  <button @click="$router.forward()">Forward</button>
  
  <!-- Router link -->
  <router-link to="/">Home</router-link>
  <router-link to="/about">About</router-link>
  <router-link :to="`/user/${userId}`">User</router-link>
</template>

<script setup>
import { useRouter } from 'vue-router'

const router = useRouter()
const userId = 123

const navigate = () => {
  router.push('/about')
}
</script>
```

### Route Information

```vue
<script setup>
import { useRoute } from 'vue-router'

const route = useRoute()

// Access route params
console.log(route.params.id)

// Access query params
console.log(route.query.name)

// Access hash
console.log(route.hash)
</script>
```

## Pinia State Management

### Store Definition

```javascript
// stores/user.js
import { defineStore } from 'pinia'

export const useUserStore = defineStore('user', {
  state: () => ({
    name: 'John',
    email: 'john@example.com',
    age: 25
  }),
  
  getters: {
    isAdult: (state) => state.age >= 18,
    displayName: (state) => `${state.name} (${state.age})`
  },
  
  actions: {
    updateName(newName) {
      this.name = newName
    },
    async fetchUser() {
      const data = await fetch('/api/user')
      const user = await data.json()
      this.$patch(user)
    }
  }
})
```

### Using Store

```vue
<script setup>
import { storeToRefs } from 'pinia'
import { useUserStore } from '../stores/user'

const userStore = useUserStore()

// Access state directly
console.log(userStore.name)

// Using storeToRefs for reactivity
const { name, email } = storeToRefs(userStore)

// Call actions
userStore.updateName('Jane')

// Async action
userStore.fetchUser()
</script>
```

## Best Practices

### 1. Use Composition API

```javascript
// Recommended
import { ref, computed } from 'vue'

export default {
  setup() {
    const count = ref(0)
    return { count }
  }
}
```

### 2. Use Script Setup

```vue
<script setup>
// More concise syntax
</script>
```

### 3. Define Props with Validation

```javascript
defineProps({
  title: {
    type: String,
    required: true,
    default: 'Default Title'
  },
  count: {
    type: Number,
    default: 0
  }
})
```

### 4. Use Computed for Derived State

```javascript
const firstName = ref('John')
const lastName = ref('Doe')

// Good - computed
const fullName = computed(() => `${firstName.value} ${lastName.value}`)

// Bad - method
const getFullName = () => `${firstName.value} ${lastName.value}`
```

### 5. Keep Components Small and Focused

```vue
<!-- Good - focused component -->
<template>
  <UserAvatar :user="user" size="large" />
</template>

<!-- Good - separate concerns -->
<template>
  <UserProfile :user="user" />
  <UserActivity :userId="user.id" />
  <UserSettings :user="user" />
</template>
```

## Common Patterns

### Loading State

```vue
<template>
  <div v-if="loading">Loading...</div>
  <div v-else-if="error">{{ error }}</div>
  <div v-else>{{ data }}</div>
</template>
```

### Async Component

```vue
<script setup>
import { defineAsyncComponent } from 'vue'

const AsyncComponent = defineAsyncComponent(() =>
  import('./HeavyComponent.vue')
)
</script>

<template>
  <Suspense>
    <AsyncComponent />
    <template #fallback>
      Loading...
    </template>
  </Suspense>
</template>
```

### Teleport

```vue
<template>
  <button @click="show = true">Open Modal</button>
  
  <Teleport to="body">
    <div v-if="show" class="modal">
      Modal Content
      <button @click="show = false">Close</button>
    </div>
  </Teleport>
</template>
```
