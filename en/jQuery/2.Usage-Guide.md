## Basic Concepts

### What is jQuery?

jQuery is a JavaScript library that simplifies DOM manipulation, event handling, and Ajax.

### The $ Function

```javascript
// Select element
$('#myId')
$('.myClass')
$('div')

// Execute when ready
$(document).ready(function() {
  // Code here
})

// Shorthand
$(function() {
  // Code here
})
```

## Selectors

### Basic Selectors

```javascript
// ID selector
$('#myId')

// Class selector
$('.myClass')

// Element selector
$('div')

// Universal selector
$('*')

// Attribute selector
$('[href]')
$('a[href="https://example.com"]')

// Pseudo selectors
$('li:first')
$('li:last')
$('li:even')
$('li:odd')
$('li:eq(0)')  // First element
$('li:gt(0)')  // Greater than
$('li:lt(2)')  // Less than
```

### Traversing

```javascript
// Find descendants
$('#parent').find('.child')

// Children
$('#parent').children('.child')

// Parent
$('.child').parent()

// Siblings
$('.item').siblings()
$('.item').next()
$('.item').prev()
$('.item').nextAll()
$('.item').prevAll()

// Filter
$('li').filter('.active')
$('li').first()
$('li').last()
$('li').eq(0)
```

## DOM Manipulation

### Get/Set Content

```javascript
// Get text
var text = $('#myElement').text()

// Set text
$('#myElement').text('New text')

// Get HTML
var html = $('#myElement').html()

// Set HTML
$('#myElement').html('<span>New content</span>')

// Get value
var value = $('#myInput').val()

// Set value
$('#myInput').val('New value')
```

### Attributes

```javascript
// Get attribute
var href = $('a').attr('href')

// Set attribute
$('a').attr('href', 'https://example.com')

// Remove attribute
$('a').removeAttr('href')

// Get data attribute
var data = $('#myElement').data('name')

// Set data attribute
$('#myElement').data('name', 'value')
```

### CSS

```javascript
// Get CSS
var color = $('#myElement').css('color')

// Set CSS
$('#myElement').css('color', 'red')
$('#myElement').css({
  'color': 'red',
  'font-size': '16px',
  'backgroundColor': 'blue'
})

// Add class
$('#myElement').addClass('active')

// Remove class
$('#myElement').removeClass('active')

// Toggle class
$('#myElement').toggleClass('active')

// Check/has class
$('#myElement').hasClass('active')
```

### DOM Creation/Removal

```javascript
// Create element
var newDiv = $('<div>', {
  id: 'newDiv',
  class: 'container',
  text: 'Hello!'
})

// Append
$('#parent').append(newDiv)
$('#parent').prepend(newDiv)

// Insert after/before
$('#existing').after(newDiv)
$('#existing').before(newDiv)

// Wrap
$('.content').wrap('<div class="wrapper"></div>')

// Remove
$('#myElement').remove()

// Empty
$('#myElement').empty()
```

## Events

### Event Binding

```javascript
// Click event
$('#myButton').click(function() {
  console.log('Clicked!')
})

// Multiple events
$('#myElement').on('click hover', function() {
  console.log('Event!')
})

// Event object
$('#myElement').click(function(event) {
  console.log(event.type)    // click
  console.log(event.target)  // Element that triggered
  console.log(event.currentTarget)  // Element bound to
})

// Event delegation
$('#parent').on('click', '.child', function() {
  console.log('Child clicked!')
})
```

### Common Events

```javascript
// Mouse events
$('#myElement').click(function() {})
$('#myElement').dblclick(function() {})
$('#myElement').hover(function() {}, function() {})
$('#myElement').mousedown(function() {})
$('#myElement').mouseup(function() {})
$('#myElement').mouseenter(function() {})
$('#myElement').mouseleave(function() {})
$('#myElement').mousemove(function(e) {})

// Keyboard events
$('#myInput').keydown(function(e) {})
$('#myInput').keyup(function(e) {})
$('#myInput').keypress(function(e) {})

// Form events
$('#myForm').submit(function(e) {})
$('#myInput').focus(function() {})
$('#myInput').blur(function() {})
$('#myInput').change(function() {})
$('#myInput').select(function() {})

// Document events
$(document).ready(function() {})
$(window).resize(function() {})
$(window).scroll(function() {})
```

### Event Methods

```javascript
// Prevent default
$('a').click(function(e) {
  e.preventDefault()
})

// Stop propagation
$('.child').click(function(e) {
  e.stopPropagation()
})

// One-time event
$('#myElement').one('click', function() {
  console.log('This will only run once!')
})

// Unbind events
$('#myElement').off('click')
$('#myElement').off()
```

## Effects and Animation

### Basic Effects

```javascript
// Hide/Show
$('#myElement').hide()
$('#myElement').show()
$('#myElement').toggle()

// Fade
$('#myElement').fadeIn()
$('#myElement').fadeOut()
$('#myElement').fadeToggle()
$('#myElement').fadeTo(300, 0.5)

// Slide
$('#myElement').slideDown()
$('#myElement').slideUp()
$('#myElement').slideToggle()
```

### Custom Animation

```javascript
// Animate
$('#myElement').animate({
  'left': '100px',
  'opacity': 0.5
}, 300)  // Duration in ms

// Animate with callback
$('#myElement').animate({
  'left': '100px'
}, 300, function() {
  console.log('Animation complete!')
})

// Stop animation
$('#myElement').stop()

// Delay
$('#myElement').delay(300).fadeOut()
```

### Queue

```javascript
$('#myElement')
  .slideUp()
  .delay(300)
  .fadeIn()
  .animate({ left: '+=100' })
```

## AJAX

### GET Request

```javascript
$.get('https://api.example.com/data', function(data) {
  console.log(data)
})

// With parameters
$.get('https://api.example.com/search', { q: 'term' }, function(data) {
  console.log(data)
})
```

### POST Request

```javascript
$.post('https://api.example.com/submit', {
  name: 'John',
  email: 'john@example.com'
}, function(data) {
  console.log(data)
})
```

### AJAX

```javascript
$.ajax({
  url: 'https://api.example.com/data',
  method: 'GET',
  data: { id: 1 },
  success: function(data) {
    console.log(data)
  },
  error: function(xhr, status, error) {
    console.error(error)
  },
  complete: function() {
    console.log('Request complete')
  }
})
```

### JSON

```javascript
$.getJSON('https://api.example.com/data.json', function(data) {
  console.log(data)
})

// With JSONP
$.ajax({
  url: 'https://api.example.com/data',
  dataType: 'jsonp',
  success: function(data) {
    console.log(data)
  }
})
```

## Utilities

### Type Checking

```javascript
$.isArray([])           // true
$.isFunction(function(){}) // true
$.isNumeric('42')        // true
$.isPlainObject({})     // true
$.isEmptyObject({})     // true
$.isWindow(window)       // true
```

### Array/Object

```javascript
$.each([1, 2, 3], function(index, value) {
  console.log(value)
})

$.each({ name: 'John', age: 30 }, function(key, value) {
  console.log(key + ': ' + value)
})

$.map([1, 2, 3], function(value) {
  return value * 2
})

$.merge([1, 2], [3, 4])  // [1, 2, 3, 4]
$.unique([])  // Remove duplicates
$.inArray(2, [1, 2, 3])  // Returns index: 1
```

### Strings

```javascript
$.trim('  hello  ')  // 'hello'
$.escapeSelector('.my-class')  // '\\.my-class'
$.htmlEscape('<script>')  // '&lt;script&gt;'
$.parseJSON('{"key": "value"}')  // Object
```

### Other

```javascript
$.noop()  // Empty function
$.proxy(function() {}, context)  // Bind context
$.extend({ a: 1 }, { b: 2 })  // Merge objects
$.param({ a: 1, b: 2 })  // 'a=1&b=2'
```

## Best Practices

1. **Use latest version**: jQuery 3.x
2. **Cache selectors**: Store in variables
3. **Use event delegation**: For dynamic elements
4. **Chain methods**: For better performance
5. **Use native when possible**: Some native JS is faster
