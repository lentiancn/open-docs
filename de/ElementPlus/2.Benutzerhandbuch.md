## 1. Komponenten-Grundlagen

### 1.1 Basis-Button

Buttons sind eine der am häufigsten verwendeten Komponenten. Element Plus bietet mehrere Button-Typen.

```vue
<template>
  <!-- Primärer Button -->
  <el-button type="primary">Primärer Button</el-button>
  
  <!-- Erfolgs-Button -->
  <el-button type="success">Erfolgs-Button</el-button>
  
  <!-- Info-Button -->
  <el-button type="info">Info-Button</el-button>
  
  <!-- Warnungs-Button -->
  <el-button type="warning">Warnungs-Button</el-button>
  
  <!-- Gefahr-Button -->
  <el-button type="danger">Gefahr-Button</el-button>
  
  <!-- Text-Button -->
  <el-button type="text">Text-Button</el-button>
  
  <!-- Einfacher Button -->
  <el-button plain>Einfacher Button</el-button>
  
  <!-- Abgerundeter Button -->
  <el-button round>Abgerundeter Button</el-button>
  
  <!-- Icon-Button -->
  <el-button icon="el-icon-search">Suchen</el-button>
</template>
```

### 1.2 Button-Attribute

| Attribut | Beschreibung | Typ | Optionale Werte | Standard |
|------|------|------|-----------|----------|
| type | Button-Typ | string | primary / success / warning / danger / info / text | — |
| plain | Ob einfacher Button | boolean | — | false |
| round | Ob abgerundeter Button | boolean | — | false |
| circle | Ob Kreis-Button | boolean | — | false |
| loading | Ob Ladend | boolean | — | false |
| disabled | Ob deaktiviert | boolean | — | false |
| icon | Icon-Klassenname | string | — | — |
| size | Größe | string | large / default / small | — |

### 1.3 Button-Ereignisse

```vue
<template>
  <el-button @click="handleClick">Klick-Ereignis</el-button>
</template>

<script setup>
const handleClick = () => {
  console.log('Button geklickt')
}
</script>
```

## 2. Formular-Komponenten

### 2.1 Basis-Formular

```vue
<template>
  <el-form :model="form" :rules="rules" ref="formRef" label-width="100px">
    <el-form-item label="Benutzername" prop="username">
      <el-input v-model="form.username" placeholder="Benutzername eingeben" />
    </el-form-item>
    
    <el-form-item label="Passwort" prop="password">
      <el-input v-model="form.password" type="password" placeholder="Passwort eingeben" show-password />
    </el-form-item>
    
    <el-form-item label="E-Mail" prop="email">
      <el-input v-model="form.email" placeholder="E-Mail eingeben" />
    </el-form-item>
    
    <el-form-item label="Geschlecht" prop="gender">
      <el-radio-group v-model="form.gender">
        <el-radio label="male">Männlich</el-radio>
        <el-radio label="female">Weiblich</el-radio>
      </el-radio-group>
    </el-form-item>
    
    <el-form-item label="Hobby" prop="hobby">
      <el-checkbox-group v-model="form.hobby">
        <el-checkbox label="Basketball" />
        <el-checkbox label="FußBall" />
        <el-checkbox label="Schwimmen" />
      </el-checkbox-group>
    </el-form-item>
    
    <el-form-item label="Biografie" prop="bio">
      <el-input v-model="form.bio" type="textarea" :rows="3" placeholder="Biografie eingeben" />
    </el-form-item>
    
    <el-form-item>
      <el-button type="primary" @click="submitForm">Absenden</el-button>
      <el-button @click="resetForm">Zurücksetzen</el-button>
    </el-form-item>
  </el-form>
</template>

<script setup>
import { reactive, ref } from 'vue'

const formRef = ref(null)

const form = reactive({
  username: '',
  password: '',
  email: '',
  gender: '',
  hobby: [],
  bio: ''
})

const rules = {
  username: [
    { required: true, message: 'Bitte Benutzername eingeben', trigger: 'blur' },
    { min: 3, max: 20, message: 'Benutzername muss 3-20 Zeichen haben', trigger: 'blur' }
  ],
  password: [
    { required: true, message: 'Bitte Passwort eingeben', trigger: 'blur' },
    { min: 6, message: 'Passwort muss mindestens 6 Zeichen haben', trigger: 'blur' }
  ],
  email: [
    { required: true, message: 'Bitte E-Mail eingeben', trigger: 'blur' },
    { type: 'email', message: 'Bitte gültiges E-Mail-Format eingeben', trigger: 'blur' }
  ],
  gender: [
    { required: true, message: 'Bitte Geschlecht auswählen', trigger: 'change' }
  ]
}

const submitForm = async () => {
  if (!formRef.value) return
  
  await formRef.value.validate((valid) => {
    if (valid) {
      console.log('Formulardaten:', form)
      ElMessage.success('Erfolgreich abgesendet')
    } else {
      ElMessage.error('Bitte Formular überprüfen')
    }
  })
}

const resetForm = () => {
  formRef.value?.resetFields()
}
</script>
```

## 3. Tisch-Komponente

### 3.1 Basis-Tisch

```vue
<template>
  <el-table :data="tableData" stripe border>
    <el-table-column prop="date" label="Datum" width="180" />
    <el-table-column prop="name" label="Name" width="180" />
    <el-table-column prop="address" label="Adresse" />
    <el-table-column label="Aktionen" width="200">
      <template #default="{ row }">
        <el-button type="primary" size="small" @click="handleEdit(row)">Bearbeiten</el-button>
        <el-button type="danger" size="small" @click="handleDelete(row)">Löschen</el-button>
      </template>
    </el-table-column>
  </el-table>
</template>

<script setup>
import { ref } from 'vue'

const tableData = ref([
  { date: '2024-01-01', name: 'Max Mustermann', address: 'Berlin' },
  { date: '2024-01-02', name: 'Maria Müller', address: 'München' },
  { date: '2024-01-03', name: 'Hans Schmidt', address: 'Hamburg' }
])

const handleEdit = (row) => {
  console.log('Bearbeiten:', row)
}

const handleDelete = (row) => {
  console.log('Löschen:', row)
}
</script>
```

### 3.2 Tisch mit Seitennummerierung

```vue
<template>
  <div>
    <el-table :data="tableData" v-loading="loading">
      <el-table-column prop="id" label="ID" width="80" />
      <el-table-column prop="title" label="Titel" />
      <el-table-column prop="author" label="Autor" width="120" />
      <el-table-column prop="date" label="Veröffentlichungsdatum" width="180" />
    </el-table>
    
    <el-pagination
      v-model:current-page="currentPage"
      v-model:page-size="pageSize"
      :total="total"
      :page-sizes="[10, 20, 50, 100]"
      layout="total, sizes, prev, pager, next, jumper"
      @size-change="handleSizeChange"
      @current-change="handleCurrentChange"
    />
  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue'

const tableData = ref([])
const loading = ref(false)
const currentPage = ref(1)
const pageSize = ref(10)
const total = ref(0)

const loadData = async () => {
  loading.value = true
  // API-Aufruf simulieren
  setTimeout(() => {
    tableData.value = [
      { id: 1, title: 'Artikeltitel 1', author: 'Autor 1', date: '2024-01-01' },
      { id: 2, title: 'Artikeltitel 2', author: 'Autor 2', date: '2024-01-02' }
    ]
    total.value = 100
    loading.value = false
  }, 500)
}

const handleSizeChange = (val) => {
  pageSize.value = val
  loadData()
}

const handleCurrentChange = (val) => {
  currentPage.value = val
  loadData()
}

onMounted(() => {
  loadData()
})
</script>
```

## 4. Dialog und Schublade

### 4.1 Dialog

```vue
<template>
  <el-button type="primary" @click="dialogVisible = true">Dialog öffnen</el-button>
  
  <el-dialog
    v-model="dialogVisible"
    title="Informationen bearbeiten"
    width="500px"
    :close-on-click-modal="false"
    @close="handleClose"
  >
    <el-form :model="form" label-width="80px">
      <el-form-item label="Name">
        <el-input v-model="form.name" />
      </el-form-item>
      <el-form-item label="Beschreibung">
        <el-input v-model="form.desc" type="textarea" :rows="3" />
      </el-form-item>
    </el-form>
    
    <template #footer>
      <el-button @click="dialogVisible = false">Abbrechen</el-button>
      <el-button type="primary" @click="handleConfirm">Bestätigen</el-button>
    </template>
  </el-dialog>
</template>

<script setup>
import { ref, reactive } from 'vue'

const dialogVisible = ref(false)

const form = reactive({
  name: '',
  desc: ''
})

const handleClose = () => {
  form.name = ''
  form.desc = ''
}

const handleConfirm = () => {
  console.log('Absenden:', form)
  dialogVisible.value = false
  ElMessage.success('Erfolgreich')
}
</script>
```

## 5. Nachrichtenbenachrichtigungen

### 5.1 Globale Nachrichten

```vue
<script setup>
import { ElMessage } from 'element-plus'

// Erfolgsnachricht
const showSuccess = () => {
  ElMessage.success('Operation erfolgreich')
}

// Warnungsnachricht
const showWarning = () => {
  ElMessage.warning('Warnmeldung')
}

// Fehlernachricht
const showError = () => {
  ElMessage.error('Operation fehlgeschlagen')
}

// Informationsnachricht
const showInfo = () => {
  ElMessage.info('Informationsmeldung')
}
</script>
```

### 5.2 Benachrichtigungen

```vue
<script setup>
import { ElNotification } from 'element-plus'

// Erfolgsbenachrichtigung
const notifySuccess = () => {
  ElNotification({
    title: 'Erfolg',
    message: 'Operation erfolgreich abgeschlossen',
    type: 'success',
  })
}

// Fehlerbenachrichtigung
const notifyError = () => {
  ElNotification({
    title: 'Fehler',
    message: 'Operation fehlgeschlagen',
    type: 'error',
  })
}
</script>
```
