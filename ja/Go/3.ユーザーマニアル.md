# Go ユーザーマニアル

> Go プログラミング完全ガイド

## 変数と型
```go
var name = "Go"
age := 10

// 型
var i int = 10
var f float64 = 3.14
var s string = "Hello"
var b bool = true
```

## 関数
```go
func add(a, b int) int {
    return a + b
}

func divide(a, b int) (int, error) {
    if b == 0 {
        return 0, errors.New("zero division")
    }
    return a / b, nil
}
```

## Goroutine と Channel
```go
go func() {
    fmt.Println("Hello from goroutine")
}()

ch := make(chan int)
ch <- 10
value := <-ch
```

## テスト
```go
import "testing"

func TestAdd(t *testing.T) {
    result := add(2, 3)
    if result != 5 {
        t.Errorf("Expected 5, got %d", result)
    }
}
```
