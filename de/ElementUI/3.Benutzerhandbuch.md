### Form (Formular)

```html
<el-form :model="ruleForm" :rules="rules" ref="ruleForm" label-width="100px">
  <el-form-item label="Aktionsname" prop="name">
    <el-input v-model="ruleForm.name"></el-input>
  </el-form-item>
  <el-form-item label="Aktionsbereich" prop="region">
    <el-select v-model="ruleForm.region" placeholder="Bitte auswählen">
      <el-option label="Bereich 1" value="shanghai"></el-option>
      <el-option label="Bereich 2" value="beijing"></el-option>
    </el-select>
  </el-form-item>
  <el-form-item label="Aktionsdatum" prop="date1">
    <el-date-picker
      v-model="ruleForm.date1"
      type="date"
      placeholder="Datum auswählen">
    </el-date-picker>
  </el-form-item>
  <el-form-item>
    <el-button type="primary" @click="submitForm('ruleForm')">Absenden</el-button>
    <el-button @click="resetForm('ruleForm')">Zurücksetzen</el-button>
  </el-form-item>
</el-form>
```

Validierungsregeln:

```javascript
export default {
  data() {
    return {
      ruleForm: {
        name: '',
        region: '',
        date1: ''
      },
      rules: {
        name: [
          { required: true, message: 'Bitte Aktionsnamen eingeben', trigger: 'blur' },
          { min: 3, max: 5, message: 'Länge zwischen 3 und 5 Zeichen', trigger: 'blur' }
        ],
        region: [
          { required: true, message: 'Bitte Bereich auswählen', trigger: 'change' }
        ],
        date1: [
          { required: true, message: 'Bitte Datum auswählen', trigger: 'change' }
        ]
      }
    }
  },
  methods: {
    submitForm(formName) {
      this.$refs[formName].validate((valid) => {
        if (valid) {
          alert('submit!')
        } else {
          return false
        }
      })
    },
    resetForm(formName) {
      this.$refs[formName].resetFields()
    }
  }
}
```

### Formularvalidierung

```javascript
// Benutzerdefinierte Validierungsregel
const validatePass = (rule, value, callback) => {
  if (value === '') {
    callback(new Error('Bitte Passwort eingeben'))
  } else {
    if (this.ruleForm.checkPass !== '') {
      this.$refs.ruleForm.validateField('checkPass')
    }
    callback()
  }
}
```
