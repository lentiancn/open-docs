# Руководство пользователя Node.js

В этом руководстве показано, как использовать Node.js для создания серверных приложений и инструментов командной строки.

## Быстрый старт

### Создание первого приложения

```javascript
// hello.js
console.log('Hello, Node.js!');

// Запуск
node hello.js
```

### Создание HTTP-сервера

```javascript
// server.js
const http = require('http');

const server = http.createServer((req, res) => {
  res.writeHead(200, { 'Content-Type': 'text/plain' });
  res.end('Hello World!');
});

server.listen(3000, () => {
  console.log('Сервер работает на http://localhost:3000/');
});
```

Запуск:

```bash
node server.js
```

## Основные модули

### fs - Файловая система

```javascript
const fs = require('fs');

// Чтение файла
fs.readFile('file.txt', 'utf8', (err, data) => {
  if (err) throw err;
  console.log(data);
});

// Запись в файл
fs.writeFile('file.txt', 'Hello!', (err) => {
  if (err) throw err;
  console.log('Файл сохранён');
});

// Синхронное чтение
const data = fs.readFileSync('file.txt', 'utf8');
```

### path - Пути

```javascript
const path = require('path');

path.join('/foo', 'bar', 'baz'); // /foo/bar/baz
path.resolve(__dirname, 'file.txt'); // Абсолютный путь
path.extname('file.txt'); // .txt
path.basename('/foo/bar.txt'); // bar.txt
```

### os - Операционная система

```javascript
const os = require('os');

os.platform(); // 'win32', 'linux', 'darwin'
os.arch(); // 'x64', 'arm64'
os.cpus(); // Информация о CPU
os.freemem(); // Свободная память
os.totalmem(); // Всего памяти
```

### events - События

```javascript
const EventEmitter = require('events');

class MyEmitter extends EventEmitter {}

const myEmitter = new MyEmitter();

myEmitter.on('event', () => {
  console.log('Событие вызвано!');
});

myEmitter.emit('event');
```

### crypto - Криптография

```javascript
const crypto = require('crypto');

const hash = crypto.createHash('sha256');
hash.update('password');
const password = hash.digest('hex');
```

## Фреймворк Express

### Установка

```bash
npm install express
```

### Создание приложения

```javascript
const express = require('express');
const app = express();

app.get('/', (req, res) => {
  res.send('Hello World!');
});

app.listen(3000, () => {
  console.log('Приложение работает на http://localhost:3000');
});
```

### Маршруты

```javascript
// GET-маршрут
app.get('/api/users', (req, res) => {
  res.json({ users: [] });
});

// POST-маршрут
app.post('/api/users', (req, res) => {
  const user = req.body;
  res.json({ success: true, user });
});

// Маршрут с параметрами
app.get('/api/users/:id', (req, res) => {
  const { id } = req.params;
  res.json({ id });
});
```

### Промежуточное ПО

```javascript
# Парсинг JSON
app.use(express.json());

# Парсинг URL-кодированных данных
app.use(express.urlencoded({ extended: true }));

# Логирующее промежуточное ПО
app.use((req, res, next) => {
  console.log(`${req.method} ${req.url}`);
  next();
});

# Обработка ошибок
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).send('Something broke!');
});
```

## Асинхронное программирование

### Callback

```javascript
fs.readFile('file.txt', 'utf8', (err, data) => {
  if (err) throw err;
  console.log(data);
});
```

### Promise

```javascript
const fs = require('fs').promises;

async function read() {
  try {
    const data = await fs.readFile('file.txt', 'utf8');
    console.log(data);
  } catch (err) {
    console.error(err);
  }
}

read();
```

### Async/Await

```javascript
const axios = require('axios');

async function fetchData() {
  try {
    const response = await axios.get('https://api.example.com/data');
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}

fetchData();
```

## Лучшие практики

### 1. Использовать async/await

```javascript
// Избегать
getData((err, data) => {
  if (err) handleError(err);
  processData(data, (err, result) => {
    if (err) handleError(err);
    saveResult(result);
  });
});

// Рекомендуется
async function main() {
  try {
    const data = await getData();
    const result = await processData(data);
    await saveResult(result);
  } catch (err) {
    handleError(err);
  }
}
```

### 2. Обработка ошибок

```javascript
// Всегда перехватывать асинхронные ошибки
async function fetchData() {
  try {
    const response = await fetch(url);
    return await response.json();
  } catch (error) {
    console.error('Ошибка получения данных:', error);
    throw error;
  }
}
```

### 3. Использовать переменные окружения

```javascript
require('dotenv').config();

const port = process.env.PORT || 3000;
const dbUrl = process.env.DATABASE_URL;
```
