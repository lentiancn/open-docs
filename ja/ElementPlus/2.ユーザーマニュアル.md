## 1. コンポーネント使用基礎

### 1.1 基礎ボタン

ボタンは最も一般的に使用されるコンポーネントの1つで、Element Plusは 多種類のボタンを提供します。

```vue
<template>
  <!-- プライマリーボタン -->
  <el-button type="primary">プライマリーボタン</el-button>
  
  <!-- 成功ボタン -->
  <el-button type="success">成功ボタン</el-button>
  
  <!-- 情報ボタン -->
  <el-button type="info">情報ボタン</el-button>
  
  <!-- 警告ボタン -->
  <el-button type="warning">警告ボタン</el-button>
  
  <!-- 危険ボタン -->
  <el-button type="danger">危険ボタン</el-button>
  
  <!-- テキストボタン -->
  <el-button type="text">テキストボタン</el-button>
  
  <!-- プレーナーボタン -->
  <el-button plain>プレーナーボタン</el-button>
  
  <!-- ラウンドボタン -->
  <el-button round>ラウンドボタン</el-button>
  
  <!-- アイコンボタン -->
  <el-button icon="el-icon-search">検索</el-button>
</template>
```

### 1.2 ボタン属性

| 属性 | 説明 | 型 | オプション値 | デフォルト |
|------|------|------|-----------|----------|
| type | ボタンタイプ | string | primary / success / warning / danger / info / text | — |
| plain | プレーナーボタンかどうか | boolean | — | false |
| round | ラウンドボタンかどうか | boolean | — | false |
| circle | サークルトンかどうか | boolean | — | false |
| loading | ローディング中かどうか | boolean | — | false |
| disabled | 無効かどうか | boolean | — | false |
| icon | アイコンクラス名 | string | — | — |
| size | サイズ | string | large / default / small | — |

### 1.3 ボタンイベント

```vue
<template>
  <el-button @click="handleClick">クリックイベント</el-button>
</template>

<script setup>
const handleClick = () => {
  console.log('ボタンがクリックされました')
}
</script>
```

## 2. フォームコンポーネント

### 2.1 基礎フォーム

```vue
<template>
  <el-form :model="form" :rules="rules" ref="formRef" label-width="100px">
    <el-form-item label="ユーザー名" prop="username">
      <el-input v-model="form.username" placeholder="ユーザー名を入力" />
    </el-form-item>
    
    <el-form-item label="パスワード" prop="password">
      <el-input v-model="form.password" type="password" placeholder="パスワードを入力" show-password />
    </el-form-item>
    
    <el-form-item label="メール" prop="email">
      <el-input v-model="form.email" placeholder="メールを入力" />
    </el-form-item>
    
    <el-form-item label="性別" prop="gender">
      <el-radio-group v-model="form.gender">
        <el-radio label="male">男</el-radio>
        <el-radio label="female">女</el-radio>
      </el-radio-group>
    </el-form-item>
    
    <el-form-item label="趣味" prop="hobby">
      <el-checkbox-group v-model="form.hobby">
        <el-checkbox label="バスケットボール" />
        <el-checkbox label="サッカー" />
        <el-checkbox label="水泳" />
      </el-checkbox-group>
    </el-form-item>
    
    <el-form-item label="自己紹介" prop="bio">
      <el-input v-model="form.bio" type="textarea" :rows="3" placeholder="自己紹介を入力" />
    </el-form-item>
    
    <el-form-item>
      <el-button type="primary" @click="submitForm">提出</el-button>
      <el-button @click="resetForm">リセット</el-button>
    </el-form-item>
  </el-form>
</template>

<script setup>
import { reactive, ref } from 'vue'

const formRef = ref(null)

const form = reactive({
  username: '',
  password: '',
  email: '',
  gender: '',
  hobby: [],
  bio: ''
})

const rules = {
  username: [
    { required: true, message: 'ユーザー名を入力してください', trigger: 'blur' },
    { min: 3, max: 20, message: 'ユーザー名は3〜20文字', trigger: 'blur' }
  ],
  password: [
    { required: true, message: 'パスワードを入力してください', trigger: 'blur' },
    { min: 6, message: 'パスワードは6文字以上', trigger: 'blur' }
  ],
  email: [
    { required: true, message: 'メールを入力してください', trigger: 'blur' },
    { type: 'email', message: '有効なメール形式を入力', trigger: 'blur' }
  ],
  gender: [
    { required: true, message: '性別を選択してください', trigger: 'change' }
  ]
}

const submitForm = async () => {
  if (!formRef.value) return
  
  await formRef.value.validate((valid) => {
    if (valid) {
      console.log('フォームデータ：', form)
      ElMessage.success('提出成功')
    } else {
      ElMessage.error('フォームを確認してください')
    }
  })
}

const resetForm = () => {
  formRef.value?.resetFields()
}
</script>
```

## 3. テーブルコンポーネント

### 3.1 基礎テーブル

```vue
<template>
  <el-table :data="tableData" stripe border>
    <el-table-column prop="date" label="日付" width="180" />
    <el-table-column prop="name" label="名前" width="180" />
    <el-table-column prop="address" label="住所" />
    <el-table-column label="操作" width="200">
      <template #default="{ row }">
        <el-button type="primary" size="small" @click="handleEdit(row)">編集</el-button>
        <el-button type="danger" size="small" @click="handleDelete(row)">削除</el-button>
      </template>
    </el-table-column>
  </el-table>

<script setup>
import { ref } from 'vue'

const tableData = ref([
  { date: '2024-01-01', name: '山田太郎', address: '東京都渋谷区' },
  { date: '2024-01-02', name: '鈴木花子', address: '大阪府大阪市' },
  { date: '2024-01-03', name: '佐藤一郎', address: '神奈川県横浜市' }
])

const handleEdit = (row) => {
  console.log('編集：', row)
}

const handleDelete = (row) => {
  console.log('削除：', row)
}
</script>
```

### 3.2 ページネーション付きテーブル

```vue
<template>
  <div>
    <el-table :data="tableData" v-loading="loading">
      <el-table-column prop="id" label="ID" width="80" />
      <el-table-column prop="title" label="タイトル" />
      <el-table-column prop="author" label="著者" width="120" />
      <el-table-column prop="date" label="公開日" width="180" />
    </el-table>
    
    <el-pagination
      v-model:current-page="currentPage"
      v-model:page-size="pageSize"
      :total="total"
      :page-sizes="[10, 20, 50, 100]"
      layout="total, sizes, prev, pager, next, jumper"
      @size-change="handleSizeChange"
      @current-change="handleCurrentChange"
    />
  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue'

const tableData = ref([])
const loading = ref(false)
const currentPage = ref(1)
const pageSize = ref(10)
const total = ref(0)

const loadData = async () => {
  loading.value = true
  // API呼び出しをシミュレート
  setTimeout(() => {
    tableData.value = [
      { id: 1, title: '記事タイトル1', author: '著者1', date: '2024-01-01' },
      { id: 2, title: '記事タイトル2', author: '著者2', date: '2024-01-02' }
    ]
    total.value = 100
    loading.value = false
  }, 500)
}

const handleSizeChange = (val) => {
  pageSize.value = val
  loadData()
}

const handleCurrentChange = (val) => {
  currentPage.value = val
  loadData()
}

onMounted(() => {
  loadData()
})
</script>
```

## 4. ダイアログとドロワー

### 4.1 ダイアログ

```vue
<template>
  <el-button type="primary" @click="dialogVisible = true">ダイアログを開く</el-button>
  
  <el-dialog
    v-model="dialogVisible"
    title="情報を編集"
    width="500px"
    :close-on-click-modal="false"
    @close="handleClose"
  >
    <el-form :model="form" label-width="80px">
      <el-form-item label="名前">
        <el-input v-model="form.name" />
      </el-form-item>
      <el-form-item label="説明">
        <el-input v-model="form.desc" type="textarea" :rows="3" />
      </el-form-item>
    </el-form>
    
    <template #footer>
      <el-button @click="dialogVisible = false">キャンセル</el-button>
      <el-button type="primary" @click="handleConfirm">確定</el-button>
    </template>
  </el-dialog>
</template>

<script setup>
import { ref, reactive } from 'vue'

const dialogVisible = ref(false)

const form = reactive({
  name: '',
  desc: ''
})

const handleClose = () => {
  form.name = ''
  form.desc = ''
}

const handleConfirm = () => {
  console.log('提出：', form)
  dialogVisible.value = false
  ElMessage.success('操作成功')
}
</script>
```

## 5. メッセージ通知

### 5.1 グローバルメッセージ

```vue
<script setup>
import { ElMessage } from 'element-plus'

// 成功メッセージ
const showSuccess = () => {
  ElMessage.success('操作成功')
}

// 警告メッセージ
const showWarning = () => {
  ElMessage.warning('警告メッセージ')
}

// エラーメッセージ
const showError = () => {
  ElMessage.error('操作失敗')
}

// 情報メッセージ
const showInfo = () => {
  ElMessage.info('情報メッセージ')
}
</script>
```

### 5.2 通知

```vue

import { ElNotification } from 'element-plus'

// 成功通知
```
