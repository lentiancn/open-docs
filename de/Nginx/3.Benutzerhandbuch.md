# Nginx Benutzerhandbuch

## Grundlegende Operationen

### Nginx starten

```bash
# Nginx starten
nginx

# Oder systemctl verwenden (für systemd)
systemctl start nginx
```

### Nginx stoppen

```bash
# Schnelles Stoppen (sofort beenden)
nginx -s stop

# Anmutiges Stoppen (auf Abschluss der Anfragen warten)
nginx -s quit
```

### Konfiguration neu laden

```bash
# Konfigurationsdatei neu laden
nginx -s reload
```

### Logs neu öffnen

```bash
# Log-Dateien neu öffnen
nginx -s reopen
```

### Konfiguration prüfen

```bash
# Konfigurationssyntax testen
nginx -t

# Vollständige Konfiguration anzeigen
nginx -T
```

## Konfigurationsdateistruktur

### Hauptkonfigurationsstruktur

```
nginx.conf
├── main (globaler Block)
│   ├── user
│   ├── worker_processes
│   ├── error_log
│   └── pid
│
├── events (events-Block)
│   └── worker_connections
│
└── http (http-Block)
    ├── include
    ├── default_type
    ├── log_format
    ├── access_log
    ├── sendfile
    ├── keepalive_timeout
    ├── gzip
    │
    └── server (server-Block)
        ├── listen
        ├── server_name
        ├── location
        │   ├── root
        │   ├── index
        │   └── proxy_pass
        │
        └── error_page
```

### Häufige Direktiven

#### Main-Block

```nginx
# Benutzer der Nginx ausführt
user nginx;

# Anzahl der Worker-Prozesse (auto erkennt CPU-Kerne automatisch)
worker_processes auto;

# Fehler-Log-Pfad
error_log /var/log/nginx/error.log warn;

# PID-Datei-Pfad
pid /var/run/nginx.pid;
```

#### Events-Block

```nginx
events {
    # Maximale Verbindungen pro Worker
    worker_connections 1024;
    
    # epoll-Modell verwenden (Linux)
    use epoll;
    
    # Mehrere Verbindungen gleichzeitig akzeptieren
    multi_accept on;
}
```

#### HTTP-Block

```nginx
http {
    # MIME-Typen einbeziehen
    include /etc/nginx/mime.types;
    
    # Standard-Dateityp
    default_type application/octet-stream;
    
    # Zugriffs-Log
    access_log /var/log/nginx/access.log;
    
    # Datei senden
    sendfile on;
    
    # TCP_NOPUSH und TCP_NODELAY
    tcp_nopush on;
    tcp_nodelay on;
    
    # Keep-alive-Timeout
    keepalive_timeout 65;
    
    # Gzip-Komprimierung aktivieren
    gzip on;
}
```

## Statische Dateien bereitstellen

### Grundkonfiguration

```nginx
server {
    listen 80;
    server_name beispiel.de www.beispiel.de;
    
    # Stammverzeichnis
    root /var/www/html;
    
    # Standard-Indexdateien
    index index.html index.htm;
    
    location / {
        try_files $uri $uri/ =404;
    }
    
    # Zugriffs-Logs
    access_log /var/log/nginx/beispiel.de.access.log;
    error_log /var/log/nginx/beispiel.de.error.log;
}
```

### Verzeichnisauflistung

```nginx
location / {
    # Verzeichnisauflistung aktivieren
    autoindex on;
    
    # Dateigrößen anzeigen
    autoindex_exact_size on;
    
    # Lokale Zeit anzeigen
    autoindex_localtime on;
}
```

### Statisches Caching

```nginx
location ~* \.(jpg|jpeg|png|gif|ico|css|js|woff|woff2|ttf|svg)$ {
    expires 30d;
    add_header Cache-Control "public, no-transform";
}
```

## Reverse-Proxy

### Grundlegender Reverse-Proxy

```nginx
server {
    listen 80;
    server_name beispiel.de;
    
    location / {
        proxy_pass http://backend_server;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
```

### Reverse-Proxy mit Caching

```nginx
proxy_cache_path /var/cache/nginx levels=1:2 keys_zone=my_cache:10m max_size=100m inactive=60m use_temp_path=off;

server {
    listen 80;
    server_name beispiel.de;
    
    location / {
        proxy_cache my_cache;
        proxy_cache_valid 200 60m;
        proxy_cache_use_stale error timeout http_500 http_502 http_503 http_504;
        proxy_cache_lock on;
        
        proxy_pass http://backend_server;
        proxy_set_header Host $host;
    }
}
```

### WebSocket-Proxy

```nginx
location /ws/ {
    proxy_pass http://backend_server;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection "upgrade";
    proxy_set_header Host $host;
    proxy_read_timeout 86400;
}
```

## Load Balancing

### Round-Robin Load Balancing

```nginx
upstream backend {
    server backend1.beispiel.de;
    server backend2.beispiel.de;
    server backend3.beispiel.de;
}

server {
    listen 80;
    server_name beispiel.de;
    
    location / {
        proxy_pass http://backend;
    }
}
```

### Gewichtetes Load Balancing

```nginx
upstream backend {
    server backend1.beispiel.de weight=5;
    server backend2.beispiel.de weight=2;
    server backend3.beispiel.de;
}
```

### IP-Hash Load Balancing

```nginx
upstream backend {
    ip_hash;
    server backend1.beispiel.de;
    server backend2.beispiel.de;
    server backend3.beispiel.de down;
}
```

### Least Connections

```nginx
upstream backend {
    least_conn;
    server backend1.beispiel.de;
    server backend2.beispiel.de;
}
```

## SSL/TLS-Konfiguration

### Grundlegende HTTPS-Konfiguration

```nginx
server {
    listen 443 ssl http2;
    server_name beispiel.de;
    
    ssl_certificate /etc/nginx/ssl/beispiel.de.crt;
    ssl_certificate_key /etc/nginx/ssl/beispiel.de.key;
    
    # SSL-Protokollversionen
    ssl_protocols TLSv1.2 TLSv1.3;
    
    # SSL-Chiffre-Suiten
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;
    ssl_prefer_server_ciphers off;
    
    # HSTS
    add_header Strict-Transport-Security "max-age=63072000" always;
    
    location / {
        root /var/www/html;
        index index.html;
    }
}
```

### HTTP-zu-HTTPS-Umleitung

```nginx
server {
    listen 80;
    server_name beispiel.de;
    return 301 https://$server_name$request_uri;
}
```

## Häufige Konfigurationsbeispiele

### PHP-FPM-Konfiguration

```nginx
server {
    listen 80;
    server_name beispiel.de;
    root /var/www/html;
    index index.php index.html;
    
    location ~ \.php$ {
        fastcgi_pass unix:/var/run/php/php-fpm.sock;
        fastcgi_index index.php;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        include fastcgi_params;
    }
    
    location ~ /\.ht {
        deny all;
    }
}
```

### Node.js-Proxy

```nginx
server {
    listen 80;
    server_name beispiel.de;
    
    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}
```

### Ratenbegrenzung-Konfiguration

```nginx
# Verbindungsbasierte Ratenbegrenzung
limit_conn_zone $binary_remote_addr zone=addr:10m;

server {
    location / {
        limit_conn addr 10;
    }
}

# Anfragebasierte Ratenbegrenzung
limit_req_zone $binary_remote_addr zone=one:10m rate=10r/s;

server {
    location / {
        limit_req zone=one burst=20 nodelay;
    }
}
```

### Hotlink-Schutz-Konfiguration

```nginx
location ~* \.(jpg|jpeg|png|gif|ico|css|js)$ {
    valid_referers none blocked beispiel.de *.beispiel.de;
    
    if ($invalid_referer) {
        return 403;
    }
}
```

## Log-Verwaltung

### Benutzerdefiniertes Log-Format

```nginx
log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                '$status $body_bytes_sent "$http_referer" '
                '"$http_user_agent" "$http_x_forwarded_for"';

access_log /var/log/nginx/access.log main;
```

### Bedingtes Logging

```nginx
map $request_uri $loggable {
    ~^/health-check 0;
    default 1;
}

access_log /var/log/nginx/access.log combined if=$loggable;
```

### Log-Rotation

Logrotate verwenden:

```bash
# /etc/logrotate.d/nginx
/var/log/nginx/*.log {
    daily
    missingok
    rotate 14
    compress
    delaycompress
    notifempty
    create 0640 www-data adm
    sharedscripts
    postrotate
        [ -f /var/run/nginx.pid ] && kill -USR1 `cat /var/run/nginx.pid`
    endscript
}
```
