# Guía de Usuario de Node.js

Esta guía muestra cómo usar Node.js para construir aplicaciones backend y herramientas de línea de comandos.

## Inicio Rápido

### Crear tu Primera Aplicación

```javascript
// hello.js
console.log('Hello, Node.js!');

// Ejecutar
node hello.js
```

### Crear un Servidor HTTP

```javascript
// server.js
const http = require('http');

const server = http.createServer((req, res) => {
  res.writeHead(200, { 'Content-Type': 'text/plain' });
  res.end('Hello World!');
});

server.listen(3000, () => {
  console.log('Servidor ejecutándose en http://localhost:3000/');
});
```

Ejecutar:

```bash
node server.js
```

## Módulos Principales

### fs - Sistema de Archivos

```javascript
const fs = require('fs');

// Leer archivo
fs.readFile('file.txt', 'utf8', (err, data) => {
  if (err) throw err;
  console.log(data);
});

// Escribir archivo
fs.writeFile('file.txt', 'Hello!', (err) => {
  if (err) throw err;
  console.log('Archivo guardado');
});

// Lectura síncrona
const data = fs.readFileSync('file.txt', 'utf8');
```

### path - Rutas

```javascript
const path = require('path');

path.join('/foo', 'bar', 'baz'); // /foo/bar/baz
path.resolve(__dirname, 'file.txt'); // Ruta absoluta
path.extname('file.txt'); // .txt
path.basename('/foo/bar.txt'); // bar.txt
```

### os - Sistema Operativo

```javascript
const os = require('os');

os.platform(); // 'win32', 'linux', 'darwin'
os.arch(); // 'x64', 'arm64'
os.cpus(); // Información de CPU
os.freemem(); // Memoria libre
os.totalmem(); // Memoria total
```

### events - Eventos

```javascript
const EventEmitter = require('events');

class MyEmitter extends EventEmitter {}

const myEmitter = new MyEmitter();

myEmitter.on('event', () => {
  console.log('¡Evento disparado!');
});

myEmitter.emit('event');
```

### crypto - Criptografía

```javascript
const crypto = require('crypto');

const hash = crypto.createHash('sha256');
hash.update('password');
const password = hash.digest('hex');
```

## Framework Express

### Instalación

```bash
npm install express
```

### Crear Aplicación

```javascript
const express = require('express');
const app = express();

app.get('/', (req, res) => {
  res.send('Hello World!');
});

app.listen(3000, () => {
  console.log('Aplicación ejecutándose en http://localhost:3000');
});
```

### Rutas

```javascript
// Ruta GET
app.get('/api/users', (req, res) => {
  res.json({ users: [] });
});

// Ruta POST
app.post('/api/users', (req, res) => {
  const user = req.body;
  res.json({ success: true, user });
});

// Ruta con parámetros
app.get('/api/users/:id', (req, res) => {
  const { id } = req.params;
  res.json({ id });
});
```

### Middleware

```javascript
// Parsear JSON
app.use(express.json());

// Parsear datos URL-encoded
app.use(express.urlencoded({ extended: true }));

// Middleware de registro
app.use((req, res, next) => {
  console.log(`${req.method} ${req.url}`);
  next();
});

// Manejo de errores
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).send('Something broke!');
});
```

## Programación Asíncrona

### Callback

```javascript
fs.readFile('file.txt', 'utf8', (err, data) => {
  if (err) throw err;
  console.log(data);
});
```

### Promise

```javascript
const fs = require('fs').promises;

async function read() {
  try {
    const data = await fs.readFile('file.txt', 'utf8');
    console.log(data);
  } catch (err) {
    console.error(err);
  }
}

read();
```

### Async/Await

```javascript
const axios = require('axios');

async function fetchData() {
  try {
    const response = await axios.get('https://api.example.com/data');
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}

fetchData();
```

## Mejores Prácticas

### 1. Usar async/await

```javascript
// Evitar
getData((err, data) => {
  if (err) handleError(err);
  processData(data, (err, result) => {
    if (err) handleError(err);
    saveResult(result);
  });
});

// Recomendado
async function main() {
  try {
    const data = await getData();
    const result = await processData(data);
    await saveResult(result);
  } catch (err) {
    handleError(err);
  }
}
```

### 2. Manejo de Errores

```javascript
// Siempre capturar errores asíncronos
async function fetchData() {
  try {
    const response = await fetch(url);
    return await response.json();
  } catch (error) {
    console.error('Error al obtener datos:', error);
    throw error;
  }
}
```

### 3. Usar Variables de Entorno

```javascript
require('dotenv').config();

const port = process.env.PORT || 3000;
const dbUrl = process.env.DATABASE_URL;
```
