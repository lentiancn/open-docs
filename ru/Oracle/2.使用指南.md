# Руководство пользователя Oracle Database

В этом руководстве описано использование Oracle Database.

## Быстрый старт

### Подключение к базе данных

```bash
# Использование SQL*Plus
sqlplus / as sysdba

# Подключение с именем пользователя
sqlplus username/password@hostname:port/service_name
```

### Основные операции

```sql
-- Показать пользователя
SHOW USER;

-- Показать табличные пространства
SELECT tablespace_name FROM dba_tablespaces;

-- Создать табличное пространство
CREATE TABLESPACE mydata 
DATAFILE '/u01/app/oracle/oradata/orcl/mydata01.dbf' 
SIZE 100M 
AUTOEXTEND ON;

-- Создать пользователя
CREATE USER myuser IDENTIFIED BY password
DEFAULT TABLESPACE mydata;

-- Предоставить права
GRANT CONNECT, RESOURCE TO myuser;
```

## Часто используемый SQL

### Операции с таблицами

```sql
-- Создать таблицу
CREATE TABLE employees (
    employee_id NUMBER(10) PRIMARY KEY,
    first_name VARCHAR2(50),
    last_name VARCHAR2(50) NOT NULL,
    email VARCHAR2(100) UNIQUE,
    hire_date DATE DEFAULT SYSDATE,
    salary NUMBER(10,2)
);

-- Вставить данные
INSERT INTO employees (employee_id, first_name, last_name, email, salary)
VALUES (1, 'John', 'Doe', 'john@example.com', 5000);

-- Запрос данных
SELECT * FROM employees WHERE salary > 3000;

-- Обновление данных
UPDATE employees SET salary = salary * 1.1 WHERE employee_id = 1;

-- Удаление данных
DELETE FROM employees WHERE employee_id = 1;
```

### Ограничения

```sql
-- Первичный ключ
ALTER TABLE employees ADD CONSTRAINT pk_emp PRIMARY KEY (employee_id);

-- Внешний ключ
CREATE TABLE departments (
    dept_id NUMBER(10) PRIMARY KEY,
    dept_name VARCHAR2(50)
);

ALTER TABLE employees ADD CONSTRAINT fk_dept 
FOREIGN KEY (department_id) REFERENCES departments(dept_id);

-- Уникальное ограничение
ALTER TABLE employees ADD CONSTRAINT uk_email UNIQUE (email);

-- Проверочное ограничение
ALTER TABLE employees ADD CONSTRAINT chk_salary 
CHECK (salary > 0);
```

### Индексы

```sql
-- Создать индекс
CREATE INDEX idx_emp_name ON employees(last_name, first_name);

-- Создать уникальный индекс
CREATE UNIQUE INDEX idx_emp_email ON employees(email);

-- Удалить индекс
DROP INDEX idx_emp_name;
```

### Представления

```sql
-- Создать представление
CREATE VIEW emp_details AS
SELECT e.employee_id, e.first_name, e.last_name, d.dept_name
FROM employees e
JOIN departments d ON e.department_id = d.dept_id;

-- Использовать представление
SELECT * FROM emp_details;
```

### Хранимые процедуры

```sql
CREATE OR REPLACE PROCEDURE update_salary(
    p_emp_id IN NUMBER,
    p_new_salary IN NUMBER
) AS
BEGIN
    UPDATE employees 
    SET salary = p_new_salary 
    WHERE employee_id = p_emp_id;
    COMMIT;
END;
/

-- Вызов
BEGIN
    update_salary(1, 6000);
END;
/
```

## Резервное копирование и восстановление

### Резервное копирование

```sql
-- Экспорт
EXP username/password FILE=exp.dmp FULL=y

-- RMAN резервное копирование
RMAN TARGET /
BACKUP DATABASE PLUS ARCHIVELOG;
```

### Восстановление

```sql
-- Импорт
IMP username/password FILE=exp.dmp FULL=y

-- RMAN восстановление
RESTORE DATABASE;
RECOVER DATABASE;
```

## Управление пользователями

### Создание пользователя

```sql
CREATE USER username IDENTIFIED BY password
DEFAULT TABLESPACE USERS
TEMPORARY TABLESPACE TEMP
QUOTA UNLIMITED ON USERS;
```

### Предоставление прав

```sql
-- Системные привилегии
GRANT CREATE SESSION TO username;
GRANT CREATE TABLE TO username;
GRANT CREATE VIEW TO username;

-- Объектные привилегии
GRANT SELECT ON employees TO username;
GRANT UPDATE ON employees TO username;
```
