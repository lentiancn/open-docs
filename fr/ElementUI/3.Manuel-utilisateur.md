### Form avec validation

```html
<el-form :model="ruleForm" :rules="rules" ref="ruleForm" label-width="100px">
  <el-form-item label="Nom de l'activité" prop="name">
    <el-input v-model="ruleForm.name"></el-input>
  </el-form-item>
  <el-form-item label="Région" prop="region">
    <el-select v-model="ruleForm.region" placeholder="Veuillez sélectionner">
      <el-option label="Région 1" value="shanghai"></el-option>
      <el-option label="Région 2" value="beijing"></el-option>
    </el-select>
  </el-form-item>
  <el-form-item label="Date" prop="date1">
    <el-date-picker
      v-model="ruleForm.date1"
      type="date"
      placeholder="Sélectionner une date">
    </el-date-picker>
  </el-form-item>
  <el-form-item>
    <el-button type="primary" @click="submitForm('ruleForm')">Soumettre</el-button>
    <el-button @click="resetForm('ruleForm')">Réinitialiser</el-button>
  </el-form-item>
</el-form>
```

Règles de validation:

```javascript
export default {
  data() {
    return {
      ruleForm: {
        name: '',
        region: '',
        date1: ''
      },
      rules: {
        name: [
          { required: true, message: 'Veuillez entrer le nom', trigger: 'blur' },
          { min: 3, max: 5, message: 'Longueur entre 3 et 5', trigger: 'blur' }
        ],
        region: [
          { required: true, message: 'Veuillez sélectionner la région', trigger: 'change' }
        ],
        date1: [
          { required: true, message: 'Veuillez sélectionner une date', trigger: 'change' }
        ]
      }
    }
  },
  methods: {
    submitForm(formName) {
      this.$refs[formName].validate((valid) => {
        if (valid) {
          alert('soumettre!')
        } else {
          return false
        }
      })
    },
    resetForm(formName) {
      this.$refs[formName].resetFields()
    }
  }
}
```

### Validation de formulaire

```javascript
// Validateur personnalisé
const validatePass = (rule, value, callback) => {
  if (value === '') {
    callback(new Error('Veuillez entrer le mot de passe'))
  } else {
    if (this.ruleForm.checkPass !== '') {
      this.$refs.ruleForm.validateField('checkPass')
    }
    callback()
  }
}
```
