## Introduction

### What is Angular?

Angular is a TypeScript-based open-source web application framework developed by Google. It provides a comprehensive solution for building modern single-page applications (SPAs).

### Key Features

- **Component-Based Architecture**: Build applications with reusable components
- **Two-Way Data Binding**: Automatic synchronization between model and view
- **Dependency Injection**: Efficient service management
- **Routing**: Client-side navigation
- **Forms**: Robust form handling with validation
- **HTTP Client**: Communicate with backend services
- **Testing**: Built-in testing utilities
- **Cross-Platform**: Web, mobile web, native mobile, and desktop

## Project Structure

### Standard Directory Layout

```
my-app/
├── src/
│   ├── app/
│   │   ├── components/        # Reusable components
│   │   ├── pages/             # Page-level components
│   │   ├── services/          # Angular services
│   │   ├── models/            # Data models/interfaces
│   │   ├── guards/            # Route guards
│   │   ├── interceptors/     # HTTP interceptors
│   │   ├── pipes/             # Custom pipes
│   │   ├── directives/        # Custom directives
│   │   ├── app.component.ts  # Root component
│   │   ├── app.config.ts     # Application config (Standalone)
│   │   ├── app.routes.ts     # Application routes
│   │   └── main.ts           # Application entry point
│   ├── assets/               # Static files
│   ├── environments/         # Environment configs
│   ├── styles.css            # Global styles
│   └── index.html            # Main HTML file
├── angular.json              # Angular CLI config
├── package.json              # NPM dependencies
└── tsconfig.json             # TypeScript config
```

## Components

### Creating Components

```bash
# Using Angular CLI
ng generate component components/user-list
# or
ng g c components/user-list
```

### Component Example

```typescript
import { Component, Input, Output, EventEmitter } from '@angular/core';

@Component({
  selector: 'app-user-card',
  templateUrl: './user-card.component.html',
  styleUrls: ['./user-card.component.css']
})
export class UserCardComponent {
  @Input() user: User | null = null;
  @Output() delete = new EventEmitter<number>();

  onDelete() {
    if (this.user) {
      this.delete.emit(this.user.id);
    }
  }
}
```

```typescript
import { Component, OnInit, OnDestroy, AfterViewInit } from '@angular/core';

@Component({...})
export class ExampleComponent implements OnInit, OnDestroy, AfterViewInit {

  ngAfterViewInit() {
    // Called after component view is initialized
  }
```

## Templates

### Template Syntax

#### Interpolation
```html
<h1>Hello, {{ name }}</h1>
<p>Total: {{ 2 + 3 }}</p>
```

#### Property Binding
```html
<img [src]="imageUrl" [alt]="imageAlt">
<button [class.active]="isActive">Click</button>
```

#### Event Binding
```html
<button (click)="onClick()">Click Me</button>
<input (input)="onInput($event)">
```

#### Two-Way Binding
```html
<input [(ngModel)]="username" name="username">
```

### Structural Directives

#### *ngIf
```html
<div *ngIf="isLoggedIn">
  Welcome, {{ username }}!
</div>

<div *ngIf="isLoggedIn; else loggedOut">Welcome!</div>
<ng-template #loggedOut>Please log in.</ng-template>
```

#### *ngFor
```html
<li *ngFor="let user of users; let i = index">
  {{ i + 1 }}. {{ user.name }}
</li>

<!-- With trackBy for performance -->
<li *ngFor="let user of users; trackBy: trackById">
```

#### *ngSwitch
```html
<div [ngSwitch]="status">
  <p *ngSwitchCase="'pending'">Pending...</p>
  <p *ngSwitchCase="'approved'">Approved!</p>
  <p *ngSwitchDefault>Unknown status</p>
</div>
```

### Pipes

```html
<p>{{ today | date:'fullDate' }}</p>
<p>{{ price | currency:'USD' }}</p>
<p>{{ name | uppercase }}</p>
<p>{{ text | slice:0:10 }}</p>
```

### Custom Pipe

```typescript
import { Pipe, PipeTransform } from '@angular/core';

@Pipe({ name: 'truncate' })
export class TruncatePipe implements PipeTransform {
  transform(value: string, limit: number = 10): string {
    return value.length > limit ? value.substring(0, limit) + '...' : value;
  }
}
```

## Services

### Creating Services

```bash
ng generate service services/user
# or
ng g s services/user
```

### Service Example

```typescript
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { User } from '../models/user';

@Injectable({ providedIn: 'root' })
export class UserService {
  private apiUrl = 'https://api.example.com/users';

  constructor(private http: HttpClient) {}

  getUsers(): Observable<User[]> {
    return this.http.get<User[]>(this.apiUrl);
  }

  getUser(id: number): Observable<User> {
    return this.http.get<User>(`${this.apiUrl}/${id}`);
  }

  createUser(user: User): Observable<User> {
    return this.http.post<User>(this.apiUrl, user);
  }

  updateUser(id: number, user: User): Observable<User> {
    return this.http.put<User>(`${this.apiUrl}/${id}`, user);
  }

  deleteUser(id: number): Observable<void> {
    return this.http.delete<void>(`${this.apiUrl}/${id}`);
  }
}
```

## Routing

### Setting Up Routes

```typescript
// app.routes.ts
import { Routes } from '@angular/router';

export const routes: Routes = [
  { path: '', redirectTo: '/home', pathMatch: 'full' },
  { 
    path: 'home', 
    loadComponent: () => import('./pages/home/home.component')
      .then(m => m.HomeComponent) 
  },
  { 
    path: 'users', 
    loadComponent: () => import('./pages/users/users.component')
      .then(m => m.UsersComponent) 
  },
  { 
    path: '**', 
    loadComponent: () => import('./pages/not-found/not-found.component')
      .then(m => m.NotFoundComponent) 
  }
];
```

### Route Parameters

```typescript
// Route: { path: 'user/:id', ... }

// In component
import { ActivatedRoute } from '@angular/router';

constructor(private route: ActivatedRoute) {}

ngOnInit() {
  this.route.paramMap.subscribe(params => {
    const id = params.get('id');
  });
}
```

### Route Guards

```typescript
import { Injectable } from '@angular/core';
import { CanActivate, Router } from '@angular/router';

@Injectable({ providedIn: 'root' })
export class AuthGuard implements CanActivate {
  constructor(private router: Router) {}

  canActivate(): boolean {
    const isLoggedIn = /* check auth status */;
    if (!isLoggedIn) {
      this.router.navigate(['/login']);
      return false;
    }
    return true;
  }
}
```

### Lazy Loading

## Forms

### Template-Driven Forms

```typescript
import { Component } from '@angular/core';

@Component({
  selector: 'app-login',
  template: `
    <form #loginForm="ngForm" (ngSubmit)="onSubmit(loginForm)">
      <input type="email" [(ngModel)]="user.email" name="email" required email #email="ngModel">
      <input type="password" [(ngModel)]="user.password" name="password" required minlength="6">
      <button type="submit" [disabled]="loginForm.invalid">Login</button>
    </form>
  `
})
export class LoginComponent {
  user = { email: '', password: '' };

  onSubmit(form: any) {
    console.log('Form value:', form.value);
  }
}
```

### Reactive Forms

```typescript
import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';

@Component({
  selector: 'app-register',
  templateUrl: './register.component.html'
})
export class RegisterComponent implements OnInit {
  registerForm!: FormGroup;

  constructor(private fb: FormBuilder) {}

  ngOnInit() {
    this.registerForm = this.fb.group({
      username: ['', [Validators.required, Validators.minLength(3)]],
      email: ['', [Validators.required, Validators.email]],
      password: ['', [Validators.required, Validators.minLength(8)]]
    });
  }

  onSubmit() {
    if (this.registerForm.valid) {
      console.log(this.registerForm.value);
    }
  }
}
```

```html
<form [formGroup]="registerForm" (ngSubmit)="onSubmit()">
  <input formControlName="username">
  <div *ngIf="registerForm.get('username')?.invalid">Username required</div>
  <button type="submit" [disabled]="registerForm.invalid">Register</button>
</form>
```

## HTTP Client

### Setup

```typescript
// app.config.ts (Standalone)
import { ApplicationConfig } from '@angular/core';
import { provideHttpClient } from '@angular/common/http';

export const appConfig: ApplicationConfig = {
  providers: [provideHttpClient()]
};
```

### HTTP Methods

```typescript
constructor(private http: HttpClient) {}

// GET
this.http.get<User[]>(url).subscribe(data => console.log(data));

// POST
this.http.post<User>(url, user).subscribe(newUser => ...);

// PUT
this.http.put<User>(`${url}/${id}`, user).subscribe();

// DELETE
this.http.delete(`${url}/${id}`).subscribe();

// With Headers
this.http.get(url, { headers: new HttpHeaders({ 'Authorization': 'Bearer token' }) });
```

### HTTP Interceptor

```typescript
import { HttpInterceptorFn } from '@angular/common/http';

export const authInterceptor: HttpInterceptorFn = (req, next) => {
  const token = localStorage.getItem('token');
  const cloned = req.clone({
    setHeaders: { Authorization: token ? `Bearer ${token}` : '' }
  });
  return next(cloned);
};
```

## State Management

### Using Services with BehaviorSubject

```typescript
import { Injectable } from '@angular/core';
import { BehaviorSubject, Observable } from 'rxjs';

@Injectable({ providedIn: 'root' })
export class CartService {
  private items$ = new BehaviorSubject<CartItem[]>([]);
  
  getItems(): Observable<CartItem[]> {
    return this.items$.asObservable();
  }
  
  addItem(item: CartItem) {
    this.items$.next([...this.items$.value, item]);
  }
  
  removeItem(id: number) {
    this.items$.next(this.items$.value.filter(i => i.id !== id));
  }
}
```

### Using Signals (Angular 16+)

```typescript
import { Component, signal, computed } from '@angular/core';

@Component({...})
export class CounterComponent {
  count = signal(0);
  doubled = computed(() => this.count() * 2);
  
  increment() {
    this.count.update(c => c + 1);
  }
}
```

## Testing

### Unit Testing

```typescript
import { ComponentFixture, TestBed } from '@angular/core/testing';
import { UserListComponent } from './user-list.component';
import { UserService } from '../services/user.service';
import { of } from 'rxjs';

describe('UserListComponent', () => {
  let component: UserListComponent;
  let fixture: ComponentFixture<UserListComponent>;

  beforeEach(async () => {
    await TestBed.configureTestingModule({
      declarations: [UserListComponent],
      providers: [
        { provide: UserService, useValue: jasmine.createSpyObj('UserService', ['getUsers']) }
      ]
    }).compileComponents();

    fixture = TestBed.createComponent(UserListComponent);
    component = fixture.componentInstance;
  });

  it('should create', () => {
    expect(component).toBeTruthy();
  });

  it('should load users', () => {
    const userService = TestBed.inject(UserService);
    userService.getUsers.and.returnValue(of([{ id: 1, name: 'John' }]));
    
    component.ngOnInit();
    
    expect(component.users.length).toBe(1);
  });
});
```

## Build and Deploy

### Development Build
```bash
ng build

```

### Production Build
```bash
ng build --configuration production
# or
ng build -c production
```

### Deploy Options

#### Netlify
```bash
npm install -g netlify-cli
netlify deploy --prod --dir=dist/your-app
```

#### Vercel
```bash
npm i -g vercel
vercel --prod
```

#### Docker
```dockerfile
FROM node:18-alpine as build
WORKDIR /app
COPY package*.json ./
RUN npm ci
COPY . .
RUN npm run build -- --configuration production

FROM nginx:alpine
COPY --from=build /app/dist/your-app/browser /usr/share/nginx/html
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
```

## Best Practices

### Project Structure
```
src/app/
├── core/           # Singleton services, guards
├── shared/         # Reusable components, pipes, directives
├── features/       # Feature modules
└── app.component.ts
```

### Performance Tips
1. Use OnPush change detection
2. Implement trackBy in *ngFor
3. Lazy load routes
4. Use signals for state
5. Unsubscribe from observables (or use async pipe)

### Code Style
- Use strict TypeScript
- Follow Angular style guide
- Use standalone components
- Keep components small and focused
