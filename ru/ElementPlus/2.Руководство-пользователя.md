## 1. Основы компонентов

### 1.1 Основная кнопка

Кнопки — один из наиболее часто используемых компонентов. Element Plus предоставляет несколько типов кнопок.

```vue
<template>
  <!-- Основная кнопка -->
  <el-button type="primary">Основная кнопка</el-button>
  
  <!-- Кнопка успеха -->
  <el-button type="success">Кнопка успеха</el-button>
  
  <!-- Информационная кнопка -->
  <el-button type="info">Информационная кнопка</el-button>
  
  <!-- Предупреждающая кнопка -->
  <el-button type="warning">Предупреждающая кнопка</el-button>
  
  <!-- Кнопка опасности -->
  <el-button type="danger">Кнопка опасности</el-button>
  
  <!-- Текстовая кнопка -->
  <el-button type="text">Текстовая кнопка</el-button>
  
  <!-- Простая кнопка -->
  <el-button plain>Простая кнопка</el-button>
  
  <!-- Круглая кнопка -->
  <el-button round>Круглая кнопка</el-button>
  
  <!-- Кнопка с иконкой -->
  <el-button icon="el-icon-search">Поиск</el-button>
</template>
```

### 1.2 Атрибуты кнопки

| Атрибут | Описание | Тип | Допустимые значения | По умолчанию |
|------|------|------|-----------|----------|
| type | Тип кнопки | string | primary / success / warning / danger / info / text | — |
| plain | Простая кнопка | boolean | — | false |
| round | Круглая кнопка | boolean | — | false |
| circle | Круглая кнопка | boolean | — | false |
| loading | Загрузка | boolean | — | false |
| disabled | Отключена | boolean | — | false |
| icon | Класс иконки | string | — | — |
| size | Размер | string | large / default / small | — |

### 1.3 События кнопки

```vue
<template>
  <el-button @click="handleClick">Событие клика</el-button>
</template>

<script setup>
const handleClick = () => {
  console.log('Кнопка нажата')
}
</script>
```

## 2. Компоненты формы

### 2.1 Базовая форма

```vue
<template>
  <el-form :model="form" :rules="rules" ref="formRef" label-width="100px">
    <el-form-item label="Имя пользователя" prop="username">
      <el-input v-model="form.username" placeholder="Введите имя пользователя" />
    </el-form-item>
    
    <el-form-item label="Пароль" prop="password">
      <el-input v-model="form.password" type="password" placeholder="Введите пароль" show-password />
    </el-form-item>
    
    <el-form-item label="Email" prop="email">
      <el-input v-model="form.email" placeholder="Введите email" />
    </el-form-item>
    
    <el-form-item label="Пол" prop="gender">
      <el-radio-group v-model="form.gender">
        <el-radio label="male">Мужской</el-radio>
        <el-radio label="female">Женский</el-radio>
      </el-radio-group>
    </el-form-item>
    
    <el-form-item label="Хобби" prop="hobby">
      <el-checkbox-group v-model="form.hobby">
        <el-checkbox label="Баскетбол" />
        <el-checkbox label="Футбол" />
        <el-checkbox label="Плавание" />
      </el-checkbox-group>
    </el-form-item>
    
    <el-form-item label="О себе" prop="bio">
      <el-input v-model="form.bio" type="textarea" :rows="3" placeholder="Введите о себе" />
    </el-form-item>
    
    <el-form-item>
      <el-button type="primary" @click="submitForm">Отправить</el-button>
      <el-button @click="resetForm">Сбросить</el-button>
    </el-form-item>
  </el-form>
</template>

<script setup>
import { reactive, ref } from 'vue'

const formRef = ref(null)

const form = reactive({
  username: '',
  password: '',
  email: '',
  gender: '',
  hobby: [],
  bio: ''
})

const rules = {
  username: [
    { required: true, message: 'Пожалуйста, введите имя пользователя', trigger: 'blur' },
    { min: 3, max: 20, message: 'Имя пользователя от 3 до 20 символов', trigger: 'blur' }
  ],
  password: [
    { required: true, message: 'Пожалуйста, введите пароль', trigger: 'blur' },
    { min: 6, message: 'Пароль должен быть не менее 6 символов', trigger: 'blur' }
  ],
  email: [
    { required: true, message: 'Пожалуйста, введите email', trigger: 'blur' },
    { type: 'email', message: 'Пожалуйста, введите корректный email', trigger: 'blur' }
  ],
  gender: [
    { required: true, message: 'Пожалуйста, выберите пол', trigger: 'change' }
  ]
}

const submitForm = async () => {
  if (!formRef.value) return
  
  await formRef.value.validate((valid) => {
    if (valid) {
      console.log('Данные формы:', form)
      ElMessage.success('Успешно отправлено')
    } else {
      ElMessage.error('Пожалуйста, проверьте форму')
    }
  })
}

const resetForm = () => {
  formRef.value?.resetFields()
}
</script>
```

## 3. Компонент таблицы

### 3.1 Базовая таблица

```vue
<template>
  <el-table :data="tableData" stripe border>
    <el-table-column prop="date" label="Дата" width="180" />
    <el-table-column prop="name" label="Имя" width="180" />
    <el-table-column prop="address" label="Адрес" />
    <el-table-column label="Действия" width="200">
      <template #default="{ row }">
        <el-button type="primary" size="small" @click="handleEdit(row)">Редактировать</el-button>
        <el-button type="danger" size="small" @click="handleDelete(row)">Удалить</el-button>
      </template>
    </el-table-column>
  </el-table>
</template>

<script setup>
import { ref } from 'vue'

const tableData = ref([

  { date: '2024-01-02', name: 'Пётр Петров', address: 'Санкт-Петербург' },
  { date: '2024-01-03', name: 'Алексей Сидоров', address: 'Новосибирск' }
])

const handleEdit = (row) => {
  console.log('Редактировать:', row)
}

const handleDelete = (row) => {
  console.log('Удалить:', row)
}
</script>
```

### 3.2 Таблица с пагинацией

```vue
<template>
  <div>
    <el-table :data="tableData" v-loading="loading">
      <el-table-column prop="id" label="ID" width="80" />
      <el-table-column prop="title" label="Заголовок" />
      <el-table-column prop="author" label="Автор" width="120" />
      <el-table-column prop="date" label="Дата публикации" width="180" />
    </el-table>
    
    <el-pagination
      v-model:current-page="currentPage"
      v-model:page-size="pageSize"
      :total="total"
      :page-sizes="[10, 20, 50, 100]"
      layout="total, sizes, prev, pager, next, jumper"
      @size-change="handleSizeChange"
      @current-change="handleCurrentChange"
    />
  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue'

const tableData = ref([])
const loading = ref(false)
const currentPage = ref(1)
const pageSize = ref(10)
const total = ref(0)

const loadData = async () => {
  loading.value = true
  // Имитация вызова API
  setTimeout(() => {
    tableData.value = [
      { id: 1, title: 'Заголовок статьи 1', author: 'Автор 1', date: '2024-01-01' },
      { id: 2, title: 'Заголовок статьи 2', author: 'Автор 2', date: '2024-01-02' }
    ]
    total.value = 100
    loading.value = false
  }, 500)
}

const handleSizeChange = (val) => {
  pageSize.value = val
  loadData()
}

const handleCurrentChange = (val) => {
  currentPage.value = val
  loadData()
}

onMounted(() => {
  loadData()
})
</script>
```

## 4. Диалог и выдвижная панель

### 4.1 Диалог

```vue
<template>
  <el-button type="primary" @click="dialogVisible = true">Открыть диалог</el-button>
  
  <el-dialog
    v-model="dialogVisible"
    title="Редактирование информации"
    width="500px"
    :close-on-click-modal="false"
    @close="handleClose"
  >
    <el-form :model="form" label-width="80px">
      <el-form-item label="Имя">
        <el-input v-model="form.name" />
      </el-form-item>
      <el-form-item label="Описание">
        <el-input v-model="form.desc" type="textarea" :rows="3" />
      </el-form-item>
    </el-form>
    
    <template #footer>
      <el-button @click="dialogVisible = false">Отмена</el-button>
      <el-button type="primary" @click="handleConfirm">Подтвердить</el-button>
    </template>
  </el-dialog>
</template>

<script setup>
import { ref, reactive } from 'vue'

const dialogVisible = ref(false)

const form = reactive({
  name: '',
  desc: ''
})

const handleClose = () => {
  form.name = ''
  form.desc = ''
}

const handleConfirm = () => {
  console.log('Отправить:', form)
  dialogVisible.value = false
  ElMessage.success('Успешно')
}
</script>
```

## 5. Уведомления

### 5.1 Глобальные сообщения

```vue
<script setup>
import { ElMessage } from 'element-plus'

// Сообщение об успехе
const showSuccess = () => {
  ElMessage.success('Операция успешна')
}

// Предупреждающее сообщение
const showWarning = () => {
  ElMessage.warning('Предупреждение')
}

// Сообщение об ошибке
const showError = () => {
  ElMessage.error('Операция не удалась')
}

// Информационное сообщение
const showInfo = () => {
  ElMessage.info('Информационное сообщение')
}
</script>
```

### 5.2 Уведомления

```vue
<script setup>
import { ElNotification } from 'element-plus'

// Уведомление об успехе
const notifySuccess = () => {
  ElNotification({
    title: 'Успех',
    message: 'Операция успешно завершена',
    type: 'success',
  })
}

// Уведомление об ошибке
const notifyError = () => {
  ElNotification({
    title: 'Ошибка',
    message: 'Операция не удалась',
    type: 'error',
  })
}
</script>
```
