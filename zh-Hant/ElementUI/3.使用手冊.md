# Element UI 使用手冊

Element UI 核心元件的使用方法和示例。

## 基礎元件

### Button 按鈕

```html
<el-button>預設</el-button>
<el-button type="primary">主要</el-button>
<el-button type="success">成功</el-button>
<el-button type="warning">警告</el-button>
<el-button type="danger">危險</el-button>
<el-button type="info">資訊</el-button>
<el-button plain>樸素</el-button>
<el-button round>圓角</el-button>
<el-button disabled>停用</el-button>
```

帶圖標：

```html
<el-button icon="el-icon-search" circle></el-button>
<el-button type="primary" icon="el-icon-edit">編輯</el-button>
<el-button type="primary" icon="el-icon-share">分享</el-button>
<el-button type="primary" icon="el-icon-delete">刪除</el-button>
```

按鈕組：

```html
<el-button-group>
  <el-button type="primary" icon="el-icon-arrow-left">上一頁</el-button>
  <el-button type="primary">下一頁<i class="el-icon-arrow-right el-icon--right"></i></el-button>
</el-button-group>
```

### Layout 佈局（24欄柵格系統）

```html
<el-row>
  <el-col :span="24"><div class="grid-content bg-purple-dark"></div></el-col>
</el-row>
<el-row>
  <el-col :span="12"><div class="grid-content bg-purple"></div></el-col>
  <el-col :span="12"><div class="grid-content bg-light-purple"></div></el-col>
</el-row>
<el-row :gutter="20">
  <el-col :span="6"><div class="grid-content bg-purple"></div></el-col>
  <el-col :span="6"><div class="grid-content bg-purple"></div></el-col>
  <el-col :span="6"><div class="grid-content bg-purple"></div></el-col>
  <el-col :span="6"><div class="grid-content bg-purple"></div></el-col>
</el-row>
```

響應式佈局：

```html
<el-row :gutter="10">
  <el-col :xs="8" :sm="6" :md="4" :lg="3" :xl="1">
    <div class="grid-content"></div>
  </el-col>
</el-row>
```

### Container 容器

```html
<el-container>
  <el-header>Header</el-header>
  <el-container>
    <el-aside width="200px">Aside</el-aside>
    <el-container>
      <el-main>Main</el-main>
```

## 表單元件

### Input 輸入框

基本用法：

```html
<el-input v-model="input" placeholder="請輸入內容"></el-input>
```

帶圖標：

```html
<el-input placeholder="請輸入" v-model="input" prefix-icon="el-icon-search"></el-input>
```

文字域：

```html
<el-input type="textarea" :rows="3" placeholder="請輸入" v-model="textarea"></el-input>
```

### Select 選擇器

基本用法：

```html
<el-select v-model="value" placeholder="請選擇">
  <el-option v-for="item in options" :key="item.value" :label="item.label" :value="item.value"></el-option>
</el-select>
```

可過濾：

```html
<el-select v-model="value" filterable placeholder="請選擇"></el-select>
```

多選：

```html
<el-select v-model="value" multiple placeholder="請選擇"></el-select>
```

### Form 表單

```html
<el-form :model="ruleForm" :rules="rules" ref="ruleForm" label-width="100px">
  <el-form-item label="活動名稱" prop="name">
    <el-input v-model="ruleForm.name"></el-input>
  </el-form-item>
  <el-form-item label="活動區域" prop="region">
    <el-select v-model="ruleForm.region" placeholder="請選擇">
      <el-option label="區域一" value="shanghai"></el-option>
      <el-option label="區域二" value="beijing"></el-option>
    </el-select>
  </el-form-item>
  <el-form-item label="活動日期" prop="date1">
    <el-date-picker v-model="ruleForm.date1" type="date" placeholder="選擇日期"></el-date-picker>
  </el-form-item>
  <el-form-item>
    <el-button type="primary" @click="submitForm('ruleForm')">提交</el-button>
    <el-button @click="resetForm('ruleForm')">重置</el-button>
  </el-form-item>
</el-form>
```

## 資料元件

### Table 表格

基本用法：

```html
<el-table :data="tableData" style="width: 100%">
  <el-table-column prop="date" label="日期" width="180"></el-table-column>
  <el-table-column prop="name" label="姓名" width="180"></el-table-column>
  <el-table-column prop="address" label="地址"></el-table-column>
</el-table>
```

帶斑馬紋：

```html
<el-table :data="tableData" stripe style="width: 100%"></el-table>
```

帶邊框：

```html
<el-table :data="tableData" border style="width: 100%"></el-table>
```

固定列和表頭：

```html
<el-table :data="tableData" height="250" border>
  <el-table-column fixed prop="date" label="日期" width="150"></el-table-column>
  <el-table-column prop="name" label="姓名" width="120"></el-table-column>
  <el-table-column prop="province" label="省份" width="120"></el-table-column>
  <el-table-column prop="city" label="市區" width="120"></el-table-column>
  <el-table-column prop="address" label="地址" width="300"></el-table-column>
  <el-table-column prop="zip" label="郵編" width="120"></el-table-column>
</el-table>
```

### Pagination 分頁

```html
<el-pagination background layout="prev, pager, next" :total="1000"></el-pagination>
```

### Tree 樹形控件

```html
<el-tree :data="data" :props="defaultProps" @node-click="handleNodeClick"></el-tree>
```

## 通知元件

### Message 消息提示

```javascript
this.$message({ message: '恭喜你，這是一條成功消息', type: 'success' })
this.$message({ message: '警告哦，這是一條警告消息', type: 'warning' })
this.$message.error('錯了哦，這是一條錯誤消息')
```

### Notification 通知

```javascript
this.$notify({ title: '標題名稱', message: '這是通知的內容' })
```

### Alert 警告

```html
<el-alert title="成功提示的文案" type="success"></el-alert>
<el-alert title="成功提示的文案" type="success" description="這是成功提示的描述文字"></el-alert>
```

```html
  <el-menu-item index="1"><i class="el-icon-menu"></i><span slot="title">導航一</span></el-menu-item>
  <el-menu-item index="2"><i class="el-icon-document"></i><span slot="title">導航二</span></el-menu-item>
  <el-menu-item index="3"><i class="el-icon-setting"></i><span slot="title">導航三</span></el-menu-item>
</el-menu>
```

```html
<el-tabs v-model="activeName" @tab-click="handleClick">

  <el-tab-pane label="定時任務" name="fourth">定時任務</el-tab-pane>
</el-tabs>
```

## 其他元件

### Dialog 對話框

```html
<el-button type="text" @click="dialogVisible = true">點擊打開 Dialog</el-button>
<el-dialog title="提示" :visible.sync="dialogVisible" width="30%" :before-close="handleClose">
  <span>這是一段訊息</span>
  <span slot="footer">
    <el-button @click="dialogVisible = false">取消</el-button>
    <el-button type="primary" @click="dialogVisible = false">確定</el-button>
  </span>
</el-dialog>
```

### Drawer 抽屜

```html
<el-button type="primary" @click="drawer = true">打開</el-button>
<el-drawer title="我是標題" :visible.sync="drawer" direction="rtl">
  <div>我是抽屜內容</div>
</el-drawer>
```

### Upload 上傳

```html
<el-upload class="upload-demo" action="https://jsonplaceholder.typicode.com/posts/" :on-preview="handlePreview" :on-remove="handleRemove" :file-list="fileList" list-type="picture">
  <el-button size="small" type="primary">點擊上傳</el-button>
  <div slot="tip" class="el-upload__tip">只能上傳jpg/png文件，且不超過500kb</div>
</el-upload>
```
