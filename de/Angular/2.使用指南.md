# Angular Benutzerhandbuch

Diese Anleitung zeigt Ihnen, wie Sie moderne Web-Anwendungen mit Angular erstellen.

## Schnellstart

### Komponente erstellen

```bash
ng generate component my-component
# oder Kurzform
ng g c my-component
```

### Komponenten-Grundlagen

```typescript
// my-component.component.ts
import { Component } from '@angular/core';

@Component({
  selector: 'app-my-component',
  templateUrl: './my-component.component.html',
  styleUrls: ['./my-component.component.css']
})
export class MyComponentComponent {
  title = 'My Angular App';
  
  // Eigenschaft
  count = 0;
  
  // Methode
  increment() {
    this.count++;
  }
}
```

```html
<!-- my-component.component.html -->
<h1>{{ title }}</h1>
<button (click)="increment()">Klicks: {{ count }}</button>
```

## Kernkonzepte

### Module

```typescript
// app.module.ts
import { NgModule } from '@angular/core';
import { BrowserModule } from '@angular/platform-browser';
import { FormsModule } from '@angular/forms';

import { AppComponent } from './app.component';

@NgModule({
  declarations: [
    AppComponent
  ],
  imports: [
    BrowserModule,
    FormsModule,
    HttpClientModule
  ],
  providers: [],
  bootstrap: [AppComponent]
})
export class AppModule { }
```

### Komponenten

```typescript
import { Component, OnInit, OnDestroy } from '@angular/core';

@Component({
  selector: 'app-root',
  template: `
    <h1>{{ title }}</h1>
    <button (click)="onClick()">Klick</button>
  `
})
export class AppComponent implements OnInit, OnDestroy {
  title = 'My App';
  
  ngOnInit() {
    console.log('Komponente initialisiert');
  }
  
  ngOnDestroy() {
    console.log('Komponente zerstört');
  }
  
  onClick() {
    console.log('Button geklickt');
  }
}
```

### Dienste

```typescript
// my-service.service.ts
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';

@Injectable({
  providedIn: 'root'
})
export class MyServiceService {
  constructor(private http: HttpClient) { }
  
  getData(): Observable<any> {
    return this.http.get('https://api.example.com/data');
  }
  
  postData(data: any): Observable<any> {
    return this.http.post('https://api.example.com/data', data);
  }
}
```

## Häufig verwendete Direktiven

### *ngIf

```html
<div *ngIf="isVisible">Inhalt</div>

<div *ngIf="isLoggedIn; else loginTemplate">
  Willkommen zurück!
</div>
<ng-template #loginTemplate>
  <button>Anmelden</button>
</ng-template>
```

### *ngFor

```ul>
  <li *ngFor="let item of items; let i = index">
    {{ i + 1 }}. {{ item.name }}
  </li>
</ul>

<!-- Mit Bedingung -->
<li *ngFor="let item of items; trackBy: trackById">
  {{ item.name }}
</li>
```

### [ngClass]

```html
<div [ngClass]="{'active': isActive, 'disabled': isDisabled}">
  Inhalt
</div>
```

### [(ngModel)]

```html
<input [(ngModel)]="username" placeholder="Benutzername">
<p>Hallo, {{ username }}!</p>
```

## Formularverarbeitung

### Reaktive Formulare

```typescript
import { Component } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';

@Component({
  selector: 'app-reactive-form',
  template: `
    <form [formGroup]="form" (ngSubmit)="onSubmit()">
      <input formControlName="email" placeholder="E-Mail">
      <div *ngIf="form.get('email')?.invalid">Bitte geben Sie eine gültige E-Mail ein</div>
      <button type="submit" [disabled]="form.invalid">Absenden</button>
    </form>
  `
})
export class ReactiveFormComponent {
  form: FormGroup;
  
  constructor(private fb: FormBuilder) {
    this.form = this.fb.group({
      email: ['', [Validators.required, Validators.email]],
      password: ['', [Validators.required, Validators.minLength(6)]]
    });
  }
  
  onSubmit() {
    console.log(this.form.value);
  }
}
```

### Vorlagengesteuerte Formulare

```html
<form #f="ngForm" (ngSubmit)="onSubmit(f)">
  <input name="username" [(ngModel)]="username" required #name="ngModel">
  <div *ngIf="name.invalid">Name ist erforderlich</div>
  <button type="submit" [disabled]="f.invalid">Absenden</button>
</form>
```

## HTTP-Anfragen

```typescript
import { Component, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';

@Component({
  selector: 'app-data-list',
  template: `
    <ul>
      <li *ngFor="let item of data">{{ item.name }}</li>
    </ul>
  `
})
export class DataListComponent implements OnInit {
  data: any[] = [];
  
  constructor(private http: HttpClient) { }
  
  ngOnInit() {
    this.http.get<any[]>('https://api.example.com/items')
      .subscribe(response => {
        this.data = response;
      });
  }
}
```

## Lebenszyklus-Hooks

| Hook | Beschreibung |
|------|--------------|
| ngOnInit | Wird nach der Komponenteninitialisierung aufgerufen |
| ngOnChanges | Wird aufgerufen, wenn sich Eingabeeigenschaften ändern |
| ngDoCheck | Wird während der Änderungserkennung aufgerufen |
| ngAfterViewInit | Wird nach der Ansichtsinitialisierung aufgerufen |
| ngOnDestroy | Wird vor der Komponentenzerstörung aufgerufen |

## Statusverwaltung

### Signal verwenden (Angular 16+)

```typescript
import { Component, signal, computed, effect } from '@angular/core';

@Component({
  selector: 'app-counter',
  template: `
    <p>Zähler: {{ count() }}</p>
    <p>Doppelt: {{ doubleCount() }}</p>
    <button (click)="increment()">Erhöhen</button>
  `
})
export class CounterComponent {
  count = signal(0);
  
  doubleCount = computed(() => this.count() * 2);
  
  constructor() {
    effect(() => {
      console.log('Zähler geändert:', this.count());
    });
  }
  
  increment() {
    this.count.update(c => c + 1);
  }
}
```

## Best Practices

### 1. Standalone-Komponenten verwenden

```typescript
// Empfohlen in Angular 14+
@Component({
  selector: 'app-standalone',
  standalone: true,
  imports: [CommonModule, FormsModule],
  template: `...`
})
export class StandaloneComponent { }
```

### 2. Lazy Loading von Modulen

```typescript
const routes: Routes = [
  {
    path: 'admin',
    loadChildren: () => import('./admin/admin.module').then(m => m.AdminModule)
  }
];
```

### 3. OnPush-Änderungserkennung verwenden

```typescript
@Component({
  selector: 'app-optimized',
  changeDetection: ChangeDetectionStrategy.OnPush,
  template: `...`
})
export class OptimizedComponent { }
```
