# React User Manual

## JSX Basics

JSX is a syntax extension that lets you write HTML-like code in JavaScript.

```jsx
function App() {
  return (
    <div>
      <h1>Hello World!</h1>
      <p>Welcome to React</p>
    </div>
  );
}
```

---

## Components

### Functional Components

```jsx
function Welcome(props) {
  return <h1>Hello, {props.name}!</h1>;
}

function App() {
  return <Welcome name="Alice" />;
}
```

### Components with State (Hooks)

```jsx
import { useState } from 'react';

function Counter() {
  const [count, setCount] = useState(0);

  return (
    <div>
      <p>Count: {count}</p>
      <button onClick={() => setCount(count + 1)}>
        Increment
      </button>
    </div>
  );
}
```

## Props

Pass data from parent to child components:

```jsx
// Parent
function App() {
  return <User name="Bob" age={30} />;
}

// Child
function User(props) {
  return (
    <div>
      <h2>{props.name}</h2>
      <p>Age: {props.age}</p>
    </div>
  );
}
```

## State Management

### useState for Local State

```jsx
import { useState } from 'react';

function Form() {
  const [name, setName] = useState('');

  return (
    <input
      value={name}
      onChange={(e) => setName(e.target.value)}
    />
  );
}
```

### useEffect for Side Effects

```jsx
import { useEffect, useState } from 'react';

function DataFetcher() {
  const [data, setData] = useState(null);

  useEffect(() => {
    fetch('https://api.example.com/data')
      .then(res => res.json())
      .then(setData);
  }, []);

  return <div>{data}</div>;
}
```

## Conditional Rendering

```jsx
function Greeting({ isLoggedIn }) {
  if (isLoggedIn) {
    return <h1>Welcome back!</h1>;
  }
  return <h1>Please log in.</h1>;
}

// Using ternary
function Greeting({ isLoggedIn }) {
  return (
    <div>
      {isLoggedIn ? <h1>Welcome!</h1> : <h1>Login</h1>}
    </div>
  );
}
```

## Lists and Keys

```jsx
function ItemList({ items }) {
  return (
    <ul>
      {items.map((item) => (
        <li key={item.id}>{item.name}</li>
      ))}
    </ul>
  );
}
```

## Styling

### Inline Styles

```jsx
function Button() {
  const style = {
    backgroundColor: 'blue',
    color: 'white',
    padding: '10px 20px',
  };
  return <button style={style}>Click</button>;
}
```

### CSS Modules

```jsx
// Button.module.css
.button {
  background: blue;
}

// Button.jsx
import styles from './Button.module.css';

function Button() {
  return <button className={styles.button}>Click</button>;
}
```

## Event Handling

```jsx
function Form() {
  function handleSubmit(e) {
    e.preventDefault();
    console.log('Form submitted!');
  }

  return (
    <form onSubmit={handleSubmit}>
      <button type="submit">Submit</button>
    </form>
  );
}
```
