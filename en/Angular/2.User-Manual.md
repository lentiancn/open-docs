## 1. Introduction to Angular

### 1.1 What is Angular?

Angular is an open-source web application framework maintained by Google. It is built with TypeScript and features:

- **Component-based**: Component-based architecture
- **Two-way Data Binding**: Automatically sync template and data
- **Dependency Injection**: Powerful dependency management system
- **Routing**: Complete client-side routing solution
- **Forms**: Powerful form validation features
- **HTTP**: Concise HTTP client
- **Testing**: Complete testing toolchain

### 1.2 Angular Version History

| Version | Release Date | Key Features |
|---------|--------------|--------------|
| Angular 17 | November 2023 | Standalone Components, Signal, if/for block syntax |
| Angular 16 | May 2023 | Signal, Experimental SSR |
| Angular 15 | November 2022 | Standalone Components API Stabilized |
| Angular 14 | June 2022 | Strongly Typed Forms |
| Angular 13 | November 2021 | View Engine Removed |
| Angular 2 | September 2016 | New Architecture |

> üí° **Tip**: This manual is written based on Angular 17+, using **Standalone Components** as the recommended approach.

## 3. Core Concepts

### 3.1 Modules

Modules are the basic building blocks of Angular applications, used to organize code.

#### Creating a Module

```bash

```

#### Module Basic Structure

```typescript

import { MyComponentComponent } from './my-component/my-component.component';

  ],

  ]
```

### 3.2 Components

Components are the core UI units of Angular applications.

#### Creating a Component

```bash
# Create component
ng generate component my-component

# Create standalone component (Angular 17+ recommended)
ng generate component my-component --standalone
```

#### Component Basic Structure

```typescript
import { Component } from '@angular/core';

@Component({
  selector: 'app-my-component',
  standalone: true,  // Standalone component

  // Component logic
  title: string = 'Hello Angular';
  
  // Method

    console.log('Hello!');
  }
```

### 3.3 Template Syntax

#### Interpolation Binding

```html
<!-- Variable binding -->

<!-- Evaluate expression -->
<p>{{ 1 + 1 }}</p>

<!-- Call method -->
```


#### Event Binding


### 3.4 Directives

#### Built-in Directives

```html
<!-- Conditional rendering -->
<div *ngIf="isVisible">Content</div>

<!-- Branching -->
<div *ngIf="isLoggedIn; else loginTemplate">
  Welcome back!
</div>
<ng-template #loginTemplate>
  Please login
</ng-template>

<!-- Loop rendering -->
<li *ngFor="let item of items; let i = index">
  {{ i + 1 }}. {{ item.name }}
</li>

<!-- Class toggle -->
<div [ngClass]="{'active': isActive, 'disabled': isDisabled}">

<!-- Style toggle -->
<div [ngStyle]="{'color': textColor, 'font-size': fontSize + 'px'}">
```

#### Angular 17+ Control Flow Syntax

```html
<!-- @if replaces *ngIf -->
@if (isVisible) {
  <p>Content</p>
} @else {
  <p>Other content</p>
}

<!-- @for replaces *ngFor -->
@for (item of items; track item.id) {
  <li>{{ item.name }}</li>
} @empty {
  <li>No data</li>
}

<!-- @switch -->
@switch (type) {
  @case ('A') { <p>Type A</p> }
  @case ('B') { <p>Type B</p> }
  @default { <p>Other type</p> }
}
```

## 4. Component Development

### 4.1 Component Communication

#### Parent to Child (Input Properties)

```typescript
// Child component
import { Component, Input } from '@angular/core';

@Component({
  selector: 'app-child',
  standalone: true,
  template: `<p>{{ message }}</p>`
})
export class ChildComponent {
  @Input() message: string = '';
  @Input() count: number = 0;
}
```

```html
<!-- Parent component template -->
<app-child message="Hello" [count]="5"></app-child>
```

#### Child to Parent (Output Properties)

```typescript
// Child component
import { Component, Output, EventEmitter } from '@angular/core';

@Component({
  selector: 'app-child',
  standalone: true,
  template: `<button (click)="sendEvent()">Send</button>`
})
export class ChildComponent {
  @Output() messageEvent = new EventEmitter<string>();
  
  sendEvent(): void {
    this.messageEvent.emit('Message from child component');
  }
}
```

```typescript
// Parent component
@Component({
  standalone: true,
  imports: [ChildComponent],
  template: `<app-child (messageEvent)="onMessage($event)"></app-child>`
})
export class ParentComponent {
  onMessage(message: string): void {
    console.log(message);
  }
}
```

#### Using ViewChild to Access Child Components

```typescript
import { Component, ViewChild, AfterViewInit } from '@angular/core';
import { ChildComponent } from './child.component';

@Component({
  standalone: true,
  imports: [ChildComponent],
  template: `<app-child #childRef></app-child>`
})
export class ParentComponent implements AfterViewInit {
  @ViewChild('childRef') childComponent!: ChildComponent;
  
  ngAfterViewInit(): void {
    // Access child component's properties and methods
    this.childComponent.someMethod();
  }
}
```

## 5. Services and Dependency Injection

### 5.1 Creating a Service

```bash
ng generate service services/data
```

### 5.2 Service Basic Structure

```typescript
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';

@Injectable({
  providedIn: 'root'  // Global singleton
})
export class DataService {
  private apiUrl = 'https://api.example.com';
  
  constructor(private http: HttpClient) { }
  
  // GET request
  getData(): Observable<any> {
    return this.http.get(this.apiUrl);
  }
  
  // POST request
  postData(data: any): Observable<any> {
    return this.http.post(this.apiUrl, data);
  }
}
```

### 5.3 Using Services in Components

```typescript
import { Component, OnInit } from '@angular/core';
import { DataService } from './services/data.service';

@Component({
  selector: 'app-my',
  standalone: true,
  template: `<p>{{ data | json }}</p>`
})
export class MyComponent implements OnInit {
  data: any = null;
  
  constructor(private dataService: DataService) { }
  
  ngOnInit(): void {
    this.dataService.getData().subscribe({
      next: (result) => this.data = result,
      error: (err) => console.error(err)
    });
  }
}
```

## 7. Form Handling

### 7.1 Reactive Forms

```typescript
import { Component } from '@angular/core';
import { FormBuilder, FormGroup, Validators, ReactiveFormsModule } from '@angular/forms';

@Component({
  selector: 'app-login',
  standalone: true,
  imports: [ReactiveFormsModule],
  template: `
    <form [formGroup]="loginForm" (ngSubmit)="onSubmit()">
      <div>
        <label>Email</label>
        <input type="email" formControlName="email">
        <span *ngIf="loginForm.get('email')?.invalid">
          Please enter a valid email
        </span>
      </div>
      <div>
        <label>Password</label>
        <input type="password" formControlName="password">
      </div>
      <button type="submit" [disabled]="loginForm.invalid">Login</button>
    </form>
  `
})
export class LoginComponent {
  loginForm: FormGroup;
  
  constructor(private fb: FormBuilder) {
    this.loginForm = this.fb.group({
      email: ['', [Validators.required, Validators.email]],
      password: ['', [Validators.required, Validators.minLength(6)]]
    });
  }
  
  onSubmit(): void {
    if (this.loginForm.valid) {
      console.log(this.loginForm.value);
    }
  }
}
```

### 7.2 Form Validators

#### Built-in Validators

```typescript
this.form = this.fb.group({
  name: ['', Validators.required],
  email: ['', [Validators.required, Validators.email]],
  age: ['', [Validators.required, Validators.min(18), Validators.max(100)]],
  url: ['', Validators.pattern(/^https?:\/\/.+/)]
});
```

#### Custom Validators

```typescript
import { AbstractControl, ValidationErrors, ValidatorFn } from '@angular/forms';

export function forbiddenValidator(forbidden: RegExp): ValidatorFn {
  return (control: AbstractControl): ValidationErrors | null => {
    const forbiddenValue = forbidden.test(control.value);
    return forbiddenValue ? { forbidden: { value: control.value } } : null;
  };
}

// Usage
this.form = this.fb.group({
  username: ['', forbiddenValidator(/^admin/)]
});
```

## 9. State Management

### 9.1 Services + Signal (Angular 16+ Recommended)

```typescript
import { Injectable, signal, computed } from '@angular/core';

@Injectable({ providedIn: 'root' })
export class CounterService {
  // Signal
  count = signal(0);
  
  // Computed property
  doubleCount = computed(() => this.count() * 2);
  
  // Method
  increment(): void {
    this.count.update(c => c + 1);
  }
  
  decrement(): void {
    this.count.update(c => c - 1);
  }
  
  reset(): void {
    this.count.set(0);
  }
}
```

```typescript
import { Component } from '@angular/core';
import { CounterService } from './services/counter.service';

@Component({
  selector: 'app-counter',
  standalone: true,
  template: `
    <p>Count: {{ counterService.count() }}</p>
    <p>Double: {{ counterService.doubleCount() }}</p>
    <button (click)="counterService.increment()">+</button>
    <button (click)="counterService.decrement()">-</button>
  `
})
export class CounterComponent {
  constructor(public counterService: CounterService) { }
}
```
