## 1. Component Basics

### 1.1 Basic Button

Buttons are among the most commonly used components. Element Plus provides multiple button types.

```vue
<template>
  <!-- Primary Button -->
  <el-button type="primary">Primary Button</el-button>
  
  <!-- Success Button -->
  <el-button type="success">Success Button</el-button>
  
  <!-- Info Button -->
  <el-button type="info">Info Button</el-button>
  
  <!-- Warning Button -->
  <el-button type="warning">Warning Button</el-button>
  
  <!-- Danger Button -->
  <el-button type="danger">Danger Button</el-button>
  
  <!-- Text Button -->
  <el-button type="text">Text Button</el-button>
  
  <!-- Plain Button -->
  <el-button plain>Plain Button</el-button>
  
  <!-- Round Button -->
  <el-button round>Round Button</el-button>
  
  <!-- Icon Button -->
  <el-button icon="el-icon-search">Search</el-button>
</template>
```

### 1.2 Button Attributes

| Attribute | Description | Type | Optional Values | Default |
|-----------|-------------|------|-----------------|---------|
| type | Button type | string | primary / success / warning / danger / info / text | — |
| plain | Whether to use plain style | boolean | — | false |
| round | Whether to use round style | boolean | — | false |
| circle | Whether to use circle style | boolean | — | false |
| loading | Whether to show loading state | boolean | — | false |
| disabled | Whether to be disabled | boolean | — | false |
| icon | Icon class name | string | — | — |
| size | Button size | string | large / default / small | — |

### 1.3 Button Events

```vue
<template>
  <el-button @click="handleClick">Click Event</el-button>
</template>

<script setup>
const handleClick = () => {
  console.log('Button clicked')
}
</script>
```

## 2. Form Components

### 2.1 Basic Form

```vue
<template>
  <el-form :model="form" :rules="rules" ref="formRef" label-width="100px">
    <el-form-item label="Username" prop="username">
      <el-input v-model="form.username" placeholder="Please enter username" />
    </el-form-item>
    
    <el-form-item label="Password" prop="password">
      <el-input v-model="form.password" type="password" placeholder="Please enter password" show-password />
    </el-form-item>
    
    <el-form-item label="Email" prop="email">
      <el-input v-model="form.email" placeholder="Please enter email" />
    </el-form-item>
    
    <el-form-item label="Gender" prop="gender">
      <el-radio-group v-model="form.gender">
        <el-radio label="male">Male</el-radio>
        <el-radio label="female">Female</el-radio>
      </el-radio-group>
    </el-form-item>
    
    <el-form-item label="Hobby" prop="hobby">
      <el-checkbox-group v-model="form.hobby">
        <el-checkbox label="Basketball" />
        <el-checkbox label="Football" />
        <el-checkbox label="Swimming" />
      </el-checkbox-group>
    </el-form-item>
    
    <el-form-item label="Bio" prop="bio">
      <el-input v-model="form.bio" type="textarea" :rows="3" placeholder="Please enter bio" />
    </el-form-item>
    
    <el-form-item>
      <el-button type="primary" @click="submitForm">Submit</el-button>
      <el-button @click="resetForm">Reset</el-button>
    </el-form-item>
  </el-form>
</template>

<script setup>
import { reactive, ref } from 'vue'

const formRef = ref(null)

const form = reactive({
  username: '',
  password: '',
  email: '',
  gender: '',
  hobby: [],
  bio: ''
})

const rules = {
  username: [
    { required: true, message: 'Please enter username', trigger: 'blur' },
    { min: 3, max: 20, message: 'Username must be 3-20 characters', trigger: 'blur' }
  ],
  password: [
    { required: true, message: 'Please enter password', trigger: 'blur' },
    { min: 6, message: 'Password must be at least 6 characters', trigger: 'blur' }
  ],
  email: [
    { required: true, message: 'Please enter email', trigger: 'blur' },
    { type: 'email', message: 'Please enter valid email format', trigger: 'blur' }
  ],
  gender: [
    { required: true, message: 'Please select gender', trigger: 'change' }
  ]
}

const submitForm = async () => {
  if (!formRef.value) return
  
  await formRef.value.validate((valid) => {
    if (valid) {
      console.log('Form data:', form)
      ElMessage.success('Submitted successfully')
    } else {
      ElMessage.error('Please check the form')
    }
  })
}

const resetForm = () => {
  formRef.value?.resetFields()
}
</script>
```

### 2.2 Form Validation Rules

| Rule Type | Description | Example |
|-----------|-------------|---------|
| required | Required field | { required: true, message: 'Required', trigger: 'blur' } |
| type | Data type | type: 'email', type: 'number', type: 'date' |
| min/max | Length limit | { min: 3, max: 20, message: 'Invalid length', trigger: 'blur' } |
| pattern | Regex pattern | { pattern: /^1[3-9]\d{9}$/, message: 'Invalid phone number' } |
| validator | Custom validator | { validator: validatePass, trigger: 'blur' } |

### 3.2 Table with Pagination

```vue
<template>
  <div>
    <el-table :data="tableData" v-loading="loading">
      <el-table-column prop="id" label="ID" width="80" />
      <el-table-column prop="title" label="Title" />
      <el-table-column prop="author" label="Author" width="120" />
      <el-table-column prop="date" label="Publish Date" width="180" />
    </el-table>
    
    <el-pagination
      v-model:current-page="currentPage"
      v-model:page-size="pageSize"
      :total="total"
      :page-sizes="[10, 20, 50, 100]"
      layout="total, sizes, prev, pager, next, jumper"
      @size-change="handleSizeChange"
      @current-change="handleCurrentChange"
    />
  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue'

const tableData = ref([])
const loading = ref(false)
const currentPage = ref(1)
const pageSize = ref(10)
const total = ref(0)

const loadData = async () => {
  loading.value = true
  // Simulate API call
  setTimeout(() => {
    tableData.value = [
      { id: 1, title: 'Article Title 1', author: 'Author 1', date: '2024-01-01' },
      { id: 2, title: 'Article Title 2', author: 'Author 2', date: '2024-01-02' }
    ]
    total.value = 100
    loading.value = false
  }, 500)
}

const handleSizeChange = (val) => {
  pageSize.value = val
  loadData()
}

const handleCurrentChange = (val) => {
  currentPage.value = val
  loadData()
}

onMounted(() => {
  loadData()
})
</script>
```

## 4. Dialog and Drawer

### 4.1 Dialog

```vue
<template>
  <el-button type="primary" @click="dialogVisible = true">Open Dialog</el-button>
  
  <el-dialog
    v-model="dialogVisible"
    title="Edit Information"
    width="500px"
    :close-on-click-modal="false"
    @close="handleClose"
  >
    <el-form :model="form" label-width="80px">
      <el-form-item label="Name">
        <el-input v-model="form.name" />
      </el-form-item>
      <el-form-item label="Description">
        <el-input v-model="form.desc" type="textarea" :rows="3" />
      </el-form-item>
    </el-form>
    
    <template #footer>
      <el-button @click="dialogVisible = false">Cancel</el-button>
      <el-button type="primary" @click="handleConfirm">Confirm</el-button>
    </template>
  </el-dialog>
</template>

<script setup>
import { ref, reactive } from 'vue'

const dialogVisible = ref(false)

const form = reactive({
  name: '',
  desc: ''
})

const handleClose = () => {
  form.name = ''
  form.desc = ''
}

const handleConfirm = () => {
  console.log('Submit:', form)
  dialogVisible.value = false
  ElMessage.success('Success')
}
</script>
```

## 8. Component Event Handling

### 8.1 Event Modifiers

```vue
<template>
  <!-- Stop event propagation -->
  <el-button @click.stop="handleClick">Click</el-button>
  
  <!-- Prevent default behavior -->
  <el-form @submit.prevent="handleSubmit">
    <el-button native-type="submit">Submit</el-button>
  </el-form>
  
  <!-- Trigger only once -->
  <el-button @click.once="handleOnce">Once Only</el-button>
</template>
```

### 8.2 Form Validation Events

```vue
<script setup>
import { ref } from 'vue'

const formRef = ref(null)

// Validate single field
const validateField = async () => {
  await formRef.value?.validateField('username')

  console.log(formRef.value?.fields)
```

## 10. Performance Optimization Suggestions

### 10.1 On-Demand Import

Use the auto-import plugin to only import needed components, reducing bundle size.

### 10.2 Virtual Scrolling

For large data tables, use el-table-v2 virtual scrolling component.

### 10.3 Avoid Unnecessary Reactivity

Use shallowRef or markRaw for objects that don't need reactivity:

```typescript
import { shallowRef, markRaw } from 'vue'

// Use shallowRef for large data objects
const largeData = shallowRef([])

// Use markRaw for non-reactive objects
const complexObj = markRaw({ /* ... */ })
```
