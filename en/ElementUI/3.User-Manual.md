# Element UI User Manual

Core component usage and examples.

## Basic Components

### Button

```html
<el-button>Default</el-button>
<el-button type="primary">Primary</el-button>
<el-button type="success">Success</el-button>
<el-button type="warning">Warning</el-button>
<el-button type="danger">Danger</el-button>
<el-button plain>Plain</el-button>
<el-button round>Round</el-button>
<el-button disabled>Disabled</el-button>
```

### Layout

```html
<el-row>
  <el-col :span="12"><div class="grid-content"></div></el-col>
  <el-col :span="12"><div class="grid-content"></div></el-col>
</el-row>

<el-row :gutter="20">
  <el-col :span="6"><div class="grid-content"></div></el-col>
  <el-col :span="6"><div class="grid-content"></div></el-col>
  <el-col :span="6"><div class="grid-content"></div></el-col>
  <el-col :span="6"><div class="grid-content"></div></el-col>
</el-row>
```

### Container

```html
<el-container>
  <el-header>Header</el-header>
  <el-container>
    <el-aside width="200px">Aside</el-aside>
    <el-main>Main</el-main>
  </el-container>
</el-container>
```

## Form Components

### Input

```html
<el-input v-model="input" placeholder="Enter content"></el-input>

<el-input prefix-icon="el-icon-search" placeholder="Search"></el-input>

<el-input type="textarea" :rows="3" placeholder="Enter content"></el-input>
```

### Select

```html
<el-select v-model="value" placeholder="Select">
  <el-option
    v-for="item in options"
    :key="item.value"
    :label="item.label"
    :value="item.value">
  </el-option>
</el-select>

<el-select v-model="value" filterable placeholder="Filterable"></el-select>

<el-select v-model="value" multiple placeholder="Multiple"></el-select>
```

### Form with Validation

```html
<el-form :model="ruleForm" :rules="rules" ref="ruleForm" label-width="100px">
  <el-form-item label="Name" prop="name">
    <el-input v-model="ruleForm.name"></el-input>
  </el-form-item>
  <el-form-item>
    <el-button type="primary" @click="submitForm">Submit</el-button>
    <el-button @click="resetForm">Reset</el-button>
  </el-form-item>
</el-form>
```

## Data Components

### Table

```html
<el-table :data="tableData" style="width: 100%">
  <el-table-column prop="date" label="Date" width="180"></el-table-column>
  <el-table-column prop="name" label="Name" width="180"></el-table-column>
  <el-table-column prop="address" label="Address"></el-table-column>
</el-table>

<el-table :data="tableData" stripe border></el-table>
```

### Pagination

```html
<el-pagination
  background
  layout="prev, pager, next"
  :total="1000">
</el-pagination>
```

### Tree

```html
<el-tree
  :data="data"
  :props="defaultProps"
  @node-click="handleNodeClick">
</el-tree>
```

## Notification Components

### Message

```javascript
this.$message({ message: 'Success', type: 'success' })
this.$message({ message: 'Warning', type: 'warning' })
this.$message.error('Error')
```

### Notification

```javascript
this.$notify({ title: 'Title', message: 'Message' })
this.$notify({ title: 'Success', type: 'success' })
```

### Alert

```html
<el-alert title="Success" type="success"></el-alert>
<el-alert title="Warning" type="warning"></el-alert>
<el-alert title="Error" type="error"></el-alert>
```

## Navigation Components

### Menu

```html
<el-menu :default-active="activeIndex">
  <el-menu-item index="1">
    <i class="el-icon-menu"></i>
    <span>Nav 1</span>
  </el-menu-item>
</el-menu>
```

### Tabs

```html
<el-tabs v-model="activeName">
  <el-tab-pane label="Tab 1" name="first">Content 1</el-tab-pane>
  <el-tab-pane label="Tab 2" name="second">Content 2</el-tab-pane>
</el-tabs>
```

## Other Components

### Dialog

```html
<el-button @click="dialogVisible = true">Open Dialog</el-button>

<el-dialog title="Title" :visible.sync="dialogVisible" width="30%">
  <span>Content</span>
  <span slot="footer">
    <el-button @click="dialogVisible = false">Cancel</el-button>
    <el-button @click="dialogVisible = false">OK</el-button>
  </span>
</el-dialog>
```

### Drawer

```html
<el-drawer title="Title" :visible.sync="drawer" direction="rtl">
  <div>Content</div>
</el-drawer>
```

### Upload

```html
<el-upload action="/upload" list-type="picture">
  <el-button size="small" type="primary">Upload</el-button>
</el-upload>
```
