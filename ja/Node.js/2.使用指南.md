# Node.js ユーザーガイド

このガイドでは、Node.jsを使用してバックエンドアプリケーションやコマンドラインツールを構築する方法を説明します。

## クイックスタート

### 最初のアプリケーションを作成

```javascript
// hello.js
console.log('Hello, Node.js!');

// 実行
node hello.js
```

### HTTPサーバーを作成

```javascript
// server.js
const http = require('http');

const server = http.createServer((req, res) => {
  res.writeHead(200, { 'Content-Type': 'text/plain' });
  res.end('Hello World!');
});

server.listen(3000, () => {
  console.log('サーバーは http://localhost:3000 で実行中');
});
```

実行：

```bash
node server.js
```

## コアモジュール

### fs - ファイルシステム

```javascript
const fs = require('fs');

// ファイルを読み取り
fs.readFile('file.txt', 'utf8', (err, data) => {
  if (err) throw err;
  console.log(data);
});

// ファイルに書き込み
fs.writeFile('file.txt', 'Hello!', (err) => {
  if (err) throw err;
  console.log('ファイルを保存しました');
});

// 同期読み取り
const data = fs.readFileSync('file.txt', 'utf8');
```

### path - パス

```javascript
const path = require('path');

path.join('/foo', 'bar', 'baz'); // /foo/bar/baz
path.resolve(__dirname, 'file.txt'); // 絶対パス
path.extname('file.txt'); // .txt
path.basename('/foo/bar.txt'); // bar.txt
```

### os - オペレーティングシステム

```javascript
const os = require('os');

os.platform(); // 'win32', 'linux', 'darwin'
os.arch(); // 'x64', 'arm64'
os.cpus(); // CPU情報
os.freemem(); // 空きメモリ
os.totalmem(); // 総メモリ
```

### events - イベント

```javascript
const EventEmitter = require('events');

class MyEmitter extends EventEmitter {}

const myEmitter = new MyEmitter();

myEmitter.on('event', () => {
  console.log('イベントが発生しました!');
});

myEmitter.emit('event');
```

### crypto - 暗号化

```javascript
const crypto = require('crypto');

const hash = crypto.createHash('sha256');
hash.update('password');
const password = hash.digest('hex');
```

## Expressフレームワーク

### インストール

```bash
npm install express
```

### アプリケーションを作成

```javascript
const express = require('express');
const app = express();

app.get('/', (req, res) => {
  res.send('Hello World!');
});

app.listen(3000, () => {
  console.log('アプリケーションは http://localhost:3000 で実行中');
});
```

### ルーティング

```javascript
// GETルート
app.get('/api/users', (req, res) => {
  res.json({ users: [] });
});

// POSTルート
app.post('/api/users', (req, res) => {
  const user = req.body;
  res.json({ success: true, user });
});

// パラメータルート
app.get('/api/users/:id', (req, res) => {
  const { id } = req.params;
  res.json({ id });
});
```

### ミドルウェア

```javascript
// JSONを解析
app.use(express.json());

// URLエンコードデータを解析
app.use(express.urlencoded({ extended: true }));

// ロギングミドルウェア
app.use((req, res, next) => {
  console.log(`${req.method} ${req.url}`);
  next();
});

// エラー処理
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).send('Something broke!');
});
```

## 非同期プログラミング

### コールバック

```javascript
fs.readFile('file.txt', 'utf8', (err, data) => {
  if (err) throw err;
  console.log(data);
});
```

### Promise

```javascript
const fs = require('fs').promises;

async function read() {
  try {
    const data = await fs.readFile('file.txt', 'utf8');
    console.log(data);
  } catch (err) {
    console.error(err);
  }
}

read();
```

### Async/Await

```javascript
const axios = require('axios');

async function fetchData() {
  try {
    const response = await axios.get('https://api.example.com/data');
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}

fetchData();
```

## ベストプラクティス

### 1. async/awaitを使用

```javascript
// 避けるべき
getData((err, data) => {
  if (err) handleError(err);
  processData(data, (err, result) => {
    if (err) handleError(err);
    saveResult(result);
  });
});

// 推奨
async function main() {
  try {
    const data = await getData();
    const result = await processData(data);
    await saveResult(result);
  } catch (err) {
    handleError(err);
  }
}
```

### 2. エラー処理

```javascript
// 非同期エラーは常にキャッチ
async function fetchData() {
  try {
    const response = await fetch(url);
    return await response.json();
  } catch (error) {
    console.error('データの取得に失敗:', error);
    throw error;
  }
}
```

```javascript
require('dotenv').config();

const dbUrl = process.env.DATABASE_URL;
```
