## Linux

### Ubuntu / Debian

#### Method 1: APT Repository (Recommended)

```bash
# Update package index
sudo apt update

# Install MySQL Server
sudo apt install mysql-server

# Start MySQL service
sudo systemctl start mysql
sudo systemctl enable mysql

# Secure MySQL installation
sudo mysql_secure_installation
```

#### Method 2: Docker

```bash
# Pull MySQL image
docker pull mysql:latest

# Run MySQL container
docker run -d \
  --name mysql \
  -e MYSQL_ROOT_PASSWORD=your_password \
  -v mysql-data:/var/lib/mysql \
  -p 3306:3306 \
  mysql:latest
```

### CentOS / RHEL

#### Using YUM Repository

```bash
# Download MySQL YUM repository
sudo yum install https://dev.mysql.com/get/mysql80-community-release-el8-4.noarch.rpm

# Install MySQL Server
sudo yum install mysql-community-server

# Start MySQL service
sudo systemctl start mysqld
sudo systemctl enable mysqld

# Get temporary password
sudo grep 'temporary password' /var/log/mysqld.log

# Secure MySQL installation
mysql_secure_installation
```

### Fedora

```bash
# Install MySQL
sudo dnf install @mysql

# Start service
sudo systemctl start mysqld
sudo systemctl enable mysqld
```

### Arch Linux

```bash
# Install MySQL
sudo pacman -S mariadb

# Initialize database
sudo mysql_install_db --user=mysql --basedir=/usr --datadir=/var/lib/mysql

# Start service
sudo systemctl start mariadb
sudo systemctl enable mariadb
```

## macOS

### Homebrew (Recommended)

```bash
# Install Homebrew (if not installed)
/bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"

# Install MySQL
brew install mysql

# Start MySQL
brew services start mysql

# Or run manually
mysqld_safe --datadir=/usr/local/var/mysql
```

### DMG Package

1. Download MySQL from [dev.mysql.com](https://dev.mysql.com/downloads/mysql/)
2. Install the `.dmg` package
3. Run MySQL from System Preferences or command line

### Configuration

```bash
# Set root password
mysql_secure_installation

# Or connect and set password
mysql -u root
ALTER USER 'root'@'localhost' IDENTIFIED BY 'YourPassword';
```

## Windows

### MSI Installer (Recommended)

1. Download MySQL Installer from [dev.mysql.com](https://dev.mysql.com/downloads/installer/)
2. Run the installer
3. Choose "Full" or "Custom" installation
4. Follow the wizard
5. Configure as needed

### ZIP Archive

```powershell
# Download ZIP archive
# Extract to C:\mysql

# Initialize MySQL
cd C:\mysql
mysqld --initialize-insecure

# Install MySQL service
mysqld --install MySQL

# Start MySQL
net start MySQL

# Set root password
mysql -u root
ALTER USER 'root'@'localhost' IDENTIFIED BY 'YourPassword';
```

### Docker

```powershell
# Pull and run MySQL
docker run -d --name mysql -e MYSQL_ROOT_PASSWORD=secret -p 3306:3306 -v mysql-data:/var/lib/mysql mysql:latest
```

## Initial Setup

### Create Database and User

```sql
-- Connect as root
mysql -u root -p

-- Create database
CREATE DATABASE app_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

-- Create user
CREATE USER 'app_user'@'localhost' IDENTIFIED BY 'user_password';

-- Grant privileges
GRANT ALL PRIVILEGES ON app_db.* TO 'app_user'@'localhost';

-- Apply changes
FLUSH PRIVILEGES;
```

### Configuration File

#### Linux: `/etc/mysql/mysql.conf.d/mysqld.cnf`

```ini
[mysqld]
# Basic Settings
max_connections = 200
max_connect_errors = 100
port = 3306
datadir = /var/lib/mysql
socket = /var/run/mysqld/mysqld.sock

# Character Set
character-set-server = utf8mb4
collation-server = utf8mb4_unicode_ci

# InnoDB Settings
innodb_buffer_pool_size = 1G
innodb_log_file_size = 256M
innodb_flush_log_at_trx_commit = 2

# Query Cache (MySQL 5.7)
query_cache_type = 1
query_cache_size = 64M

# Logging
log_error = /var/log/mysql/error.log
slow_query_log = 1
slow_query_log_file = /var/log/mysql/slow.log
long_query_time = 2
```

## Uninstall

### Ubuntu/Debian

```bash
# Stop MySQL
sudo systemctl stop mysql

# Remove packages
sudo apt purge mysql-server mysql-client mysql-common
sudo apt autoremove

# Remove data directory
sudo rm -rf /var/lib/mysql
sudo rm -rf /etc/mysql
```

### macOS (Homebrew)

```bash
# Stop MySQL
brew services stop mysql

# Uninstall
brew uninstall mysql

# Remove data
rm -rf /usr/local/var/mysql
```

### Windows

```bash
# Via Programs and Features
# Or using command line
net stop MySQL
mysqld --remove
```
