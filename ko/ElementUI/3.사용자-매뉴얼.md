### Form (유효성 검사 포함)

```html
<el-form :model="ruleForm" :rules="rules" ref="ruleForm" label-width="100px">
  <el-form-item label="활동 이름" prop="name">
    <el-input v-model="ruleForm.name"></el-input>
  </el-form-item>
  <el-form-item label="활동 지역" prop="region">
    <el-select v-model="ruleForm.region" placeholder="선택">
      <el-option label="지역 1" value="shanghai"></el-option>
      <el-option label="지역 2" value="beijing"></el-option>
    </el-select>
  </el-form-item>
  <el-form-item label="날짜" prop="date1">
    <el-date-picker
      v-model="ruleForm.date1"
      type="date"
      placeholder="날짜 선택">
    </el-date-picker>
  </el-form-item>
  <el-form-item>
    <el-button type="primary" @click="submitForm('ruleForm')">제출</el-button>
    <el-button @click="resetForm('ruleForm')">초기화</el-button>
  </el-form-item>
</el-form>
```

유효성 검사 규칙:

```javascript
export default {
  data() {
    return {
      ruleForm: {
        name: '',
        region: '',
        date1: ''
      },
      rules: {
        name: [
          { required: true, message: '활동 이름을 입력하세요', trigger: 'blur' },
          { min: 3, max: 5, message: '길이는 3~5자', trigger: 'blur' }
        ],
        region: [
          { required: true, message: '지역을 선택하세요', trigger: 'change' }
        ],
        date1: [
          { required: true, message: '날짜를 선택하세요', trigger: 'change' }
        ]
      }
    }
  },
  methods: {
    submitForm(formName) {
      this.$refs[formName].validate((valid) => {
        if (valid) {
          alert('제출!')
        } else {
          return false
        }
      })
    },
    resetForm(formName) {
      this.$refs[formName].resetFields()
    }
  }
}
```

### 폼 유효성 검사

```javascript
// 커스텀 유효성 검사기
const validatePass = (rule, value, callback) => {
  if (value === '') {
    callback(new Error('비밀번호를 입력하세요'))
  } else {
    if (this.ruleForm.checkPass !== '') {
      this.$refs.ruleForm.validateField('checkPass')
    }
    callback()
  }
}
```
