# Angular Bedienungsanleitung

Diese Anleitung behandelt die Verwendung von Angular zum Erstellen von Webanwendungen.

## Schnellstart

### Entwicklungsserver

```bash
# Entwicklungsserver starten
ng serve

# Mit bestimmtem Port starten
ng serve --port 4201

# Browser automatisch öffnen
ng serve --open

# Hot Reload aktivieren
ng serve --hmr
```

### Build

```bash
# Entwicklungs-Build
ng build

# Produktions-Build
ng build --configuration production

# Mit Statistiken bauen
ng build --stats-json
```

## Komponenten

### Komponente erstellen

```bash
# Komponente generieren
ng generate component components/user-list

# Kurzform
ng g c components/user-list
```

### Komponentenbeispiel

```typescript
import { Component } from '@angular/core';

@Component({
  selector: 'app-user-list',
  template: `
    <h1>Benutzerliste</h1>
    <ul>
      <li *ngFor="let user of users">{{ user.name }}</li>
    </ul>
  `,
  standalone: true
})
export class UserListComponent {
  users = [
    { name: 'Max' },
    { name: 'Anna' }
  ];
}
```

## Dienste

### Dienst erstellen

```bash
ng generate service services/user
# oder
ng g s services/user
```

### Dienstbeispiel

```typescript
import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';

@Injectable({
  providedIn: 'root'
})
export class UserService {
  constructor(private http: HttpClient) {}

  getUsers(): Observable<User[]> {
    return this.http.get<User[]>('/api/users');
  }
}
```

## Routing

### Routen konfigurieren

```typescript
// app.routes.ts
import { Routes } from '@angular/router';

export const routes: Routes = [
  { path: '', redirectTo: '/home', pathMatch: 'full' },
  { path: 'home', component: HomeComponent },
  { path: 'users', component: UserListComponent },
  { path: 'users/:id', component: UserDetailComponent },
  { path: '**', component: NotFoundComponent }
];
```

### Navigation


## Formulare

### Vorlagengesteuerte Formulare

```typescript
import { FormsModule } from '@angular/forms';

@Component({
  imports: [FormsModule],
  template: `
    <input [(ngModel)]="name" name="name">
    <button (click)="submit()">Absenden</button>
  `
})
export class LoginComponent {
  name = '';
}
```

### Reaktive Formulare

```typescript
import { ReactiveFormsModule, FormBuilder, FormGroup, Validators } from '@angular/forms';

@Component({
  imports: [ReactiveFormsModule],
  template: `
    <form [formGroup]="form" (ngSubmit)="submit()">
      <input formControlName="email">
      <input formControlName="password" type="password">
      <button type="submit">Absenden</button>
    </form>
  `
})
export class LoginComponent {
  form: FormGroup;

  constructor(private fb: FormBuilder) {
    this.form = this.fb.group({
      email: ['', [Validators.required, Validators.email]],
      password: ['', [Validators.required, Validators.minLength(6)]]
    });
  }
}
```

## HTTP-Client

### Konfigurieren

```typescript
// app.config.ts
import { provideHttpClient } from '@angular/common/http';

export const appConfig: ApplicationConfig = {
  providers: [
    provideHttpClient()
  ]
};
```

### API-Aufruf

```typescript
import { HttpClient } from '@angular/common/http';

@Component({})
export class UserComponent {
  constructor(private http: HttpClient) {}

  getUsers() {
    this.http.get('/api/users').subscribe(data => {
      console.log(data);
    });
  }
}
```

## Statusverwaltung

### Signals (Angular 16+)

```typescript
import { signal, computed } from '@angular/core';

@Component({})
export class CounterComponent {
  count = signal(0);
  double = computed(() => this.count() * 2);

  increment() {
    this.count.update(c => c + 1);
  }
}
```

## Best Practices

### 1. Standalone-Komponenten verwenden

```typescript
@Component({
  standalone: true,
  imports: [CommonModule],
  ...
})
export class MyComponent {}
```

### 2. Signals für Status verwenden

```typescript
# Bevorzugt
count = signal(0);

# Vermeiden
count = 0;
```

### 3. Lazy Loading

```typescript
{
  path: 'admin',
  loadComponent: () => import('./admin/admin.component')
}
```
