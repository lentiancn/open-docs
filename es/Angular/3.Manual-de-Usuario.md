# Manual de Usuario de Angular

> Funciones

## Componentes

```typescript
@Component({
  selector: 'app-my',
  templateUrl: './my.component.html'
})
export class MyComponent {
  title = 'Hello';
}
```

## Data Binding

```html
<img [src]="imageUrl">
<button (click)="onClick()">Click</button>
<input [(ngModel)]="name">
```

## Directivas

```html
<div *ngIf="show">Content</div>
<li *ngFor="let item of items">{{item}}</li>
```

## Servicios

```typescript
@Injectable({ providedIn: 'root' })
export class MyService {
  getData() { return 'Data'; }
}
```

## Routing

```typescript
const routes: Routes = [
  { path: '', component: HomeComponent }
];
```
