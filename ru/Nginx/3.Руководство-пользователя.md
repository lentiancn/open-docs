# Руководство пользователя Nginx

## Основные операции

### Запуск Nginx

```bash
# Запуск Nginx
nginx

# Или через systemctl (для systemd)
systemctl start nginx
```

### Остановка Nginx

```bash
# Быстрая остановка (немедленное завершение)
nginx -s stop

# Плавная остановка (ожидание завершения запросов)
nginx -s quit
```

### Перезагрузка конфигурации

```bash
# Перезагрузить конфигурацию
nginx -s reload
```

### Переоткрытие логов

```bash
# Переоткрыть лог-файлы
nginx -s reopen
```

### Проверка конфигурации

```bash
# Тестирование синтаксиса конфигурации
nginx -t

# Показать полную конфигурацию
nginx -T
```

## Структура конфигурационного файла

### Структура основной конфигурации

```
nginx.conf
├── main (глобальный блок)
│   ├── user
│   ├── worker_processes
│   ├── error_log
│   └── pid
│
├── events (блок events)
│   └── worker_connections
│
└── http (блок http)
    ├── include
    ├── default_type
    ├── log_format
    ├── access_log
    ├── sendfile
    ├── keepalive_timeout
    ├── gzip
    │
    └── server (блок server)
        ├── listen
        ├── server_name
        ├── location
        │   ├── root
        │   ├── index
        │   └── proxy_pass
        │
        └── error_page
```

### Часто используемые директивы

#### Блок main

```nginx
# Пользователь, от имени которого работает Nginx
user nginx;

# Количество рабочих процессов (auto определяет CPU автоматически)
worker_processes auto;

# Путь к логу ошибок
error_log /var/log/nginx/error.log warn;

# Путь к PID-файлу
pid /var/run/nginx.pid;
```

#### Блок events

```nginx
events {
    # Максимальное количество соединений на рабочий процесс
    worker_connections 1024;
    
    # Использовать epoll (Linux)
    use epoll;
    
    # Принимать несколько соединений одновременно
    multi_accept on;
}
```

#### Блок http

```nginx
http {
    # Включить MIME-типы
    include /etc/nginx/mime.types;
    
    # Тип файла по умолчанию
    default_type application/octet-stream;
    
    # Лог доступа
    access_log /var/log/nginx/access.log;
    
    # Отправка файлов
    sendfile on;
    
    # TCP_NOPUSH и TCP_NODELAY
    tcp_nopush on;
    tcp_nodelay on;
    
    # Таймаут keep-alive
    keepalive_timeout 65;
    
    # Включить сжатие gzip
    gzip on;
}
```

## Обслуживание статических файлов

### Базовая конфигурация

```nginx
server {
    listen 80;
    server_name example.ru www.example.ru;
    
    # Корневая директория
    root /var/www/html;
    
    # Файлы индекса по умолчанию
    index index.html index.htm;
    
    location / {
        try_files $uri $uri/ =404;
    }
    
    # Логи доступа
    access_log /var/log/nginx/example.ru.access.log;
    error_log /var/log/nginx/example.ru.error.log;
}
```

### Листинг директорий

```nginx
location / {
    # Включить листинг директорий
    autoindex on;
    
    # Показывать точные размеры файлов
    autoindex_exact_size on;
    
    # Показывать местное время
    autoindex_localtime on;
}
```

### Кэширование статических ресурсов

```nginx
location ~* \.(jpg|jpeg|png|gif|ico|css|js|woff|woff2|ttf|svg)$ {
    expires 30d;
    add_header Cache-Control "public, no-transform";
}
```

## Обратный прокси

### Базовый обратный прокси

```nginx
server {
    listen 80;
    server_name example.ru;
    
    location / {
        proxy_pass http://backend_server;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
}
```

### Обратный прокси с кэшированием

```nginx
proxy_cache_path /var/cache/nginx levels=1:2 keys_zone=my_cache:10m max_size=100m inactive=60m use_temp_path=off;

server {
    listen 80;
    server_name example.ru;
    
    location / {
        proxy_cache my_cache;
        proxy_cache_valid 200 60m;
        proxy_cache_use_stale error timeout http_500 http_502 http_503 http_504;
        proxy_cache_lock on;
        
        proxy_pass http://backend_server;
        proxy_set_header Host $host;
    }
}
```

### Проксирование WebSocket

```nginx
location /ws/ {
    proxy_pass http://backend_server;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection "upgrade";
    proxy_set_header Host $host;
    proxy_read_timeout 86400;
}
```

## Балансировка нагрузки

### Round-Robin балансировка

```nginx
upstream backend {
    server backend1.example.ru;
    server backend2.example.ru;
    server backend3.example.ru;
}

server {
    listen 80;
    server_name example.ru;
    
    location / {
        proxy_pass http://backend;
    }
}
```

### Взвешенная балансировка

```nginx
upstream backend {
    server backend1.example.ru weight=5;
    server backend2.example.ru weight=2;
    server backend3.example.ru;
}
```

### IP-hash балансировка

```nginx
upstream backend {
    ip_hash;
    server backend1.example.ru;
    server backend2.example.ru;
    server backend3.example.ru down;
}
```

### Least Connections

```nginx
upstream backend {
    least_conn;
    server backend1.example.ru;
    server backend2.example.ru;
}
```

## Конфигурация SSL/TLS

### Базовая HTTPS-конфигурация

```nginx
server {
    listen 443 ssl http2;
    server_name example.ru;
    
    ssl_certificate /etc/nginx/ssl/example.ru.crt;
    ssl_certificate_key /etc/nginx/ssl/example.ru.key;
    
    # Версии SSL-протокола
    ssl_protocols TLSv1.2 TLSv1.3;
    
    # Наборы шифров SSL
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;
    ssl_prefer_server_ciphers off;
    
    # HSTS
    add_header Strict-Transport-Security "max-age=63072000" always;
    
    location / {
        root /var/www/html;
        index index.html;
    }
}
```

### Перенаправление с HTTP на HTTPS

```nginx
server {
    listen 80;
    server_name example.ru;
    return 301 https://$server_name$request_uri;
}
```

## Примеры распространённых конфигураций

### Конфигурация PHP-FPM

```nginx
server {
    listen 80;
    server_name example.ru;
    root /var/www/html;
    index index.php index.html;
    
    location ~ \.php$ {
        fastcgi_pass unix:/var/run/php/php-fpm.sock;
        fastcgi_index index.php;
        fastcgi_param SCRIPT_FILENAME $document_root$fastcgi_script_name;
        include fastcgi_params;
    }
    
    location ~ /\.ht {
        deny all;
    }
}
```

### Прокси Node.js

```nginx
server {
    listen 80;
    server_name example.ru;
    
    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}
```

### Конфигурация ограничения частоты

```nginx
# Ограничение на основе соединений
limit_conn_zone $binary_remote_addr zone=addr:10m;

server {
    location / {
        limit_conn addr 10;
    }
}

# Ограничение на основе частоты запросов
limit_req_zone $binary_remote_addr zone=one:10m rate=10r/s;

server {
    location / {
        limit_req zone=one burst=20 nodelay;
    }
}
```

### Защита от hotlinking

```nginx
location ~* \.(jpg|jpeg|png|gif|ico|css|js)$ {
    valid_referers none blocked example.ru *.example.ru;
    
    if ($invalid_referer) {
        return 403;
    }
}
```

## Управление логами

### Пользовательский формат лога

```nginx
log_format main '$remote_addr - $remote_user [$time_local] "$request" '
                '$status $body_bytes_sent "$http_referer" '
                '"$http_user_agent" "$http_x_forwarded_for"';

access_log /var/log/nginx/access.log main;
```

### Условное логирование

```nginx
map $request_uri $loggable {
    ~^/health-check 0;
    default 1;
}

access_log /var/log/nginx/access.log combined if=$loggable;
```

### Ротация логов

Использование logrotate:

```bash
# /etc/logrotate.d/nginx
/var/log/nginx/*.log {
    daily
    missingok
    rotate 14
    compress
    delaycompress
    notifempty
    create 0640 www-data adm
    sharedscripts
    postrotate
        [ -f /var/run/nginx.pid ] && kill -USR1 `cat /var/run/nginx.pid`
    endscript
}
```
