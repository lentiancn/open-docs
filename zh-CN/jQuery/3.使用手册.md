# jQuery 使用手册

## 快速开始

### 引入 jQuery

在 HTML 文件的 `<head>` 或 `<body>` 底部引入 jQuery：

```html
<!DOCTYPE html>
<html>
<head>
    <title>jQuery Demo</title>
    <script src="https://code.jquery.com/jquery-4.0.0.min.js"></script>
</head>
<body>
    <!-- 页面内容 -->
</body>
</html>
```

### 等待 DOM 加载完成

jQuery 代码需要在 DOM 完全加载后才能执行：

```javascript
$(document).ready(function() {
    // 在这里编写 jQuery 代码
});

// 简写形式
$(function() {
    // 在这里编写 jQuery 代码
});
```

## 选择器

jQuery 选择器是 jQuery 的核心功能之一，语法与 CSS 选择器基本一致。

### 基本选择器

```javascript
$("#myId")          // ID 选择器
$(".myClass")       // 类选择器
$("div")            // 元素选择器
$("*")              // 通配符选择器
$("div, p, span")   // 多选择器
```

### 属性选择器

```javascript
$('[href]')                    // 有 href 属性的元素
$('[href="value"]')           // href 属性等于指定值的元素
$('[href!="value"]')          // href 属性不等于指定值的元素
$('[href^="http"]')           // href 属性以指定值开头的元素
$('[href$=".pdf"]')           // href 属性以指定值结尾的元素
$('[href*="google"]')         // href 属性包含指定值的元素
```

### 筛选选择器

```javascript
$("div:first")        // 第一个 div 元素
$("div:last")         // 最后一个 div 元素
$("div:even")         // 偶数位置的 div 元素
$("div:odd")          // 奇数位置的 div 元素
$("div:eq(n)")        // 索引为 n 的 div 元素
$("div:gt(n)")        // 索引大于 n 的 div 元素
$("div:lt(n)")        // 索引小于 n 的 div 元素
$("div:not(.exclude") // 不含指定类的 div 元素
```

## DOM 操作

### 获取和设置内容

```javascript
// 获取/设置文本内容
var text = $("#myElement").text();
$("#myElement").text("新的文本内容");

// 获取/设置 HTML 内容
var html = $("#myElement").html();
$("#myElement").html("<p>新的 HTML 内容</p>");

// 获取/设置表单值
var value = $("#inputField").val();
$("#inputField").val("新的值");
```

### 获取和设置属性

```javascript
// 获取属性
var src = $("#myImage").attr("src");

// 设置单个属性
$("#myImage").attr("src", "image.jpg");

// 设置多个属性
$("#myImage").attr({
    src: "image.jpg",
    alt: "图片说明",
    title: "图片标题"
});

// 移除属性
$("#myImage").removeAttr("title");
```

### 获取和设置 CSS

```javascript
// 获取单个 CSS 属性
var color = $("#myElement").css("color");

// 设置单个 CSS 属性
$("#myElement").css("color", "red");

// 设置多个 CSS 属性
$("#myElement").css({
    color: "red",
    backgroundColor: "blue",
    fontSize: "16px"
});

// 添加/移除类
$("#myElement").addClass("active highlight");
$("#myElement").removeClass("highlight");
$("#myElement").toggleClass("active");  // 切换类
```

### DOM 遍历

```javascript
// 父元素
$("#myElement").parent();
$("#myElement").parents();
$("#myElement").closest(".container");

// 子元素
$("#myElement").children();
$("#myElement").find(".child");

// 兄弟元素
$("#myElement").siblings();
$("#myElement").next();
$("#myElement").prev();
$("#myElement").nextAll();
```

### 创建和删除元素

```javascript
// 创建新元素
var newDiv = $("<div>", {
    class: "new-element",
    text: "新元素内容"
});

// 添加到页面
$("#container").append(newDiv);      // 内部末尾
$("#container").prepend(newDiv);      // 内部开头
$("#container").before(newDiv);      // 外部之前
$("#container").after(newDiv);       // 外部之后

// 删除元素
$("#myElement").remove();             // 删除元素及其子元素
$("#myElement").empty();              // 清空子元素
```

## 事件处理

### 绑定事件

```javascript
// 绑定点击事件
$("#myButton").click(function() {
    alert("按钮被点击了");
});

// 使用 on 方法绑定
$("#myButton").on("click", function() {
    alert("按钮被点击了");
});

// 绑定多个事件
$("#myElement").on({
    mouseenter: function() {
        $(this).css("background", "yellow");
    },
    mouseleave: function() {
        $(this).css("background", "");
    }
});
```

### 事件委托

```javascript
// 动态元素的事件绑定
$("#container").on("click", ".dynamic-item", function() {
    alert("动态元素被点击");
});
```

### 常用事件

| 事件 | 说明 |
|------|------|
| click | 点击事件 |
| dblclick | 双击事件 |
| mouseenter | 鼠标进入 |
| mouseleave | 鼠标离开 |
| mousemove | 鼠标移动 |
| keydown | 键盘按下 |
| keyup | 键盘释放 |
| focus | 获得焦点 |
| blur | 失去焦点 |
| submit | 表单提交 |
| change | 值改变 |
| load | 加载完成 |

## 动画效果

### 基本动画

```javascript
// 显示元素
$("#myElement").show();

// 隐藏元素
$("#myElement").hide();

// 切换显示状态
$("#myElement").toggle();

// 淡入
$("#myElement").fadeIn();

// 淡出
$("#myElement").fadeOut();

// 淡入淡出切换
$("#myElement").fadeToggle();

// 滑动显示
$("#myElement").slideDown();

// 滑动隐藏
$("#myElement").slideUp();

// 滑动切换
$("#myElement").slideToggle();
```

### 自定义动画

```javascript
$("#myElement").animate({
    opacity: 0.5,
    left: "+=50",
    height: "200px"
}, 300, function() {
    // 动画完成后的回调
    console.log("动画完成");
});
```

### 停止动画

```javascript
// 停止当前动画
$("#myElement").stop();

// 停止所有动画，并到达最终状态
$("#myElement").stop(true, true);
```

## Ajax 请求

### 基本用法

```javascript
$.ajax({
    url: "/api/data",
    method: "GET",
    data: { id: 1, name: "test" },
    success: function(response) {
        console.log("成功:", response);
    },
    error: function(xhr, status, error) {
        console.log("错误:", error);
    }
});
```

### 快捷方法

```javascript
// GET 请求
$.get("/api/data", { id: 1 }, function(response) {
    console.log(response);
});

// POST 请求
$.post("/api/submit", { name: "test" }, function(response) {
    console.log(response);
});

// 加载 HTML
$("#content").load("fragment.html");
```

### 完整示例

```javascript
$.ajax({
    url: "https://api.example.com/users",
    type: "GET",
    dataType: "json",
    timeout: 5000,
    headers: {
        "Authorization": "Bearer token"
    },
    beforeSend: function(xhr) {
        // 请求发送前执行
        console.log("请求即将发送");
    },
    success: function(data) {
        // 请求成功
        $("#user-list").html("");
        data.forEach(function(user) {
            $("#user-list").append("<li>" + user.name + "</li>");
        });
    },
    error: function(xhr, status, error) {
        // 请求失败
        alert("加载失败: " + error);
    },
    complete: function(xhr, status) {
        // 请求完成（无论成功或失败）
        console.log("请求完成, 状态: " + status);
    }
});
```

## 工具函数

### 数组和对象操作

```javascript
// 遍历数组
$.each([1, 2, 3], function(index, value) {
    console.log(index + ": " + value);
});

// 遍历对象
$.each({ name: "John", age: 30 }, function(key, value) {
    console.log(key + ": " + value);
});

// 筛选数组
var filtered = $.grep([1, 2, 3, 4, 5], function(value) {
    return value > 3;
});

// 映射数组
var mapped = $.map([1, 2, 3], function(value) {
    return value * 2;
});

// 合并对象
var merged = $.extend({}, obj1, obj2);
```

### 字符串处理

```javascript
// 去除空格
$.trim("  hello  ");  // "hello"

// 截断字符串
$.escapeSelector(".myClass");  // 转义特殊字符
```

### 类型判断

```javascript
$.isArray([]);        // true
$.isFunction(fn);     // true
$.isEmptyObject({});  // true
$.isPlainObject({});  // true
```

## 链式调用

jQuery 支持链式调用，可以连续操作同一个元素：

```javascript
$("#myElement")
    .addClass("highlight")
    .css("color", "red")
    .fadeIn()
    .animate({ height: "100px" }, function() {
        console.log("完成");
    });
```

使用 `end()` 方法可以返回到上一个选择器：

```javascript
$("#myElement")
    .find(".child")
    .addClass("active")
    .end()    // 返回到 #myElement
    .removeClass("highlight");
```
