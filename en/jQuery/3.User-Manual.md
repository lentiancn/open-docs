# jQuery User Manual

## Quick Start

### Including jQuery

Include jQuery in the `<head>` or at the bottom of `<body>` in your HTML file:

```html
<!DOCTYPE html>
<html>
<head>
    <title>jQuery Demo</title>
    <script src="https://code.jquery.com/jquery-4.0.0.min.js"></script>
</head>
<body>
    <!-- Page content -->
</body>
</html>
```

### Waiting for DOM Ready

jQuery code must execute after the DOM is fully loaded:

```javascript
$(document).ready(function() {
    // Write jQuery code here
});

// Shorthand
$(function() {
    // Write jQuery code here
});
```

## Selectors

jQuery selectors are one of jQuery's core features, with syntax nearly identical to CSS selectors.

### Basic Selectors

```javascript
$("#myId")          // ID selector
$(".myClass")       // Class selector
$("div")            // Element selector
$("*")              // Universal selector
$("div, p, span")   // Multiple selectors
```

### Attribute Selectors

```javascript
$('[href]')                    // Elements with href attribute
$('[href="value"]')           // href equals specified value
$('[href!="value"]')          // href does not equal specified value
$('[href^="http"]')           // href starts with specified value
$('[href$=".pdf"]')           // href ends with specified value
$('[href*="google"]')         // href contains specified value
```

### Filter Selectors

```javascript
$("div:first")        // First div element
$("div:last")         // Last div element
$("div:even")         // Even-positioned div elements
$("div:odd")          // Odd-positioned div elements
$("div:eq(n)")        // Div at index n
$("div:gt(n)")        // Divs with index greater than n
$("div:lt(n)")        // Divs with index less than n
$("div:not(.exclude") // Divs without specified class
```

## DOM Manipulation

### Getting and Setting Content

```javascript
// Get/Set text content
var text = $("#myElement").text();
$("#myElement").text("New text content");

// Get/Set HTML content
var html = $("#myElement").html();
$("#myElement").html("<p>New HTML content</p>");

// Get/Set form value
var value = $("#inputField").val();
$("#inputField").val("New value");
```

### Getting and Setting Attributes

```javascript
// Get attribute
var src = $("#myImage").attr("src");

// Set single attribute
$("#myImage").attr("src", "image.jpg");

// Set multiple attributes
$("#myImage").attr({
    src: "image.jpg",
    alt: "Image description",
    title: "Image title"
});

// Remove attribute
$("#myImage").removeAttr("title");
```

### Getting and Setting CSS

```javascript
// Get single CSS property
var color = $("#myElement").css("color");

// Set single CSS property
$("#myElement").css("color", "red");

// Set multiple CSS properties
$("#myElement").css({
    color: "red",
    backgroundColor: "blue",
    fontSize: "16px"
});

// Add/Remove classes
$("#myElement").addClass("active highlight");
$("#myElement").removeClass("highlight");
$("#myElement").toggleClass("active");  // Toggle class
```

### DOM Traversal

```javascript
// Parent elements
$("#myElement").parent();
$("#myElement").parents();
$("#myElement").closest(".container");

// Child elements
$("#myElement").children();
$("#myElement").find(".child");

// Sibling elements
$("#myElement").siblings();
$("#myElement").next();
$("#myElement").prev();
$("#myElement").nextAll();
```

### Creating and Removing Elements

```javascript
// Create new element
var newDiv = $("<div>", {
    class: "new-element",
    text: "New element content"
});

// Add to page
$("#container").append(newDiv);      // Inside, at end
$("#container").prepend(newDiv);     // Inside, at beginning
$("#container").before(newDiv);      // Outside, before
$("#container").after(newDiv);       // Outside, after

// Remove elements
$("#myElement").remove();             // Remove element and children
$("#myElement").empty();              // Remove children only
```

## Event Handling

### Binding Events

```javascript
// Bind click event
$("#myButton").click(function() {
    alert("Button clicked");
});

// Bind using on method
$("#myButton").on("click", function() {
    alert("Button clicked");
});

// Bind multiple events
$("#myElement").on({
    mouseenter: function() {
        $(this).css("background", "yellow");
    },
    mouseleave: function() {
        $(this).css("background", "");
    }
});
```

### Event Delegation

```javascript
// Event binding for dynamic elements
$("#container").on("click", ".dynamic-item", function() {
    alert("Dynamic element clicked");
});
```

### Common Events

| Event | Description |
|-------|-------------|
| click | Click event |
| dblclick | Double click event |
| mouseenter | Mouse enters |
| mouseleave | Mouse leaves |
| mousemove | Mouse moves |
| keydown | Key pressed |
| keyup | Key released |
| focus | Gains focus |
| blur | Loses focus |
| submit | Form submitted |
| change | Value changed |
| load | Loaded |

## Animation Effects

### Basic Animations

```javascript
// Show element
$("#myElement").show();

// Hide element
$("#myElement").hide();

// Toggle visibility
$("#myElement").toggle();

// Fade in
$("#myElement").fadeIn();

// Fade out
$("#myElement").fadeOut();

// Toggle fade
$("#myElement").fadeToggle();

// Slide down
$("#myElement").slideDown();

// Slide up
$("#myElement").slideUp();

// Toggle slide
$("#myElement").slideToggle();
```

### Custom Animations

```javascript
$("#myElement").animate({
    opacity: 0.5,
    left: "+=50",
    height: "200px"
}, 300, function() {
    // Callback after animation
    console.log("Animation complete");
});
```

### Stopping Animations

```javascript
// Stop current animation
$("#myElement").stop();

// Stop all animations and reach final state
$("#myElement").stop(true, true);
```

## Ajax Requests

### Basic Usage

```javascript
$.ajax({
    url: "/api/data",
    method: "GET",
    data: { id: 1, name: "test" },
    success: function(response) {
        console.log("Success:", response);
    },
    error: function(xhr, status, error) {
        console.log("Error:", error);
    }
});
```

### Shortcut Methods

```javascript
// GET request
$.get("/api/data", { id: 1 }, function(response) {
    console.log(response);
});

// POST request
$.post("/api/submit", { name: "test" }, function(response) {
    console.log(response);
});

// Load HTML
$("#content").load("fragment.html");
```

### Complete Example

```javascript
$.ajax({
    url: "https://api.example.com/users",
    type: "GET",
    dataType: "json",
    timeout: 5000,
    headers: {
        "Authorization": "Bearer token"
    },
    beforeSend: function(xhr) {
        // Before request sends
        console.log("Request about to send");
    },
    success: function(data) {
        // Request successful
        $("#user-list").html("");
        data.forEach(function(user) {
            $("#user-list").append("<li>" + user.name + "</li>");
        });
    },
    error: function(xhr, status, error) {
        // Request failed
        alert("Load failed: " + error);
    },
    complete: function(xhr, status) {
        // Request complete (success or failure)
        console.log("Request complete, status: " + status);
    }
});
```

## Utility Functions

### Array and Object Operations

```javascript
// Iterate array
$.each([1, 2, 3], function(index, value) {
    console.log(index + ": " + value);
});

// Iterate object
$.each({ name: "John", age: 30 }, function(key, value) {
    console.log(key + ": " + value);
});

// Filter array
var filtered = $.grep([1, 2, 3, 4, 5], function(value) {
    return value > 3;
});

// Map array
var mapped = $.map([1, 2, 3], function(value) {
    return value * 2;
});

// Merge objects
var merged = $.extend({}, obj1, obj2);
```

### String Operations

```javascript
// Trim whitespace
$.trim("  hello  ");  // "hello"

// Escape special characters
$.escapeSelector(".myClass");  // "\.myClass"
```

### Type Checking

```javascript
$.isArray([]);        // true
$.isFunction(fn);     // true
$.isEmptyObject({});  // true
$.isPlainObject({});  // true
```

## Chaining

jQuery supports chaining, allowing you to continuously operate on the same element:

```javascript
$("#myElement")
    .addClass("highlight")
    .css("color", "red")
    .fadeIn()
    .animate({ height: "100px" }, function() {
        console.log("Done");
    });
```

Use the `end()` method to return to the previous selector:

```javascript
$("#myElement")
    .find(".child")
    .addClass("active")
    .end()    // Return to #myElement
    .removeClass("highlight");
```
