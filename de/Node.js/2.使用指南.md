# Node.js Benutzerhandbuch

Dieses Handbuch zeigt, wie Sie Node.js zum Erstellen von Backend-Anwendungen und Befehlszeilentools verwenden.

## Schnellstart

### Erste Anwendung erstellen

```javascript
// hello.js
console.log('Hello, Node.js!');

// Ausführen
node hello.js
```

### HTTP-Server erstellen

```javascript
// server.js
const http = require('http');

const server = http.createServer((req, res) => {
  res.writeHead(200, { 'Content-Type': 'text/plain' });
  res.end('Hello World!');
});

server.listen(3000, () => {
  console.log('Server läuft unter http://localhost:3000/');
});
```

Ausführen:

```bash
node server.js
```

## Kernmodule

### fs - Dateisystem

```javascript
const fs = require('fs');

// Datei lesen
fs.readFile('file.txt', 'utf8', (err, data) => {
  if (err) throw err;
  console.log(data);
});

// In Datei schreiben
fs.writeFile('file.txt', 'Hello!', (err) => {
  if (err) throw err;
  console.log('Datei gespeichert');
});

// Synchrones Lesen
const data = fs.readFileSync('file.txt', 'utf8');
```

### path - Pfade

```javascript
const path = require('path');

path.join('/foo', 'bar', 'baz'); // /foo/bar/baz
path.resolve(__dirname, 'file.txt'); // Absoluter Pfad
path.extname('file.txt'); // .txt
path.basename('/foo/bar.txt'); // bar.txt
```

### os - Betriebssystem

```javascript
const os = require('os');

os.platform(); // 'win32', 'linux', 'darwin'
os.arch(); // 'x64', 'arm64'
os.cpus(); // CPU-Informationen
os.freemem(); // Freier Speicher
os.totalmem(); // Gesamtspeicher
```

### events - Ereignisse

```javascript
const EventEmitter = require('events');

class MyEmitter extends EventEmitter {}

const myEmitter = new MyEmitter();

myEmitter.on('event', () => {
  console.log('Ereignis ausgelöst!');
});

myEmitter.emit('event');
```

### crypto - Kryptografie

```javascript
const crypto = require('crypto');

const hash = crypto.createHash('sha256');
hash.update('password');
const password = hash.digest('hex');
```

## Express-Framework

### Installation

```bash
npm install express
```

### Anwendung erstellen

```javascript
const express = require('express');
const app = express();

app.get('/', (req, res) => {
  res.send('Hello World!');
});

app.listen(3000, () => {
  console.log('Anwendung läuft unter http://localhost:3000');
});
```

### Routen

```javascript
// GET-Route
app.get('/api/users', (req, res) => {
  res.json({ users: [] });
});

// POST-Route
app.post('/api/users', (req, res) => {
  const user = req.body;
  res.json({ success: true, user });
});

// Parameter-Route
app.get('/api/users/:id', (req, res) => {
  const { id } = req.params;
  res.json({ id });
});
```

### Middleware

```javascript
// JSON parsen
app.use(express.json());

// URL-kodierte Daten parsen
app.use(express.urlencoded({ extended: true }));

// Logging-Middleware
app.use((req, res, next) => {
  console.log(`${req.method} ${req.url}`);
  next();
});

// Fehlerbehandlung
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).send('Something broke!');
});
```

## Asynchrone Programmierung

### Callback

```javascript
fs.readFile('file.txt', 'utf8', (err, data) => {
  if (err) throw err;
  console.log(data);
});
```

### Promise

```javascript
const fs = require('fs').promises;

async function read() {
  try {
    const data = await fs.readFile('file.txt', 'utf8');
    console.log(data);
  } catch (err) {
    console.error(err);
  }
}

read();
```

### Async/Await

```javascript
const axios = require('axios');

async function fetchData() {
  try {
    const response = await axios.get('https://api.example.com/data');
    console.log(response.data);
  } catch (error) {
    console.error(error);
  }
}

fetchData();
```

## Modulsystem

### Module exportieren

```javascript
// math.js
module.exports = {
  add: (a, b) => a + b,
  subtract: (a, b) => a - b,
};

// Oder ES6 verwenden
export const add = (a, b) => a + b;
export default { add };
```

### Module importieren

```javascript
// CommonJS
const math = require('./math');
console.log(math.add(2, 3));

// ES6
import { add } from './math';
console.log(add(2, 3));
```

## Best Practices

### 1. async/await verwenden

```javascript
// Vermeiden
getData((err, data) => {
  if (err) handleError(err);
  processData(data, (err, result) => {
    if (err) handleError(err);
    saveResult(result);
  });
});

// Empfohlen
async function main() {
  try {
    const data = await getData();
    const result = await processData(data);
    await saveResult(result);
  } catch (err) {
    handleError(err);
  }
}
```

### 2. Fehlerbehandlung

```javascript
// Asynchrone Fehler immer abfangen
async function fetchData() {
  try {
    const response = await fetch(url);
    return await response.json();
  } catch (error) {
    console.error('Datenabruf fehlgeschlagen:', error);
    throw error;
  }
}
```

### 3. Umgebungsvariablen verwenden

```javascript
require('dotenv').config();

const port = process.env.PORT || 3000;
const dbUrl = process.env.DATABASE_URL;
```
