# jQuery 使用手冊

## 快速開始

### 引入 jQuery

在 HTML 檔案的 `<head>` 或 `<body>` 底部引入 jQuery：

```html
<!DOCTYPE html>
<html>
<head>
    <title>jQuery Demo</title>
    <script src="https://code.jquery.com/jquery-4.0.0.min.js"></script>
</head>
<body>
    <!-- 頁面內容 -->
</body>
</html>
```

### 等待 DOM 載入完成

jQuery 程式碼需要在 DOM 完全載入後才能執行：

```javascript
$(document).ready(function() {
    // 在這裡編寫 jQuery 程式碼
});

// 簡寫形式
$(function() {
    // 在這裡編寫 jQuery 程式碼
});
```

## 選擇器

jQuery 選擇器是 jQuery 的核心功能之一，語法與 CSS 選擇器基本一致。

### 基本選擇器

```javascript
$("#myId")          // ID 選擇器
$(".myClass")       // 類別選擇器
$("div")            // 元素選擇器
$("*")              // 萬用字元選擇器
$("div, p, span")   // 多選擇器
```

### 屬性選擇器

```javascript
$('[href]')                    // 有 href 屬性的元素
$('[href="value"]')           // href 屬性等於指定值的元素
$('[href!="value"]')          // href 屬性不等於指定值的元素
$('[href^="http"]')           // href 屬性以指定值開頭的元素
$('[href$=".pdf"]')           // href 屬性以指定值結尾的元素
$('[href*="google"]')         // href 屬性包含指定值的元素
```

### 篩選選擇器

```javascript
$("div:first")        // 第一個 div 元素
$("div:last")         // 最後一個 div 元素
$("div:even")         // 偶數位置的 div 元素
$("div:odd")          // 奇數位置的 div 元素
$("div:eq(n)")        // 索引為 n 的 div 元素
$("div:gt(n)")        // 索引大於 n 的 div 元素
$("div:lt(n)")        // 索引小於 n 的 div 元素
$("div:not(.exclude") // 不含指定類別的 div 元素
```

## DOM 操作

### 取得和設定內容

```javascript
// 取得/設定文字內容
var text = $("#myElement").text();
$("#myElement").text("新的文字內容");

// 取得/設定 HTML 內容
var html = $("#myElement").html();
$("#myElement").html("<p>新的 HTML 內容</p>");

// 取得/設定表單值
var value = $("#inputField").val();
$("#inputField").val("新的值");
```

### 取得和設定屬性

```javascript
// 取得屬性
var src = $("#myImage").attr("src");

// 設定單一屬性
$("#myImage").attr("src", "image.jpg");

// 設定多個屬性
$("#myImage").attr({
    src: "image.jpg",
    alt: "圖片說明",
    title: "圖片標題"
});

// 移除屬性
$("#myImage").removeAttr("title");
```

### 取得和設定 CSS

```javascript
// 取得單一 CSS 屬性
var color = $("#myElement").css("color");

// 設定單一 CSS 屬性
$("#myElement").css("color", "red");

// 設定多個 CSS 屬性
$("#myElement").css({
    color: "red",
    backgroundColor: "blue",
    fontSize: "16px"
});

// 新增/移除類別
$("#myElement").addClass("active highlight");
$("#myElement").removeClass("highlight");
$("#myElement").toggleClass("active");  // 切換類別
```

### DOM 遍歷

```javascript
// 父元素
$("#myElement").parent();
$("#myElement").parents();
$("#myElement").closest(".container");

// 子元素
$("#myElement").children();
$("#myElement").find(".child");

// 兄弟元素
$("#myElement").siblings();
$("#myElement").next();
$("#myElement").prev();
$("#myElement").nextAll();
```

### 建立和刪除元素

```javascript
// 建立新元素
var newDiv = $("<div>", {
    class: "new-element",
    text: "新元素內容"
});

// 新增到頁面
$("#container").append(newDiv);      // 內部末尾
$("#container").prepend(newDiv);     // 內部開頭
$("#container").before(newDiv);      // 外部之前
$("#container").after(newDiv);       // 外部之後

// 刪除元素
$("#myElement").remove();             // 刪除元素及其子元素
$("#myElement").empty();              // 清空子元素
```

## 事件處理

### 綁定事件

```javascript
// 綁定點擊事件
$("#myButton").click(function() {
    alert("按鈕被點擊了");
});

// 使用 on 方法綁定
$("#myButton").on("click", function() {
    alert("按鈕被點擊了");
});

// 綁定多個事件
$("#myElement").on({
    mouseenter: function() {
        $(this).css("background", "yellow");
    },
    mouseleave: function() {
        $(this).css("background", "");
    }
});
```

### 事件委託

```javascript
// 動態元素的事件綁定
$("#container").on("click", ".dynamic-item", function() {
    alert("動態元素被點擊");
});
```

### 常用事件

| 事件 | 說明 |
|------|------|
| click | 點擊事件 |
| dblclick | 雙擊事件 |
| mouseenter | 滑鼠進入 |
| mouseleave | 滑鼠離開 |
| mousemove | 滑鼠移動 |
| keydown | 鍵盤按下 |
| keyup | 鍵盤釋放 |
| focus | 獲得焦點 |
| blur | 失去焦點 |
| submit | 表單提交 |
| change | 值改變 |
| load | 載入完成 |

## 動畫效果

### 基本動畫

```javascript
// 顯示元素
$("#myElement").show();

// 隱藏元素
$("#myElement").hide();

// 切換顯示狀態
$("#myElement").toggle();

// 淡入
$("#myElement").fadeIn();

// 淡出
$("#myElement").fadeOut();

// 淡入淡出切換
$("#myElement").fadeToggle();

// 滑動顯示
$("#myElement").slideDown();

// 滑動隱藏
$("#myElement").slideUp();

// 滑動切換
$("#myElement").slideToggle();
```

### 自訂動畫

```javascript
$("#myElement").animate({
    opacity: 0.5,
    left: "+=50",
    height: "200px"
}, 300, function() {
    // 動畫完成後的回調
    console.log("動畫完成");
});
```

### 停止動畫

```javascript
// 停止目前動畫
$("#myElement").stop();

// 停止所有動畫，並到達最終狀態
$("#myElement").stop(true, true);
```

## Ajax 請求

### 基本用法

```javascript
$.ajax({
    url: "/api/data",
    method: "GET",
    data: { id: 1, name: "test" },
    success: function(response) {
        console.log("成功:", response);
    },
    error: function(xhr, status, error) {
        console.log("錯誤:", error);
    }
});
```

### 快捷方法

```javascript
// GET 請求
$.get("/api/data", { id: 1 }, function(response) {
    console.log(response);
});

// POST 請求
$.post("/api/submit", { name: "test" }, function(response) {
    console.log(response);
});

// 載入 HTML
$("#content").load("fragment.html");
```

### 完整範例

```javascript
$.ajax({
    url: "https://api.example.com/users",
    type: "GET",
    dataType: "json",
    timeout: 5000,
    headers: {
        "Authorization": "Bearer token"
    },
    beforeSend: function(xhr) {
        // 請求發送前執行
        console.log("請求即將發送");
    },
    success: function(data) {
        // 請求成功
        $("#user-list").html("");
        data.forEach(function(user) {
            $("#user-list").append("<li>" + user.name + "</li>");
        });
    },
    error: function(xhr, status, error) {
        // 請求失敗
        alert("載入失敗: " + error);
    },
    complete: function(xhr, status) {
        // 請求完成（無論成功或失敗）
        console.log("請求完成, 狀態: " + status);
    }
});
```

## 工具函數

### 陣列和物件操作

```javascript
// 遍歷陣列
$.each([1, 2, 3], function(index, value) {
    console.log(index + ": " + value);
});

// 遍歷物件
$.each({ name: "John", age: 30 }, function(key, value) {
    console.log(key + ": " + value);
});

// 篩選陣列
var filtered = $.grep([1, 2, 3, 4, 5], function(value) {
    return value > 3;
});

// 映射陣列
var mapped = $.map([1, 2, 3], function(value) {
    return value * 2;
});

// 合併物件
var merged = $.extend({}, obj1, obj2);
```

### 字串處理

```javascript
// 去除空格
$.trim("  hello  ");  // "hello"

// 轉義特殊字元
$.escapeSelector(".myClass");  // 轉義特殊字元
```

### 型態判斷

```javascript
$.isArray([]);        // true
$.isFunction(fn);     // true
$.isEmptyObject({});  // true
$.isPlainObject({});  // true
```

## 鏈式呼叫

jQuery 支援鏈式呼叫，可以連續操作同一個元素：

```javascript
$("#myElement")
    .addClass("highlight")
    .css("color", "red")
    .fadeIn()
    .animate({ height: "100px" }, function() {
        console.log("完成");
    });
```

使用 `end()` 方法可以返回到上一個選擇器：

```javascript
$("#myElement")
    .find(".child")
    .addClass("active")
    .end()    // 返回到 #myElement
    .removeClass("highlight");
```
