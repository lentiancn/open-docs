# Element UI User Manual

> Component usage guide

---

## Basic Components

### Buttons

```html
<el-button>Default</el-button>
<el-button type="primary">Primary</el-button>
<el-button type="success">Success</el-button>
<el-button type="warning">Warning</el-button>
<el-button type="danger">Danger</el-button>
```

### Input

```html
<el-input v-model="input" placeholder="Enter content"></el-input>
```

---

## Form Components

### Form Validation

```html
<el-form :model="form" :rules="rules" ref="ruleForm">
  <el-form-item prop="username">
    <el-input v-model="form.username"></el-input>
  </el-form-item>
  <el-form-item>
    <el-button @click="submitForm">Submit</el-button>
  </el-form-item>
</el-form>
```

```javascript
data() {
  return {
    form: { username: '' },
    rules: {
      username: [
        { required: true, message: 'Required', trigger: 'blur' }
      ]
    }
  }
}
```

### Select

```html
<el-select v-model="selected" placeholder="Select">
  <el-option
    v-for="item in options"
    :key="item.value"
    :label="item.label"
    :value="item.value">
  </el-option>
</el-select>
```

## Table Components

### Basic Table

```html
<el-table :data="tableData" stripe border>
  <el-table-column prop="name" label="Name"></el-table-column>
  <el-table-column prop="address" label="Address"></el-table-column>
</el-table>
```

### Table with Actions

```html
<el-table-column label="Actions">
  <template slot-scope="scope">
    <el-button size="mini" @click="handleEdit(scope.row)">Edit</el-button>
    <el-button size="mini" type="danger" @click="handleDelete(scope.row)">Delete</el-button>
  </template>
</el-table-column>
```

## Dialog Components

### Dialog

```html
<el-button @click="dialogVisible = true">Open</el-button>

<el-dialog title="Title" :visible.sync="dialogVisible" width="30%">
  <span>Content</span>
  <span slot="footer">
    <el-button @click="dialogVisible = false">Cancel</el-button>
    <el-button type="primary" @click="dialogVisible = false">OK</el-button>
  </span>
</el-dialog>
```

## Navigation

### Menu

```html
<el-menu mode="horizontal">
  <el-menu-item index="1">Home</el-menu-item>
  <el-submenu index="2">
    <template slot="title">Profile</template>
    <el-menu-item index="2-1">Settings</el-menu-item>
  </el-submenu>
</el-menu>
```

### Tabs

```html
<el-tabs v-model="activeName">
  <el-tab-pane label="User" name="first">User</el-tab-pane>
  <el-tab-pane label="Config" name="second">Config</el-tab-pane>
</el-tabs>
```
