## フォームコンポーネント

### Input

基本的な使用：

```html
<el-input v-model="input" placeholder="入力してください"></el-input>
```

アイコン付き：

```html
<el-input
  placeholder="入力"
  v-model="input"
  prefix-icon="el-icon-search">
</el-input>
```

テキストエリア：

```html
<el-input
  type="textarea"
  :rows="3"
  placeholder="入力"
  v-model="textarea">
</el-input>
```

### Select

基本的な使用：

```html
<el-select v-model="value" placeholder="選択">
  <el-option
    v-for="item in options"
    :key="item.value"
    :label="item.label"
    :value="item.value">
  </el-option>
</el-select>
```

フィルタ可能：

```html
<el-select
  v-model="value"
  filterable
  placeholder="選択">
</el-select>
```

複数選択：

```html
<el-select
  v-model="value"
  multiple
  placeholder="選択">
</el-select>
```

### Form（バリデーション付き）

```html
<el-form :model="ruleForm" :rules="rules" ref="ruleForm" label-width="100px">
  <el-form-item label="アクティビティ名" prop="name">
    <el-input v-model="ruleForm.name"></el-input>
  </el-form-item>
  <el-form-item label="アクティビティ地域" prop="region">
    <el-select v-model="ruleForm.region" placeholder="選択">
      <el-option label="地域1" value="shanghai"></el-option>
      <el-option label="地域2" value="beijing"></el-option>
    </el-select>
  </el-form-item>
  <el-form-item label="日付" prop="date1">
    <el-date-picker
      v-model="ruleForm.date1"
      type="date"
      placeholder="日付を選択">
    </el-date-picker>
  </el-form-item>
  <el-form-item>
    <el-button type="primary" @click="submitForm('ruleForm')">送信</el-button>
    <el-button @click="resetForm('ruleForm')">リセット</el-button>
  </el-form-item>
</el-form>
```

バリデーションルール：

```javascript
export default {
  data() {
    return {
      ruleForm: {
        name: '',
        region: '',
        date1: ''
      },
      rules: {
        name: [
          { required: true, message: '名前を入力してください', trigger: 'blur' },
          { min: 3, max: 5, message: '3〜5文字', trigger: 'blur' }
        ],
        region: [
          { required: true, message: '地域を選択してください', trigger: 'change' }
        ],
        date1: [
          { required: true, message: '日付を選択してください', trigger: 'change' }
        ]
      }
    }
  },
  methods: {
    submitForm(formName) {
      this.$refs[formName].validate((valid) => {
        if (valid) {
          alert('submit!')
        } else {
          return false
        }
      })
    },
    resetForm(formName) {
      this.$refs[formName].resetFields()
    }
  }
}
```

### フォームバリデーション

```javascript
// カスタムバリデーター
const validatePass = (rule, value, callback) => {
  if (value === '') {
    callback(new Error('パスワードを入力'))
  } else {
    if (this.ruleForm.checkPass !== '') {
      this.$refs.ruleForm.validateField('checkPass')
    }
    callback()
  }
}
```
