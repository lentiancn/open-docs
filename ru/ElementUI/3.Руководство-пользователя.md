### Form с валидацией

```html
<el-form :model="ruleForm" :rules="rules" ref="ruleForm" label-width="100px">
  <el-form-item label="Название" prop="name">
    <el-input v-model="ruleForm.name"></el-input>
  </el-form-item>
  <el-form-item label="Регион" prop="region">
    <el-select v-model="ruleForm.region" placeholder="Выберите">
      <el-option label="Регион 1" value="shanghai"></el-option>
      <el-option label="Регион 2" value="beijing"></el-option>
    </el-select>
  </el-form-item>
  <el-form-item label="Дата" prop="date1">
    <el-date-picker
      v-model="ruleForm.date1"
      type="date"
      placeholder="Выберите дату">
    </el-date-picker>
  </el-form-item>
  <el-form-item>
    <el-button type="primary" @click="submitForm('ruleForm')">Отправить</el-button>
    <el-button @click="resetForm('ruleForm')">Сбросить</el-button>
  </el-form-item>
</el-form>
```

Правила валидации:

```javascript
export default {
  data() {
    return {
      ruleForm: {
        name: '',
        region: '',
        date1: ''
      },
      rules: {
        name: [
          { required: true, message: 'Введите название', trigger: 'blur' },
          { min: 3, max: 5, message: 'Длина от 3 до 5', trigger: 'blur' }
        ],
        region: [
          { required: true, message: 'Выберите регион', trigger: 'change' }
        ],
        date1: [
          { required: true, message: 'Выберите дату', trigger: 'change' }
        ]
      }
    }
  },
  methods: {
    submitForm(formName) {
      this.$refs[formName].validate((valid) => {
        if (valid) {
          alert('submit!')
        } else {
          return false
        }
      })
    },
    resetForm(formName) {
      this.$refs[formName].resetFields()
    }
  }
}
```

### Валидация формы

```javascript
const validatePass = (rule, value, callback) => {
  if (value === '') {
    callback(new Error('Введите пароль'))
  } else {
    if (this.ruleForm.checkPass !== '') {
      this.$refs.ruleForm.validateField('checkPass')
    }
    callback()
  }
}
```
