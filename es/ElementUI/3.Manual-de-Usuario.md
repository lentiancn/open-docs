### Form con validación

```html
<el-form :model="ruleForm" :rules="rules" ref="ruleForm" label-width="100px">
  <el-form-item label="Nombre de actividad" prop="name">
    <el-input v-model="ruleForm.name"></el-input>
  </el-form-item>
  <el-form-item label="Región" prop="region">
    <el-select v-model="ruleForm.region" placeholder="Por favor seleccione">
      <el-option label="Región 1" value="shanghai"></el-option>
      <el-option label="Región 2" value="beijing"></el-option>
    </el-select>
  </el-form-item>
  <el-form-item label="Fecha" prop="date1">
    <el-date-picker
      v-model="ruleForm.date1"
      type="date"
      placeholder="Seleccione fecha">
    </el-date-picker>
  </el-form-item>
  <el-form-item>
    <el-button type="primary" @click="submitForm('ruleForm')">Enviar</el-button>
    <el-button @click="resetForm('ruleForm')">Restablecer</el-button>
  </el-form-item>
</el-form>
```

Reglas de validación:

```javascript
export default {
  data() {
    return {
      ruleForm: {
        name: '',
        region: '',
        date1: ''
      },
      rules: {
        name: [
          { required: true, message: 'Por favor ingrese nombre', trigger: 'blur' },
          { min: 3, max: 5, message: 'Longitud entre 3 y 5', trigger: 'blur' }
        ],
        region: [
          { required: true, message: 'Por favor seleccione región', trigger: 'change' }
        ],
        date1: [
          { required: true, message: 'Por favor seleccione fecha', trigger: 'change' }
        ]
      }
    }
  },
  methods: {
    submitForm(formName) {
      this.$refs[formName].validate((valid) => {
        if (valid) {
          alert('enviar!')
        } else {
          return false
        }
      })
    },
    resetForm(formName) {
      this.$refs[formName].resetFields()
    }
  }
}
```

### Validación de formulario

```javascript
// Validador personalizado
const validatePass = (rule, value, callback) => {
  if (value === '') {
    callback(new Error('Por favor ingrese contraseña'))
  } else {
    if (this.ruleForm.checkPass !== '') {
      this.$refs.ruleForm.validateField('checkPass')
    }
    callback()
  }
}
```
